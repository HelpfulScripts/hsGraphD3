   5 *

   7 * The setting are initialized during construction and can be modified as needed. 

   8 * `graph` provides two paths to access and modify setting:

   9 * - `graph.defaults..` provides a central access to collection of 

  10 * all component settings via the `graph's` default getter. The table below lists all settings

  11 * available under `graph.defaults`.

  12 * - Each `GraphComponent` provides access to the component's default settings via `.defaults.`.

  13 * These are also accessible via the `graph's` `defaults` getter. 

  14 * 

  15 * For example, the `canvas` settings can be accessed via `graph.canvas.defaults` and 

  16 * `grpah.defaults.canvas`;

 180export const defaultStroke = (width:UnitVp, color:Color='currentColor'):Stroke => {

 181    return {

 268export function setStroke(svg:d3Base, settings:Stroke):d3Base {

 269    return (!settings || !settings.width)? svg : svg 

 270    .attr('stroke',         settings.color)

 271    .attr('stroke-width',   settings.width)

 272    .attr('stroke-opacity', settings.opacity)

 273    .attr('stroke-dasharray', settings.dashed||'none')

 274    .attr('fill-opacity',   0);

 275}

 276

 277export function setFill(svg:d3Base, settings:Fill):d3Base {

 278    return !settings? svg : svg 

 279    .attr('fill',         settings.color)

 280    .attr('fill-opacity', settings.opacity);

 281}

 283export function setArea(svg:d3Base, settings:Area):d3Base {

 284    return setStroke(svg, settings.border)

 285    .attr('fill',         settings.color)

 286    .attr('fill-opacity', settings.opacity);

 287}

 312export function setRect(svg:d3Base, settings:RectStyle):d3Base {

 313    svg .attr('rx', settings.rx)
