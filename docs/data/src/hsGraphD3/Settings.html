   5 *

   7 * The setting are initialized during construction and can be modified as needed. 

   8 * `graph` provides two paths to access and modify setting:

   9 * - `graph.defaults..` provides a central access to collection of 

  10 * all component settings via the `graph's` default getter. The table below lists all settings

  11 * available under `graph.defaults`.

  12 * - Each `GraphComponent` provides access to the component's default settings via `.defaults.`.

  13 * These are also accessible via the `graph's` `defaults` getter. 

  14 * 

  15 * For example, the `canvas` settings can be accessed via `graph.canvas.defaults` and 

  16 * `grpah.defaults.canvas`;

 134export const defaultStroke = (width:UnitVp, color:Color='currentColor'):Stroke => {

 135    return {

 136        width: width,

 137        color: color,

 138        opacity: 1,

 139        dashed: undefined   // or '5,10,5', alternating filled and blank

 140    };

 141};

 142

 143export const defaultLine = (width:UnitVp, color:Color='currentColor'):Line => {

 144    const def:any = defaultStroke(width, color);

 145    def.rendered = true;

 146    return def;

 147};

 149export const defaultFill = (areaFill:Color = 'currentColor', opacity=0.5):Fill => {

 156export const defaultFont = (size=16):Font => {

 165/**

 166 * convenience function to create a default `RectStyle` object with configurable fill color and border. 

 167 * @param area  the fill color

 168 * @param borderWidth the border width in pixel

 169 * @param borderColor the border color

 170 */

 171export const defaultRectStyle = (areaFill:Color, borderWidth:UnitVp=0, borderColor:Color='currentColor'):RectStyle => {

 172    return {

 173        rx: 0,

 174        ry: 0,

 175        fill:   defaultFill(areaFill, 1),

 176        stroke: defaultStroke(borderWidth, borderColor)

 177    };

 178};

 205export function setStroke(svg:d3Base, settings:Stroke):d3Base {

 206    return svg 

 207    .attr('stroke',         settings.color)

 208    .attr('stroke-width',   settings.width)

 209    .attr('stroke-opacity', settings.opacity)

 210    .attr('fill-opacity',   0);

 211}

 212

 213export function setFill(svg:d3Base, settings:Fill):d3Base {

 214    return svg 

 215    .attr('fill',         settings.color)

 216    .attr('fill-opacity', settings.opacity);

 217}

 218

 219export function setRect(svg:d3Base, settings:RectStyle):d3Base {

 220    svg .attr('rx', settings.rx)

 221        .attr('ry', settings.ry);

 222    setStroke(svg, settings.stroke);

 223    setFill(svg, settings.fill); 

 224    return svg;

 225}

 227export function setText(svg:d3Base, settings:TextStyle, transition:any) {

 228    svg.transition(transition)

 229        .attr('fill', settings.color)

 230        .attr('font-family', settings.font.family)

 231        .attr('font-size',   settings.font.size+'px')

 232        .attr('font-style',  settings.font.style)

 233        .attr('font-weight', settings.font.weight);

 234}
