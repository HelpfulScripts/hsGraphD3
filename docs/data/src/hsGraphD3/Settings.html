 177/** defines `Label` configurations: */

 178export interface Label extends TextStyle, Rendered {

 179    /**

 180     * indicates the horizontal alignemnt of the label. 

 181     * Either a {@link Settings.HPos HPos} or a number or percentage indicating 

 182     * the horizontal position with `0=0%=left` and `1=100%=right`. 

 183     * Defaults to `HPos.center`.

 184     */

 185    xpos: HPos | Share | UnitPercent;

 186    /**

 187     * indicates the vertical alignemnt of the label. 

 188     * Either a {@link Settings.VPos VPos} or a number or percentage indicating 

 189     * the vertical position with `0=0%=top` and `1=100%=bottom`. 

 190     * Defaults to `VPos.center`.

 191     */

 192    ypos: VPos | Share | UnitPercent;

 193    /**

 194     * offsets the label horizontally be the specified `em` value.

 195     * positive numbers shift right. Defaults to 0.

 196     */

 197    hOffset: UnitEm;    // offset in `em`

 198    /**

 199     * offsets the label vertically be the specified `em` value

 200     * positive numbers shift down. Defaults to 0.

 201     */

 202    vOffset: UnitEm;    // offset in `em`

 203    /**

 204     * if `true`, renders labels inside the marker. 

 205     * If `false`, renders labels outside the marker when `xpos` or `ypos`

 206     * are not `center`. Defaults to `true`.

 207     */

 208    inside: boolean;    // render label inside the area

 209}

 223function pct2Share(v:UnitPercent | Share | HPos | VPos):Share {

 224    if (!isNaN(v)) { 

 225        return v; 

 226    } else if ((v).endsWith('%')) {

 227        return parseInt(v)/100;

 228    } else {

 229        switch(v) {

 230            case TextHAlign.left:

 231            case TextVAlign.top:    return 0;

 232            case TextHAlign.right:

 233            case TextVAlign.bottom: return 1;

 234            case TextHAlign.center:

 235            case TextVAlign.center: return 0.5;

 236            default: log.warn(`value ${v} is neither numeric, an alignment, nor a percentage`);

 237                return 0;

 238        }

 239    }

 240}

 242export interface TextPos {

 243    x:{pos: number, shift:number, anchor:string};

 244    y:{pos: number, shift:number, baseline:string};

 245}

 246

 247export function textPos(x:UnitPercent|Share|HPos, y:UnitPercent|Share|VPos, inside=true):TextPos {

 248    let xpos = pct2Share(x);

 249    let ypos = pct2Share(y);

 250    const anchor   = (xpos<0.33)? (inside?'start':'end')        : ((xpos>0.66)? (inside?'end':'start') : 'middle');

 251    const baseline = (ypos<0.33)? (inside?'hanging':'baseline') : ((ypos>0.66)? (inside?'baseline':'hanging') : 'middle');

 252    let xShift     = (xpos<0.33)? (inside?1:-1) : ((xpos>0.66)? (inside?-1:1) : 0);

 253    let yShift     = (ypos<0.33)? (inside?1:-1) : ((ypos>0.66)? (inside?-1:1) : 0);

 254    return {x:{pos:xpos, shift:xShift, anchor:anchor}, y:{pos:ypos, shift:yShift, baseline:baseline}};

 255}

 353export function setStroke(svg:d3Base, settings:Stroke):d3Base {

 354    return (!settings || !settings.width)? svg : svg 

 355    .attr('stroke',         settings.color)

 356    .attr('stroke-width',   settings.width)

 357    .attr('stroke-opacity', settings.opacity)

 358    .attr('stroke-dasharray', settings.dashed||'none')
