 111export const defaultStroke = (width:UnitVp, color:Color='currentColor'):Stroke => {

 112    return {

 113        width: width,

 114        color: color,

 115        opacity: 1,

 116        dashed: undefined   // or '5,10,5', alternating filled and blank

 117    };

 118};

 120export const defaultFill = (areaFill:Color = '#fff') => {

 121    return {

 122        color: areaFill,

 123        opacity: 1

 124    };

 125};

 127/**

 128 * convenience function to create a default `RectStyle` object with configurable fill color and border. 

 129 * @param area  the fill color

 130 * @param borderWidth the border width in pixel

 131 * @param borderColor the border color

 132 */

 133export const defaultRect = (areaFill:Color, borderWidth:UnitVp=0, borderColor:Color='currentColor'):RectStyle => {

 134    return {

 135        rx: 0,

 136        ry: 0,

 137        fill:   defaultFill(areaFill),

 138        stroke: defaultStroke(borderWidth, borderColor)

 139    };

 140};

 168export function setStroke(svg:d3Base, settings:Stroke):d3Base {

 169    return svg 

 170    .attr('stroke',         settings.color)

 171    .attr('stroke-width',   settings.width)

 172    .attr('stroke-opacity', settings.opacity)

 173    .attr('fill-opacity',   0);

 174}

 175

 176export function setFill(svg:d3Base, settings:Fill):d3Base {

 177    return svg 

 178    .attr('fill',         settings.color)

 179    .attr('fill-opacity', settings.opacity);

 180}

 181

 182export function setRect(svg:d3Base, settings:RectStyle):d3Base {

 183    svg .attr('rx', settings.rx)

 184        .attr('ry', settings.ry);

 185    setStroke(svg, settings.stroke);

 186    setFill(svg, settings.fill); 

 187    return svg;

 188}

 190export function setText(svg:d3Base, settings:TextStyle, transition:any) {

 191    svg.transition(transition)

 192        .attr('color', settings.color)

 193        .attr('font-family', settings.font.family)

 194        .attr('font-size',   settings.font.size+'px')

 195        .attr('font-style',  settings.font.style)

 196        .attr('font-weight', settings.font.weight);

 197}
