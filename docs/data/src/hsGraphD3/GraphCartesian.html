  54import { log as gLog }      from 'hsutil';   const log = gLog('GraphCartesian');

  55

  56import { Data }             from 'hsdatab';

  57import { AbstractGraph }    from './AbstractGraph';

  58import { defaultDimScale }  from './Scale';

  59import { ScaleDefaults }    from './Scale';

  60import { Scales }           from './Scale';

  61

  62export class GraphCartesian extends AbstractGraph {

  63    private cumulativeDomains: {[colName:string]: [number, number]} = {};

  64

  65    constructor(root:any) { 

  66        super(root);

  67        const scales = this.config.defaults.scales;

  68        const margins = scales.margin;

  69        this.config.scales.hor = undefined;

  70        this.config.scales.ver = undefined;

  71        scales.dims['hor'] = defaultDimScale(margins.left, this.viewport.width-margins.right);

  72        scales.dims['ver'] = defaultDimScale(margins.top, this.viewport.height-margins.bottom);

  73        log.info('creating Graph');

  74    }

  75

  76    /**

  77     * adds a series to the plot.

  78     * @param type type of plot to use, e.g. 'bubble' or 'scatter'

  79     * @param params the column name of the parameters used to plot the series

  80     */

  81    public addSeries(type:string, x:string, y:string, ...params:string[]) {

  82        super.addSeries(type, x, y, ...params);

  83        const scales = this.config.defaults.scales.dims;

  84        // the first two dimensions are shared with the standard 2D dimensions 'hor' and 'ver'.

  85        scales[params[0]] = scales[params[0]] || scales['hor'] || defaultDimScale(0, 1);

  86        scales[params[1]] = scales[params[1]] || scales['ver'] || defaultDimScale(0, 1);

  87    }

  88

  89    protected setScales(data:Data) {

  90        const aggregateOverTime = this.config.defaults.scales.aggregateOverTime;

  91        const scalesDefaults = this.config.defaults.scales.dims;

  92        const scales = this.config.scales;

  93        if (!aggregateOverTime) {

  94            this.cumulativeDomains = {};

  95        } 

  96        this.cumulativeDomains[0] = this.cumulativeDomains[0] || [1e99, -1e99];

  97        this.cumulativeDomains[1] = this.cumulativeDomains[1] || [1e99, -1e99];

  98        const domain = this.series.expandDomain(data, this.cumulativeDomains);

  99        scales.hor = Scales.createScale(scalesDefaults.hor, this.cumulativeDomains[0], [0, this.viewport.width]);

 100        scales.ver = Scales.createScale(scalesDefaults.ver, this.cumulativeDomains[1], [0, this.viewport.height]);

 101    }
