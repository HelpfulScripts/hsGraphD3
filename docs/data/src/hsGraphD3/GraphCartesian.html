   6 * ### Example:

  22 * // setup and plot the data:

  23 * const graph = new hsGraphD3.GraphCartesian(root);

  25 * graph.addSeries('line', 'time', 'costs');

  26 * 

  27 * with (graph.defaults.canvas) {

  28 *      fill.color = '#fcfcfc';

  30 * }

  31 * 

  32 * // series defaults can be indexed by position or by name. Names are created as `series`+position index.

  33 * graph.defaults.series[0].marker.size = 15;

  73export class GraphCartesian extends Graph {

  74    constructor(root:any) { 

  75        super(root);

  76        log.debug('created GraphCartesian');

  77    }

  79    /**

  80     * adds a series to the plot.

  81     * At a minimum, an x- and y coordinate column name needs to be specified to use on the data.

  82     * Additional column names will interpreted as follows:

  83     * - index 0: x coordinate

  84     * - index 1: y coordinate

  85     * - index 2: size of marker (replaces default setting)

  86     * 

  87     * Accordingly, scales will be defined for each of the coordinate indexes.

  88     * 

  89     * @param type type of plot to use, e.g. 'bubble' or 'scatter'

  90     * @param params the column name of the parameters used to plot the series

  91     */

  92    public addSeries(type:string, x:string, y:string, ...params:string[]) {

  93        super.addSeries(type, x, y, ...params);

  94        const scales = this.config.defaults.scales;

  95        // the first two dimensions are shared with the standard 2D dimensions 'hor' and 'ver'.

  96        scales.dims[0] = scales.dims[0] || scales.dims['hor'] || scaleDefault();       // auto viewport range

  97        scales.dims[1] = scales.dims[1] || scales.dims['ver'] || scaleDefault();       // auto viewport range

  98        if (params.length > 0) { 

  99            scales.dims[2] = scales.dims[2] || scales.dims['size'] || scaleDefault();  // auto viewport range

 100        }

 101    }

 102

 103    protected makeDefaults() {

 104        super.makeDefaults();

 105        const scales = this.config.defaults.scales.dims;

 106        scales.hor  = scaleDefault();

 107        scales.ver  = scaleDefault();

 108        scales.size = scaleDefault(5, 15); // marker sizes in vpUnits

 109    }

 110

 111    protected setScales(data:DataSet) {

 112        const scalesDefaults = this.config.defaults.scales;

 113        const margins = this.config.defaults.scales.margin;

 114        const scales = this.config.scales;

 115        scales.hor  = Scales.createScale(scalesDefaults.dims.hor, this.cumulativeDomains[0], [margins.left,  this.viewport.width-margins.right]);

 116        scales.ver  = Scales.createScale(scalesDefaults.dims.ver, this.cumulativeDomains[1], [this.viewport.height-margins.bottom, margins.top]);

 117        scales.size = Scales.createScale(scalesDefaults.dims.size, this.cumulativeDomains[2]);

 118    }

 119

 120    protected prepareDomains(scalesDefaults:ScalesDefaults):Domains {

 121        if (!this.cumulativeDomains[0] || !scalesDefaults.dims.hor.aggregateOverTime)  { 

 122            this.cumulativeDomains[0] = this.cumulativeDomains.hor = [1e99, -1e99]; 

 123        }

 124        if (!this.cumulativeDomains[1] || !scalesDefaults.dims.ver.aggregateOverTime)  { 

 125            this.cumulativeDomains[1] = this.cumulativeDomains.ver = [1e99, -1e99]; 

 126        }

 127        if (!this.cumulativeDomains[2] || !scalesDefaults.dims.size.aggregateOverTime) { 

 128            this.cumulativeDomains[2] = this.cumulativeDomains.size = [1e99, -1e99]; 

 129        }

 130        return this.cumulativeDomains;

 131    }

 135    initialize(svg:d3Base): void {

 136        super.initialize(svg);

 137        this.config.scales.hor = undefined;

 138        this.config.scales.ver = undefined;

 139    }
