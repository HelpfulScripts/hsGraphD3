  15     * Returns an accessor function to access the numeric value in a data row. 

  16     * @param dim the semantic dimension ('hor', 'ver', 'size') in which to aggregate

  17     * @param v data column value definition

  18     * @param colNames 

  19     */

  20    accessor(v:ValueDef, colNames:string[], useStack=true):AccessFn {

  21        if (useStack && this.dims.stacked) {

  22            // stackDim = is 'v' a stackable dimension?

  23            const stackDim = this.getStackDim(v);

  24            const abscissaCol = this.getAbscissaCol();

  25            if (stackDim && typeof abscissaCol === 'string') {

  26                const stackIndex = colNames.indexOf(this.dims.stacked);

  27                const fn = super.accessor(v, colNames);

  28                return (row, rowIndex) => row[stackIndex] + fn(row, rowIndex);

  29            }

  30        }

  31        return super.accessor(v, colNames);

  32    }

  40    public expandDomains(dataSet:DataSet, domains:Domains) {

  41        this.updateStack(dataSet);

  42        const dims:GraphDimensions = this.dimensions;

  43        Object.keys(dims).map(dim => { // dim='hor', 'ver', size'

  44            const useStack = dim!=='size';  // donst stack-scale marker sizes

  45

  46            const type = this.cfg.graph.defaults.scales.dims[dim].type;

  47            dims[dim].map(colName => { if (colName!==undefined) { 

  48                const valueFn = this.accessor(colName, dataSet.colNames, useStack);

  49                switch(type) {

  50                    case 'ordinal':     

  51                        domains[dim] = this.expandOrdinalDomain(dataSet, domains[dim] || [], valueFn); 

  52                        break;

  53                    default:            

  54                        domains[dim] = this.expandNumDomain(dataSet, domains[dim] || [1e99, -1e99], valueFn);

  55                }

  56            }});

  57        });

  58    }

  60    protected expandNumDomain(dataSet:DataSet, domain:NumDomain, fn:(row?:DataRow, i?:number) => DataVal):NumDomain {

  61        return dataSet.rows.reduce((dom:NumDomain, row:DataRow, i:number):NumDomain => {

  62            const val = fn(row, i);

  63            dom[0] = Math.min(val, dom[0]);

  64            dom[1] = Math.max(val, dom[1]);

  65            return dom;

  66        }, domain);

  67    }

  68    

  69    protected expandOrdinalDomain(dataSet:DataSet, domain:OrdDomain, fn:(row?:DataRow, i?:number) => DataVal):OrdDomain {

  70        return dataSet.rows.reduce((dom:OrdDomain, row:DataRow, i:number):OrdDomain => {

  71            const val = fn(row, i);

  72            if (dom.indexOf(val) < 0) { dom.push(val); }

  73            return dom;

  74        }, domain);

  75    }

 119    /** clears the stack for this cycle before any series rendering happens. */

 120    public clearStack(data:DataSet) {

 121        const group = this.dims.stacked;

 122        if (group) {

 123            if (data.colNames.indexOf(group) < 0) { 

 124                data.colNames.push(group); 

 125            }

 126            const stackCol = data.colNames.indexOf(group);

 127            data.rows.forEach(row => row[stackCol] = 0);

 128            this.cfg.stack[this.dims.stacked] = {};

 129        }

 130    }

 131    

 132    /** update stack after rendering series. */

 133    protected updateStack(data:DataSet) {

 134        const group = this.dims.stacked;

 135        if (group) {

 136            const stack = this.cfg.stack[group];

 137            const stackCol = data.colNames.indexOf(group);

 138            const abscissaCol = this.getAbscissaCol();

 139            const abscissaIndex = data.colNames.indexOf(abscissaCol);

 140            const ordinateCol = this.getOrdinateCol();

 141            const ordinateIndex = data.colNames.indexOf(ordinateCol);

 142            data.rows.forEach(row => {

 143                const abscissaKey = ''+row[abscissaIndex];

 144                row[stackCol] = stack[''+abscissaKey] || 0;

 145                stack[''+abscissaKey] = (stack[''+abscissaKey]||0) + row[ordinateIndex];

 146            });

 147        }
