  22export interface CartSeriesDimensions extends SeriesDimensions {

  23    x:   NumberSet;     // name of x-axis data column, or a constant number

  24    y:   NumberSet;     // name of y-axis data column, or a constant number

  25    y0?: NumberSet;     // optional, name of y-axis data column for lower fill border, or a constant number

  26    r?:  NumberSet;     // optional, name of marker size data column, or a constant number

  27}

  28

  29export abstract class CartSeriesPlot extends SeriesPlot {

  30    /** 

  31     * a list of data column names used,

  32     * reflecting the list of column names provided during construction.

  33     */

  34    protected dims: CartSeriesDimensions;

  35

  36    constructor(cfg:GraphCfg, seriesName:string, dims:CartSeriesDimensions) {

  37        super(cfg, seriesName);

  38        this.dims = dims;

  39    }

  40

  41    get dimensions():CartDimensions { 

  42        return {

  43            hor: [this.dims.x],

  44            ver: [this.dims.y, this.dims.y0],

  45            size:[this.dims.r]

  46        };

  47    }

  48

  49

  50    //---------- lifecylce methods --------------------

  51

  52    initialize(svg:d3Base, color?:string): void {

  53        this.svg = svg.append('g').classed(this.seriesKey, true);

  54        if (color) { this.svg.style('color', color); }

  55        this.svg.append('g').classed('area', true).append('path');

  56        this.svg.append('g').classed('line', true).append('path');

  57        this.svg.append('g').classed('markers', true);

  58

  59        const defaults = this.cfg.defaults.series[this.key];

  60        if (this.dims.r) { defaults.marker.rendered = true; }

  61

  62        const line = this.svg.select('.line');

  63        setStroke(line, defaults.line);

  64

  65        const area = this.svg.select('.area');

  66        setFill(area, defaults.area);

  67

  68        const markers = this.svg.select('.markers');

  69        setStroke(markers, defaults.marker.stroke);

  70        setFill(markers, defaults.marker.fill);

  71    }

  72

  73    preRender(data:DataSet, domains:Domains): void {

  74    }

  75

  76    /** renders the component for the given data */

  77    renderComponent(data:DataSet): void {

  78        const defaults = (this.cfg.defaults.series[this.key]);

  79        if (defaults.marker.rendered) { this.svg.call(this.d3RenderMarkers.bind(this), data); }

  80        if (defaults.line.rendered)   { this.svg.call(this.d3RenderPath.bind(this), data); }

  81        if (defaults.area.rendered)   { this.svg.call(this.d3RenderFill.bind(this), data); }

  82    }

  83

  87    protected abstract d3RenderMarkers(svg:d3Base, data:DataSet):void;

  88

  89    protected abstract d3RenderPath(svg:d3Base, data:DataSet):void;

  90
