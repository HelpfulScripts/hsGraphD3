  32/**

  33 * valid {@link SeriesPlot.ValueDef `Value Definiton`} dimensions on cartesian plots:

  34 * - `x`?:  optional values for the x-axis. If omitted, the index of y-values will be used as x-values

  35 * - `y`:   values for the y-axis.

  36 * - `y0`?: optional values for lower fill border on the y-axis; defaults to `0`

  37 * - `r`?:  optional values for the size of markers. If provided, marker rendering is enabled.

  38 * - `label`?: optional values for item labels

  39 * - `popup`?: optional values to show in mouse-over popups.

  40 */

  41export interface CartSeriesDimensions extends SeriesDimensions {

  42    /** name of x-axis data column, or a function returning a value */

  43    x:   ValueDef;    

  44    /** name of y-axis data column, or a function returning a value */ 

  45    y:   ValueDef;    

  46    /** optional, name of y-axis data column for lower fill border, or a function returning a value */

  47    y0?: ValueDef;  

  48    /** optional, name of marker size data column, or a function returning a value */

  49    r?:  ValueDef;   

  50    /** optional, name of label data column, or a function returning a value */

  51    label?: ValueDef;  

  52    /** optional, name of popup data column, or a function returning a value */

  53    popup?: ValueDef;   

  54    /** optional, name of color data column, or a function returning a value */

  55    color?: ValueDef;   

  56}

  70export abstract class CartSeriesPlot extends SeriesPlot {

  71    constructor(cfg:GraphCfg, seriesName:string, dims:CartSeriesDimensions) {

  72        super(cfg, seriesName, dims);

  73    }

  74

  75    /** return the GraphDimension of the independent axis */

  76    protected abscissa:'hor'|'ver' = 'hor';

  77

  78    /** return the list of scalable Series dimesions for each Graph Dimension */

  79    get dimensions():CartDimensions { 

  80        return {

  81            hor: [this.dims.x],

  82            ver: [this.dims.y, this.dims.y0],

  83            size:[this.dims.r],

  84        };

  85    }

  87    /** 

  88     * Set the defaults for the series. Called during `addSeries`.

  89     * */

  90    public getDefaults(): SeriesPlotDefaults {

  91        const def = super.getDefaults();

  92        if (this.dims.r)    { def.marker.rendered = true; }

  93        if (this.dims.y0)   { def.area.rendered = true; }

  94        if (this.dims.label){ 

  95            def.label.rendered = true; 

  96            def.label.color = '#000';

  97        }

  98        return def;

  99    }

 163    protected renderElements(data:DataSet) {

 164        const defaults = this.defaults;

 165        if (defaults.marker.rendered) { this.svg.call(this.d3RenderMarkers.bind(this), data); }

 166        if (defaults.line.rendered)   { this.svg.call(this.d3RenderPath.bind(this), data); }

 167        if (defaults.area.rendered)   { this.svg.call(this.d3RenderFill.bind(this), data); }

 168        if (defaults.label.rendered)  { this.svg.call(this.d3RenderLabels.bind(this), data); }

 169        if (defaults.popup.rendered)  { this.svg.call(this.d3RenderPopup.bind(this), data); }

 170    }

 191    /** clears the stack for this cycle before any series rendering happens. */

 192    public clearStack(data:DataSet) {

 193        const group = this.dims.stacked;

 194        if (group) {

 195            if (data.colNames.indexOf(group) < 0) { 

 196                data.colNames.push(group); 

 197            }

 198            const stackCol = data.colNames.indexOf(group);

 199            data.rows.forEach(row => row[stackCol] = 0);

 200            this.cfg.stack[this.dims.stacked] = {};

 201        }

 202    }
