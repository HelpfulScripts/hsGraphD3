  28/**

  29 * valid standard dimensions on cartesian plots:

  30 * - x: a {@ Graph.ValueDef Value Definiton} for the x-axis.

  31 * - y: a {@ Graph.ValueDef Value Definiton} for the y-axis.

  32 * - y0: optional {@ Graph.ValueDef Value Definiton} for lower fill border on the y-axis

  33 * - r: optional {@ Graph.ValueDef Value Definiton} for the size of markers

  34 */

  35export interface CartSeriesDimensions extends SeriesDimensions {

  36    x:   ValueDef;      // name of x-axis data column, or a function returning a value

  37    y:   ValueDef;      // name of y-axis data column, or a function returning a value

  38    y0?: ValueDef;      // optional, name of y-axis data column for lower fill border, or a function returning a value

  39    r?:  ValueDef;      // optional, name of marker size data column, or a function returning a value

  40}

  41

  42/**

  43 * Abstract base class for all cartesian plots.

  44 */

  45export abstract class CartSeriesPlot extends SeriesPlot {

  46    constructor(cfg:GraphCfg, seriesName:string, dims:CartSeriesDimensions) {

  47        super(cfg, seriesName, dims);

  48    }

  49

  50    /** return the GraphDimension of the independent axis */

  51    protected get independentAxis():'hor'|'ver' { return 'hor'; }

  52

  53    /** return the list of Series dimesions for each Graph Dimension */

  54    get dimensions():CartDimensions { 

  55        return {

  56            hor: [this.dims.x],

  57            ver: [this.dims.y, this.dims.y0],

  58            size:[this.dims.r]

  59        };

  60    }

  61

  62    /** 

  63     * Set the defaults for the series. Called during `addSeries`.

  64     * */

  65    public getDefaults(): SeriesPlotDefaults {

  66        const def = super.getDefaults();

  67        if (this.dims.r)  { def.marker.rendered = true; }

  68        if (this.dims.y0) { def.area.rendered = true; }

  69        return def;

  70    }

  71

  72

  73    //---------- lifecylce methods --------------------

  74

  75    public initialize(svg:d3Base, color?:string): void {

  76        super.initialize(svg, color);

  77        this.svg.append('g').classed('area', true).append('path');

  78        this.svg.append('g').classed('line', true).append('path');

  79        this.svg.append('g').classed('markers', true);

  80

  81        const defaults = this.cfg.defaults.series[this.key];

  82        if (this.dims.r) { defaults.marker.rendered = true; }

  83

  84        const line = this.svg.select('.line');

  85        setStroke(line, defaults.line);

  86

  87        const area = this.svg.select('.area');

  88        setFill(area, defaults.area);

  89

  90        const markers = this.svg.select('.markers');

  91        setStroke(markers, defaults.marker.stroke);

  92        setFill(markers, defaults.marker.fill);

  93        this.clearStack();

  94    }

  95

  96    public preRender(data:DataSet, domains:Domains): void {

  97        super.preRender(data, domains);

  98        this.clearStack();

  99    }

 100

 101    /** renders the component for the given data */

 102    public renderComponent(data:DataSet): void {

 103        this.intializeStackGroup(data);

 104        super.renderComponent(data);

 105        this.renderElements(data);

 106        this.updateStack(data);

 107    }

 108

 109    public postRender(data:DataSet): void {

 110        super.postRender(data);

 111    }

 112        

 113    //---------- support methods during lifecylce --------------------

 114

 115    /**

 116     * returns the `ValueDef`, i.e. column name or `ValueFn`, for which to stack series.

 117     */

 118    protected getStackValueDef():ValueDef {

 119        return this.independentAxis==='hor'? this.dims.y : this.dims.x;

 120    }

 121

 122    protected renderElements(data:DataSet) {

 123        const defaults = (this.cfg.defaults.series[this.key]);

 124        if (defaults.marker.rendered) { this.svg.call(this.d3RenderMarkers.bind(this), data); }

 125        if (defaults.line.rendered)   { this.svg.call(this.d3RenderPath.bind(this), data); }

 126        if (defaults.area.rendered)   { this.svg.call(this.d3RenderFill.bind(this), data); }

 127    }

 128

 129    protected abstract d3RenderMarkers(svg:d3Base, data:DataSet):void;

 130

 131    protected abstract d3RenderPath(svg:d3Base, data:DataSet):void;

 132

 133    protected abstract d3RenderFill(svg:d3Base, data:DataSet):void;

 134

 135    protected getPathElement(svg:d3Base, cls:string):any {

 136        return svg.select(cls).selectAll('path').transition(this.cfg.transition);

 137    }
