  65export interface TitleDefaults extends ComponentDefaults { 

  66    rendered:   boolean;

  67    /** 

  68     * the horizontal position of the title on the viewport, either as a number between [0, 1]

  69     * or a percentage, or a `HPos` value: `0=0%=left edge`, `1=100%right` edge of the viewport.

  70     */

  71    x: UnitPercent | Share | HPos;

  72    /** 

  73     * the vertical position of the title on the viewport, either as a number between [0, 1]

  74     * or a percentage, or a `VPos` value: `0=0%=top` edge, `1=100%=bottom` edge of the viewport.

  75     */

  76    y: UnitPercent | Share | VPos;

  77    style: TextStyle;

  78}

  81export class Title extends GraphComponent {

  82    static type = 'title';

  83

  84    titleText = '';

  85

  86    constructor(cfg:GraphCfg) {

  87        super(cfg, Title.type);

  88    }

  89

  90    public get componentType() { return Title.type; }

  91

  92    public get defaults():TitleDefaults { return this.cfg.graph.defaults.title; }

  93

  94    public set text(title:string) { 

  95        this.titleText = title; 

  96        this.defaults.rendered = true;

  97    }

  98

  99    public createDefaults():TitleDefaults {

 100        const defs = {

 101            rendered:   false,

 104            style:       defaultTextStyle(24),

 105        };

 106        defs.style.font.weight = 'bold';    // 100-900

 107        return defs;

 108    }

 109

 110    initialize(svg:d3Base): void {

 111        this.svg.append('text');

 112    } 

 113

 114    preRender(): void {} 

 115

 116    /**

 117     * renders the Graph's background canvas

 118     * @param cfg 

 119     */

 120    public renderComponent() {

 121        const titleDefs = this.defaults;

 122        if (titleDefs.rendered) {

 123            const vp = this.cfg.viewPort;

 124            const pos = textPos(titleDefs.x, titleDefs.y);

 125            const svg = this.svg.select('text');

 126            setText(svg, titleDefs.style, this.cfg.transition);

 127            svg.attr('x', pos.x.pos * vp.width + vp.orgX).attr('text-anchor', pos.x.anchor)

 128               .attr('y', pos.y.pos * vp.height + vp.orgY).attr('dominant-baseline', pos.y.baseline)

 129               .text(this.titleText);

 130        }

 131    }
