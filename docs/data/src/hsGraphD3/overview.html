   1/**

   2 * # hsGraphD3

   3 * 

   4 * Helpful Scripts Graph convenience wrapper for D3js. 

   5 * [`[Github page]`](https://github.com/HelpfulScripts/hsGraphd3)

   6 * [`[Coverage Info]`](./data/src/hsGraphD3/coverage/)

   7 * ___

   8 * `hsGraphD3` provides convienent programmatic shortcuts to plotting data with [`D3js`](https://d3js.org/).

   9 * 

  10 * ## Usage

  11 * 1. create a {@link Graph `Graph`} object with a root DOM element to attach to: `graph = new hsGraphD3.Graph(root);'

  12 * 2. add desired series configuartions, specifying the type of series, as well as the column names to use for the x- and y-axes, 

  13 * and other variables, depending on the series type.

  14 * 3. apply any desired formatting changes to the default

  15 * 3. render the graph for a given data set: `graph.render(data);`

  16 * 

  17 * ## Data

  18 * Data is provided either as a [`Data`](https://helpfulscripts.github.io/hsDatab/#!/api/hsDatab/0) object, 

  19 * or in table form

  20 * ```

  21 * [

  22 *    colNames[],

  23 *    [],

  24 *    ...

  25 * ]

  26 * ```

  27 * or in object-literal form 

  28 * ```

  29 * {

  30 *    name?:  string;

  31 *    colNames:  string[];   

  32 *    rows:   [][];

  33 * }

  34 * ```

  35 * 

  36 * ## Example 1: `bubble` chart

  37 * hsGraphD3'}>

  38 * script.js'>

  39 * // create data set:

  40 * const data = {

  41 *    colNames:['date', 'time', 'volume', 'costs'], 

  42 *    rows:[   ['1/1/14', -1,     0.2,    0.3], 

  43 *             ['1/1/16',  0.2,   0.7,    0.2], 

  44 *             ['9/1/16',  0.4,   0.1,    0.3],

  45 *             ['5/1/17',  0.6,  -0.2,    0.1], 

  46 *             ['7/1/18',  0.8,   0.3,    0.5], 

  47 *             ['1/1/19',  1,     0.2,    0.4]]

  48 * };

  49 * 

  50 * // create the graph and define the series to plot:

  51 * const graph = new hsGraphD3.GraphCartesian(root);

  52 * graph.addSeries('bubble', 'time', 'volume', 'costs');

  53 * 

  54 * // adjust some settings:

  55 * graph.defaults.axes.color = '#88f';  // both axes appear blue

  56 * with (graph.defaults.scales.dims.size.range) {

  57 *      min = 10;                       // min marker size

  58 *      max = 50;                       // max marker size

  59 * }

  60 * 

  61 * // trigger the update loop to plot the data

  62 * graph.render(data, 2000, data => {

  63 *      // modify the data in this round:

  64 *      data.rows.map(row => { 

  65 *          row[2] = (Math.random()-0.1)*200;   // y-value

  66 *          row[3] = Math.random();             // radius

  67 *      });

  68 *      return true;

  69 * });

  70 *

  73 * ## Example 2: `line` vs. `timeseries`

  74 * hsGraphD3'}>

  75 * script.js'>

  76 * // create data set:

  77 * const data = {

  78 *    colNames:['date', 'line', 'series'], 

  79 *    rows:[   [0,       0.2,     -0.2]], 

  80 * };

  81 * let index = 0;

  82 * while (index<11) { 

  83 *    val = Math.random();

  84 *    data.rows.push([index++, val, val-1]);

  86 * 

  87 * // create the graph and define the series to plot:

  88 * const graph = new hsGraphD3.GraphCartesian(root);

  90 * graph.addSeries('timeseries', 'date', 'series');

  91 * 

  92 * //----- adjust some settings:

  93 * with (graph.defaults) {

  94 *    scales.dims.hor.aggregateOverTime = false; // forget the past

  95 *    series.series0.line.color = '#00a';  // first line blue

  98 * }

  99 *

 101 * graph.render(data, 1000, update);

 114 * }
