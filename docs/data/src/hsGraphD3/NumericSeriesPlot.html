   1/**

   2 * # NumericSeriesPlot

   3 * 

   4 * Abstract base class for all numeric series plot types, i.e. plots with numeric coordinates

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.add(, {: , ...});

   8 * ``` 

   9 * - `` is one of the registered types: 

  10 *     -   {@link plots.Line `line`} a 2D line plot

  11 *     -   {@link plots.Bubble `bubble`} a 2D scatter plot with marker sizes driven by the data

  12 *     -   {@link plots.Area `area`} a 2D area plot filling to the x-axis

  13 *     -   {@link plots.TimeSeries `timeseries`} a 2D scatter plot with marker sizes driven by the data

  14 *     -   {@link plots.Voronoi `voronoi`} a voronoi diagrom with centroids and partition

  15 * - `` is the semantic dimension to set. See {@link CartSeriesPlot.CartSeriesDimensions CartSeriesDimensions} for valid dimensions. 

  16 * - `` is the {@link SeriesPlot.ValueDef value definition}. 

  17 */

  35/**

  36 * Abstract base class of a  cartesian series plot. 

  37 */

  38export abstract class NumericSeriesPlot extends SeriesPlotCartesian { 

  39

  40    //---------- lifecylce methods --------------------

  41

  42    initialize(plot:d3Base, color?:string): void {

  43        super.initialize(plot, color);

  44    }

  45

  46    preRender(data:NumericDataSet): void {

  47        super.preRender(data);

  48        const defaults = this.defaults;

  49        if (defaults.area.rendered && this.dims.y0===undefined) { 

  50            this.dims.y0 = this.dims.stacked? this.dims.stacked : ()=>0; 

  51        } 

  52    }

  53

  54    //---------- support methods during lifecylce --------------------

  55

  56    protected markerShape() { return 'circle'; }

  57

  58    protected d3RenderFill(plot:d3Base, data:NumericDataSet) {

  59        // const scales = this.cfg.components.scales.scaleDims;

  60        let line = this.line = this.line || this.getPath(data.rows, data.colNames, this.dims.y);

  61        if (this.dims.y0!==undefined) {

  62            line += `L` + this.getPath(data.rows.reverse(), data.colNames, this.dims.y0, false).slice(1); // replace first 'M' with 'L'

  63        }

  64        return this.getPathElement(plot, '.area').attr('d', (d:any) => line);

  65    }

  70    protected d3DrawMarker(markers:d3Base, data:DataSet, defaults:SeriesPlotDefaults) {

  71        const scales = this.cfg.components.scales.scaleDims;

  72        const xAccess = this.accessor(this.dims.x, data.colNames);

  73        const yAccess = this.accessor(this.dims.y, data.colNames);

  74        // don't scale markers as 'stacked markers' -> use super instead of this:

  75        const rAccess = this.accessor(this.dims.r, data.colNames, false);

  76        const rDefault = this.defaults.marker.size;

  77        markers

  78            .attr("cx", (d:number[], i:number) => scales.hor(xAccess(d, i)))

  79            .attr("cy", (d:number[], i:number) => scales.ver(yAccess(d, i)))

  80            .attr("r",  (d:number[], i:number) => this.dims.r? scales.size(rAccess(d, i)) : rDefault);

  81    }

  82    

  83    protected d3DrawLabels(labels:d3Base, data:DataSet, defaults:SeriesPlotDefaults) {

  84        const scales = this.cfg.components.scales.scaleDims;

  85        const xAccess = this.accessor(this.dims.x, data.colNames);

  86        const yAccess = this.accessor(this.dims.y, data.colNames);

  87        const rAccess = this.accessor(this.dims.r, data.colNames, false);

  88        const rDefault = this.defaults.marker.size;

  89        const lAccess = this.accessor(this.dims.label, data.colNames, false);

  90        const cfg:Label = this.defaults.label;

  91

  92        // const [xpos, ypos, yShift] = this.labelPos(cfg, labels);

  93        const pos = textPos(cfg.xpos, cfg.ypos, cfg.inside);

  94        labels

  95            .attr("x", (d:number[], i:number) => scales.hor(xAccess(d, i))

  96                + (this.dims.r? scales.size(rAccess(d, i)) : rDefault) * (pos.x.pos-0.5)*2)

  97            .attr("y", (d:number[], i:number) => scales.ver(yAccess(d, i)) 

  98                + (this.dims.r? scales.size(rAccess(d, i)) : rDefault) * (pos.y.pos-0.5)*2)

  99            .text((d:number[], i:number) => text(lAccess(d, i)))

 100            .attr('text-anchor', pos.x.anchor)

 101            .attr('dominant-baseline', pos.y.baseline)

 102            .attr('dx', ((cfg.hOffset||0)+pos.x.shift*0.4).toFixed(1) + 'em')

 103            .attr('dy', ((cfg.vOffset||0)+pos.y.shift*0.2).toFixed(1) + 'em');

 104    }

 106    /**

 107     * returns the path rendering for the main data line 

 108     * @param rows the data rows set to render from

 109     * @param yDef a constant (defaults to 0), or the data column to render from

 110     */

 111    protected getPath(rows:NumericDataRow[], colNames:string[], yDef: ValueDef = () => 0, useStack=true):string {

 112        const scales = this.cfg.components.scales.scaleDims;

 113        const xAccess = this.accessor(this.dims.x, colNames, useStack);

 114        const yAccess = this.accessor(yDef, colNames, useStack);

 115        const line = d3line()

 116            .x((d:number[], i:number) => scales.hor(xAccess(d, i)))

 117            .y((d:number[], i:number) => scales.ver(yAccess(d, i)))

 118            .curve(curveCatmullRom.alpha(0.2));

 119        return line(<[number, number][]>rows);

 120    }
