src/NumericSeriesPlot.ts

   1/**

   2 * # NumericSeriesPlot

   3 * 

   4 * Abstract base class for all numeric series plot types.

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.addSeries(, {: , ...});

   8 * ``` 

   9 * - `` is one of the registered types: 

  10 *     -   {@link plots.Line `line`} a 2D line plot

  11 *     -   {@link plots.Bubble `bubble`} a 2D scatter plot with marker sizes driven by the data

  12 *     -   {@link plots.Area `area`} a 2D area plot filling to the x-axis

  13 *     -   {@link plots.Band `band`} a 2D area plot filling between 2 series 

  14 *     -   {@link plots.TimeSeries `timeseries`} a 2D scatter plot with marker sizes driven by the data

  15 *     -   {@link plots.Voronoi `voronoi`} a voronoi diagrom with centroids and partition

  16 * - `` is one of the semantic dimensions defined for the plot. `NumericSeriesPlots` define the 

  17 *   following dimensions:

  18 *     - `x`: required; the value to plot along the x-axis

  19 *     - `y`: required; the value to plot along the y-axis

  20 *     - `y0`: optional; for areas, the lower bound of the area. 

  21 *       Specifying a value for `y0` will automatically enable area rendering

  22 *     - `r`: optional; for markers, the value to use for the size

  23 *       Specifying a value for `r` will automatically enable markers

  24 * - `` is the value to use for the {@link Series.SeriesDimensions semantic dimension}. Values may be specified

  25 *     - by `string` to specify the column name of the data set to use

  26 *     - by `number` to specify a constant value to use.  

  27 */

  28

  42/** */

  43function accessor(v:NumberSet, colNames:string[], scale:NumberScale) {

  44    return (d:number[]) => scale(typeof(v)==='number'? v : d[colNames.indexOf(v)]);

  45}

  48/**

  49 * Abstract base class of a  cartesian series plot. 

  50 */

  51export abstract class NumericSeriesPlot extends CartSeriesPlot { 

  52    /** the main data line  */

  53    protected line: string;         // d3Line;

  54

  55    //---------- lifecylce methods --------------------

  56

  57    initialize(svg:d3Base, color?:string): void {

  58        super.initialize(svg, color);

  59    }

  60

  61    preRender(data:NumericDataSet, domains:Domains): void {

  62        super.preRender(data, domains);

  63        const defaults = (this.cfg.defaults.series[this.key]);

  64        if (defaults.area.rendered && this.dims.y0===undefined) { this.dims.y0 = 0; } 

  65        this.line = undefined;

  66    }

  67

  68

  69    //---------- support methods during lifecylce --------------------

  70

  71    protected d3RenderMarkers(svg:d3Base, data:NumericDataSet) {

  72        const defaults = (this.cfg.defaults.series[this.key]).marker;

  73        if (defaults.rendered) {

  74            const samples:any = svg.select('.markers').selectAll("circle")

  75                .data(data.rows, d => d[0]);                // bind to data, iterate over rows

  76            samples.exit().remove();                        // remove unneeded circles

  77            samples.enter().append('circle')                // add new circles

  78                .call(this.d3DrawMarker, this, data.colNames)

  79            .merge(samples).transition(this.cfg.transition) // draw markers

  80                .call(this.d3DrawMarker, this, data.colNames);

  81        }

  82    }

  83

  84    protected d3RenderPath(svg:d3Base, data:NumericDataSet) {

  85        this.line = this.line || this.getLine(data.rows, data.colNames, this.dims.y);

  86        return this.getPathElement(svg, '.line').attr('d', (d:any) => this.line);

  87    }

  88

  89    protected d3RenderFill(svg:d3Base, data:NumericDataSet) {

  90        this.line = this.line || this.getLine(data.rows, data.colNames, this.dims.y);

  91        let line0 = '';

  92        if (this.dims.y0!==undefined) {

  93            const max = data.rows.length-1;

  94            const xmax = accessor(this.dims.x, data.colNames, this.cfg.scales.hor)(data.rows[max]);

  95            const x0   = accessor(this.dims.x, data.colNames, this.cfg.scales.hor)(data.rows[0]);

  96            const y    = accessor(this.dims.y0, data.colNames, this.cfg.scales.ver)(data.rows[max]);

  97            line0 = `L${xmax},${y}`;

  98            line0 += (typeof(this.dims.y0)==='number')? `L${x0},${y}` :

  99                this.getLine(data.rows.reverse(), data.colNames, this.dims.y0).slice(8);  // remove first 'M' command

 100        }

 101        return this.getPathElement(svg, '.area').attr('d', (d:any) => this.line + line0);

 102    }

 103

 104    protected d3DrawMarker(markers:d3Base, plot:NumericSeriesPlot, colNames:string[]) {

 105        markers

 106            .attr("cx", (d:number[]) => accessor(plot.dims.x, colNames, plot.cfg.scales.hor)(d))

 107            .attr("cy", (d:number[]) => accessor(plot.dims.y, colNames, plot.cfg.scales.ver)(d))

 108            .attr("r",  (d:number[]) => (plot.dims.r!==undefined)? 

 109                                        accessor(plot.dims.r, colNames, plot.cfg.scales.size)(d)

 110                                      : plot.cfg.defaults.series[plot.key].marker.size);

 111    }

 112    

 113    /**

 114     * returns the path rendering for the main data line 

 115     * @param data the data set to render from

 116     * @param yCol a constant (defaults to 0), or the data column to render from

 117     */

 118    protected getLine(rows:NumericDataRow[], colNames:string[], yCol: NumberSet = 0):string {

 119        const line = d3line()

 120            .x(accessor(this.dims.x, colNames, this.cfg.scales.hor))

 121            .y(accessor(yCol, colNames, this.cfg.scales.ver))

 122            .curve(curveCatmullRom.alpha(0.2));

 123        return line(<[number, number][]>rows);

 124    }

 125

 126    protected getPathElement(svg:d3Base, cls:string):any {

 127        return svg.select(cls).selectAll('path').transition(this.cfg.transition);

 128    }

 129}
