   1/**

   2 * # SeriesPlot

   3 * 

   4 * Abstract base class for all series plot types.

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.series.add(, {x:, ...});

   8 * ``` 

   9 * where `dataRef` is a data reference, either

  10 * - the name of a column in the data set to use

  11 * - or a function, returning the data to use. The function will be called at runtime once for each row

  12 * in the data set supplied when calling `render`, and will receive as parameters the `dataRow`, 

  13 * the `index` of the row, and the entire `rows` array.

  14 * To specify a constant of value 5, simply supply `()=>5`.

  15 */

  16

  46/**

  47 * Returns an accessor function to access the numeric value in a data row. 

  48 * The type of `v` determines how to access the value: 

  49 * - If `v` is a function it will be valuated for the provided index `i` to return the result.

  50 * - If `v` is a number it will be returned as constant result.

  51 * - If `v` is a string and contained in `colNames` it specifies the column to index in the 

  52 * supplied `row` to return as result.

  53 * @param v the `ValueDef` specifying the value

  54 * @param colNames a list of names for the coluymns in the `DataSet`

  55 * @return an accessor function `(row?:DataRow, i?:number) => DataVal` 

  56 * that returns a `DataVal` value. The function

  57 * receives a `DataRow` and the index of the row in the `DataSet` as a parameter. 

  58 */

  59function value(v:ValueDef, colNames:string[]):(row?:DataRow, i?:number) => DataVal {

  60    switch (typeof(v)) {

  61        case 'function': return (row, i) => (v)(i);

  62        case 'number':   log.info(`accessing constant number ${v}`);

  63                         return () => v;

  64        case 'string':

  65        default:         const index = colNames.indexOf(''+v);

  66                         return (row) => row[index];

  67    }

  68}

  70/**

  71 * The base class for all series plots. It manages

  72 * - setting the series' defaults

  73 * - expanding domains to values of this series

  74 * - the rendering lifecycle for the series

  75 * - maintaining a stack mechanism that allows series to be stacked on one another

  76 */

  77export abstract class SeriesPlot { 

  78    /** 

  79     * a list of data column names used,

  81     */

  86

  88    protected cfg: GraphCfg;

  89

 105    /** 

 106     * Set the defaults for the series. Called during `addSeries`.

 107     * */

 108    public getDefaults(): SeriesPlotDefaults {

 109        const def:any = {

 110            line:   defaultStroke(5),

 111            marker: defaultMarkerStyle(),

 112            area:   defaultFill()

 113        };

 114        def.line.rendered = true;

 115        def.marker.rendered = false;

 116        def.area.rendered = false;

 117        return def;

 118    }

 120    public expandDomains(dataSet:DataSet, domains:Domains) {

 121        this.intializeStackGroup(dataSet);

 122        const dims:GraphDimensions = this.dimensions;

 123        Object.keys(dims).map(dim => {

 124            const type = this.cfg.graph.defaults.scales.dims[dim].type;

 125            dims[dim].map(col => { if (col!==undefined) { 

 126                const valueFn = this.value(dim, col, dataSet.colNames);

 127                switch(type) {

 128                    case 'ordinal':     

 129                        domains[dim] = this.expandOrdinalDomain(dataSet, domains[dim] || [], valueFn); 

 130                        break;

 131                    default:            

 132                        domains[dim] = this.expandNumDomain(dataSet, domains[dim] || [1e99, -1e99], valueFn);

 133                }

 134            }});

 135        });

 136    }

 137    

 138    protected expandNumDomain(dataSet:DataSet, domain:NumDomain, fn:(row?:DataRow, i?:number) => DataVal):NumDomain {

 139        return dataSet.rows.reduce((dom:NumDomain, row:DataRow, i:number):NumDomain => {

 140            const val = fn(row, i);

 141            dom[0] = Math.min(val, dom[0]);

 142            dom[1] = Math.max(val, dom[1]);

 143            return dom;

 144        }, domain);

 145    }

 146    

 147    protected expandOrdinalDomain(dataSet:DataSet, domain:OrdDomain, fn:(row?:DataRow, i?:number) => DataVal):OrdDomain {

 148        return dataSet.rows.reduce((dom:OrdDomain, row:DataRow, i:number):OrdDomain => {

 149            const val = fn(row, i);

 150            if (dom.indexOf(val) < 0) { dom.push(val); }

 151            return dom;

 152        }, domain);

 153    }

 155    /**

 156     * Returns an accessor function to access the numeric value in a data row. 

 157     * `v` determines how to access the value: 

 158     * - If `v` is contained in `colNames` it specifies the column to index in the 

 159     * supplied `row` to return as result.

 160     * - If `v` is a function it will be valuated for the provided index `i` to return the result.

 161     * - If 'v' is a number, it will be used as a constant to return as result

 162     * @param dim the Graph Dimension, used for stacking; e.g. 'hor', or 'ver' 

 163     * @param v the `ValueDef` specifying the value

 164     * @param colNames a list of names for the coluymns in the `DataSet`

 165     * @return an accessor function `(row?:DataRow, i?:number) => DataVal` 

 166     * that returns a `DataVal` value. The function

 167     * receives a `DataRow` and the index of the row in the `DataSet` as a parameter. 

 168     */

 169    value(dim:string, v:ValueDef, colNames:string[]):(row?:DataRow, i?:number) => DataVal {

 170        return value(v, colNames);

 171    }

 191    /** clears the stack for this cycle before any series rendering happens. */

 192    public clearStack() {

 193        if (this.dims.stacked) {

 194            this.cfg.stack[this.dims.stacked] = [];

 195        }

 196    }

 197    

 198    /** 

 199     * If the required stack group has not been initialized,

 200     * set it to all zeros before rendering this series  

 201     */

 202    protected intializeStackGroup(data:DataSet) {

 203        const group:string = this.dims.stacked;

 204        if (group && this.cfg.stack[group].length === 0) {

 205            this.cfg.stack[group] = data.rows.map(() => 0);

 206        }

 207    }

 208

 209    /** update stack after rendering series. */

 210    protected updateStack(data:DataSet) {

 211        const stacked = this.dims.stacked;

 212        if (stacked) {

 213            // use static value function to avoid stacking side effect.

 214            const valFn = value(this.getStackValueDef(), data.colNames);

 215            const stack = this.cfg.stack[stacked];

 216            data.rows.forEach((r, i) => stack[i] += valFn(r, i));

 217        }

 218    }
