  43/**

  44 * specifies the valid dimensions (e.g. 'x' for the x-axis) for a {@link SeriesPlot.SeriesPlot `SeriesPlot`}.

  45 * For each dimension, the value can be either 

  46 * - a string that identifies the column name in the data set to use

  47 * - or a {@link SeriesPlot.ValueFn `ValueFn`} function that returns the data to use.

  48 * </ul>

  49 * Optional auxiliary dimensions for all plots:<ul>

  50 * - `label`?: optional values for item {@link SeriesPlot.SeriesDimensions.label labels}

  51 * - `popup`?: optional values to show in mouse-over {@link SeriesPlot.SeriesDimensions.popup popups}.

  52 * - `color`?: optional values to determine {@link SeriesPlot.SeriesDimensions.color marker colors}

  53 * - `stacked`?: optional {@link SeriesPlot.SeriesDimensions.stacked stack group}. 

  54 *    Series with the same group will be stacked on each other

  55 */

  56export interface SeriesDimensions { 

  57    [dim:string]: ValueDef; 

  58    /** 

  59     * optional, name of label data column, or a function returning a value.

  60     * See {@link Settings.Label Settings.Label} for configuring labels.

  61     */

  62    label?: ValueDef;  

  63    /** 

  64     * optional, name of popup data column, or a function returning a value.

  65     * The popup is triggered when the mouse enters a marker or label area.

  66     * If omitted, a default popup will configured to show the marker's 

  67     * abscissa and ordinate names and values.

  68     * See {@link Popup Popup} for an example.

  69     */

  70    popup?: ValueDef;   

  71    /** 

  72     * optional, name of color data column, or a function returning a value.

  73     * The returned color will be used on the marker of that data point. If 

  74     * omitted, the `series'` color will be used, as defined in the default settinmgs:

  75     * `graph.defaults.series.<seriesKey>.marker.fill`. 

  76     * See {@link SeriesPlot.SeriesPlot.d3MarkerColors marker colors} for more details.

  77     */

  78    color?: ValueDef;   

  79    /** 

  80     * optional stack group. Series with the same group will be stacked on each other. 

  81     * `stacked` differes from other dimensions in that the `ValueDef` is always a `string`

  82     * used as a unique identifier for the series to be stacked (and not as a named index into the data set).

  83     * For example: `stacked:'myStack'` would stack all series with the ID `myStack`. 

  84     */

  85    stacked?:     string;

  86}

  88/** 

  89 * Basic `ValueDef` definition, used in {@link SeriesPlot.SeriesPlot `SeriesPlot`}. 

  90 * - `string`: the name of column in the data set, e.g. `x:'time'`

  91 * - `number`: a constant value, e.g. `r: 5`

  92 * - `ValueFn`: a {@link SeriesPlot.ValueFn `user-defined function`} returning the value.

  93 * - `string[]`: a list of column names, e.g. `['from', 'to']`

  94 */

  95export type ValueDef = string|number|ValueFn|string[];

  97/** 

  98 * a user-defined function returning the {@link Graph.DataVal `DataVal`} of a data point.

  99 * The function will be called at runtime once for each row in the data set.

 100 * @param rowIndex the index of the row in the {@link Graph.DataSet `DataSet's`} rows array. 

 101 * @return the value of type `DataVal` to use for the row.

 102 */

 103export interface ValueFn { (row:DataRow, rowIndex?:Index): DataVal; }

 105/**

 110    return typeof val==='number'? val.toFixed(1) : <string>val;

 114/**

 115 * Abstract base class for all series plot types. It manages

 116 * - setting the series' defaults

 117 * - expanding domains to values of this series

 118 * - the rendering lifecycle for the series

 119 * - maintaining a stack mechanism that allows series to be stacked on one another

 120 * To create a series plot, add the desired plot type to the graph:

 121 * ```

 122 * graph.add(<type>, {<dim>:<ValueDef>, ...});

 123 * ``` 

 124 * where `<dim>` specifies a {@link SeriesPlot.SeriesDimensions series dimension}

 125 * and {@link SeriesPlot.ValueDef `ValueDef`} provides the data values for the dimension, 

 126 * as one of:

 127 * - the name of a column in the data set to use

 128 * - a constant number

 129 * - or a user-defined {@link SeriesPlot.ValueFn function}, returning the data to use. 

 130 */

 196    /**

 197     * Returns an accessor function `(row:DataVal[], rowIndex:Index) => DataVal` to access the numeric value in a data row.

 198     * The type of `v` determines how to access the value: 

 199     * - If `v` is a function it will be valuated for the provided `rowIndex` and `row` to return the result.

 200     * - If `v` is a number it will be returned as constant result.

 201     * - If `v` is a string and contained in `colNames` it specifies the column to index in `row` 

 202     * @param v the `ValueDef` specifying the value

 203     * @param colNames a list of names for the coluymns in the `DataSet`

 204     * @return an accessor function `(row:DataVal[], rowIndex:Index) => DataVal` 

 205     * that returns a `DataVal` value. The function

 206     * receives a `DataRow` and the index of the row in the `DataSet` as a parameter. 

 207     */

 208    protected accessor(v:ValueDef, colNames:string[], useStack=true):(row:DataRow, rowIndex:Index) => DataVal {

 209        switch (typeof(v)) {

 210            case 'function':return (row, rowIndex) => (<ValueFn>v)(row, rowIndex);

 211            case 'number':  return () => <DataVal>v;

 212            case 'string':

 213            default:        const c = colNames.indexOf(''+v); 

 214                            // try row.data[c] (as passed by d3.arc), then row[c]

 215                            return row => row['data']? row['data'][c] : row[c];

 216        }

 217    }

 220    /**

 221     * applies `stroke` and `fill` colors to `items`.

 222     * Colors are determined by the `color` tag in the series definition as in

 223     * ```

 224     * graph.add('column', {x:'item', y:'Mary', color:<colorValue>}); 

 225     * ```

 226     * Various coloring modes are available via the `<colorValue>` tag:

 227     * - `string`: 

 228     *     1. addresses a data column containing

 229     *          - `numbers`: used to index the `default.marker.scheme` table

 230     *          - `strings`: used as direct colors; must be in the format `#rgb` 

 231     *     2. if not addressing a data column, the value addresses an entry in an

 232     *        internal color schemes list. Available entries are 

 233     *          [cat10](https://github.com/d3/d3-scale-chromatic#schemeCategory10),

 234     *          [blues](https://github.com/d3/d3-scale-chromatic#schemeBlues),

 235     *          [greens](https://github.com/d3/d3-scale-chromatic#schemeGreens),

 236     *          [greys](https://github.com/d3/d3-scale-chromatic#schemeGreys),

 237     *          [oranges](https://github.com/d3/d3-scale-chromatic#schemeOranges),

 238     *          [purples](https://github.com/d3/d3-scale-chromatic#schemePurples),

 239     *          [reds](https://github.com/d3/d3-scale-chromatic#schemeReds)

 240     * - `number`: used to index the `default.marker.scheme` table

 241     * - `function (rowIndex:Index)=>string` a function, returning a `#rgb` string for the 

 242     *    provided row index

 243     * @param items the items being colored

 244     * @param numRows the number of data rows

 245     */

 246    protected d3MarkerColors(items:d3Base, data:DataSet, defaults:SeriesPlotDefaults) {

 247        const colors = this.dims.color;

 248        function getColor(d:number[], i:number):string  {

 249            const scheme = schemes[defaults.marker.scheme];

 250            if (typeof colors === 'function') {

 251                const color = colors(d, i);

 252                return typeof color === 'number'? scheme((color % 10) / 10) : <string>color;

 253            } else if (typeof colors === 'string') {

 254                const col = data.colNames.indexOf(colors);

 255                if (col >= 0) {

 256                    return d['data']? scheme(d['data'][col]) : scheme(d[col]);

 257                } else if (schemes[colors]) { 

 258                    return schemes[colors](i / data.rows.length);

 259                } else {

 260                    return colors;

 261                }

 262            } else if (typeof colors === 'number') {

 263                return scheme((colors % 10) / 10);

 264            } else {

 265                return '#f00';

 266            }

 267        }

 268        if (colors) {

 269            // items.attr('stroke', getColor)

 270            //      .attr('fill', getColor);

 271            items.attr('color', getColor);

 272        }

 273    }

 304    /**

 305     * formats the popup string to display

 306     * @param colNames 

 307     */

 308    protected d3RenderPopup(data:DataSet):AccessFn {

 309        if (this.dims.popup) {

 310            const format = d3format('.4r');

 311            const popupAccess = this.accessor(this.dims.popup, data.colNames, false);

 312            const abscissa = this.dimensions[this.abscissa][0];

 313            const absAccess = this.accessor(abscissa, data.colNames, false);

 314            // if dims x present, reflect its column nname; else use 'index'

 315            const absName = typeof abscissa === 'string'? abscissa : this.abscissa;

 316            return (r:DataVal[], i:number) => {

 317                let val = popupAccess(r,i);

 318                if (typeof val === 'number') { val = format(val); }

 319                let abs = absAccess(r,i);

 320                if (typeof abs === 'number') { abs = format(abs); }

 321                return typeof this.dims.popup === 'function' ? popupAccess(r,i) :`

 322                    ${this.dims.popup} = ${val}<br>

 323                    ${absName} = ${absAccess(r,i)}

 324                `;

 325            };

 326        }

 327    }

 368    public renderComponent(data:DataSet): void {

 369        const defaults = this.defaults;

 370        if (defaults.marker.rendered) { this.svg.call(this.d3RenderMarkers.bind(this), data); }

 372        if (defaults.area.rendered)   { this.svg.call(this.d3RenderFill.bind(this), data); }

 374        if (defaults.popup)           { this.svg.call(this.d3SeriesPopup.bind(this)); }

 375    }
