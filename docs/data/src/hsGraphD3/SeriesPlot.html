src/SeriesPlot.ts

   1/**

   2 * # SeriesPlot

   3 * 

   4 * Abstract base class for all series plot types.

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.addSeries(, ...);

   8 * ``` 

   9 * -  is one of the registered types: 

  10 *     - {@Line `line`} a 2D line plot

  11 *     - {@Bubble `bubble`} a 2D scatter plot with marker sizes driven by the data

  12 * 

  13 */

  14

  15/** */

  16import { BaseType }         from 'd3';

  17import { line as d3line}    from "d3";

  18import { Line as d3Line }   from "d3";

  19import { curveCardinal }    from 'd3';

  20import { Data }             from 'hsdatab';

  21import { NumDomain }        from 'hsdatab';

  22import { GraphCfg }         from './GraphComponent';

  23import { Stroke }           from './Settings';

  24import { d3Base }           from './Settings';

  25import { MarkerStyle }      from './Settings';

  26import { defaultStroke }    from './Settings';

  27import { defaultMarker }    from './Settings';

  28import { setStroke }        from './Settings';

  29import { setFill }          from './Settings';

  30import { Scales }           from './Scale';

  31

  32export type d3Selection = d3.Selection; 

  33

  34export interface SeriesPlotDefaults {

  35    line:   Stroke;

  36    marker: MarkerStyle;

  37}

  38

  39export abstract class SeriesPlot { 

  40    /** the base svg element to render the component into */

  41    protected svg: d3Base;

  42    

  43    /** the render tree configuration */

  44    protected cfg: GraphCfg;

  45

  46    /** the unique series key assigned during cinstruction, used to index the series settings. */

  47    protected seriesKey:string;

  48

  49    /** 

  50     * a list of data column names used,

  51     * reflecting the list of column names provided during construction.

  52     */

  53    protected dims: string[] = [];

  54

  55    /** 

  56     * a list of data column indices, corresponding to `dims`.

  57     */

  58    protected cols: number[] = [];

  59

  60    constructor(cfg:GraphCfg, seriesName:string, ...params:string[]) {

  61        this.cfg = cfg; 

  62        this.seriesKey = seriesName;

  63        this.dims = params;

  64    }

  65

  66    get key() { return this.seriesKey; }

  67

  68    get dimensions() { return this.dims; }

  69

  70    /** set the defaults for the series. */

  71    getDefaults(): SeriesPlotDefaults {

  72        return {

  73            line:   defaultStroke(5),

  74            marker: defaultMarker()

  75        };

  76    }

  77

  78    //---------- lifecylce methods --------------------

  79

  80    initialize(svg:d3Base): void {

  81        this.svg = svg.append('g').classed(this.seriesKey, true);

  82        this.svg.append('g').classed('lines', true).append('path');

  83        this.svg.append('g').classed('markers', true);

  84    }

  85

  86    preRender(data:Data, domains:{[dim:string]: NumDomain}): void {

  87        const defaults = this.cfg.defaults.series[this.key];

  88        this.cols = this.dims.map(d => data.colNumber(d));

  89        if (this.dims.length > 2) {

  90            // param 2: size of marker

  91            this.setMarkerSizeScale(this.dims[2], domains[this.dims[2]]);

  92        }

  93        

  94        const markers = this.svg.select('.markers');

  95        setStroke(markers, defaults.marker.stroke);

  96        setFill(markers, defaults.marker.fill);

  97

  98        const lines = this.svg.select('.lines');

  99        setStroke(lines, defaults.line);

 100    }

 101

 102    /** renders the component for the given data */

 103    abstract renderComponent(data:Data): void;

 104

 105    //------- render elements

 106    setMarkerSizeScale(dataCol:string, domain:NumDomain) {

 107        const def = this.cfg.defaults.scales.dims[dataCol];

 108        this.cfg.scales[dataCol] = Scales.createScale(def, domain);

 109    }

 110

 111    renderMarkers(data:Data) {

 112        const defaults = this.cfg.defaults.series[this.key].marker;

 113        const rScale = this.cfg.scales[this.dims[2]];

 114

 115        const samples:any = this.svg.select('.markers').selectAll("circle").data(data.getData());

 116        samples.exit().remove();            // remove unneeded circles

 117        samples.enter().append('circle')    // add new circles

 118            .attr("cx", (d:number[]) => this.cfg.scales.hor(d[this.cols[0]]))

 119            .attr("cy", (d:number[]) => this.cfg.scales.ver(d[this.cols[1]]))

 120            .attr("r",  (d:number[]) => this.cols[2]? rScale(d[this.cols[2]]) : defaults.size)

 121        .merge(samples).transition(this.cfg.transition)   // draw markers

 122            .attr("cx", (d:number[]) => this.cfg.scales.hor(d[this.cols[0]]))

 123            .attr("cy", (d:number[]) => this.cfg.scales.ver(d[this.cols[1]]))

 124            .attr("r",  (d:number[]) => this.cols[2]? rScale(d[this.cols[2]]) : defaults.size);

 125    }

 126

 127    renderPath(data:Data) {

 128        const line:d3Line = d3line()

 129            .x(d => this.cfg.scales.hor(d[this.cols[0]]))

 130            .y(d => this.cfg.scales.ver(d[this.cols[1]]))

 131            .curve(curveCardinal.tension(0))

 132        ;

 133        const path = this.svg.select('.lines').selectAll('path').data([data.getData()]);

 134        path.transition(this.cfg.transition)

 135            .attr('d', d => line(d));

 136    }

 137}
