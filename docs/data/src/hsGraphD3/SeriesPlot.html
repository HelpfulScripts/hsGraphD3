  37/**

  38 * specify the values to use for different semantic dimension (e.g. 'x' for the x-axis) of 

  39 * each {@link SeriesPlot.SeriesPlot `SeriesPlot`}.

  40 * For each dimension, the value can be either 

  41 * - a string that identifies the column name in the data set to use

  42 * - or a {@link SeriesPlot.ValueFn `ValueFn`} function that returns the data to use.

  43 */

  44export interface SeriesDimensions { 

  45    [dim:string]: ValueDef; 

  46    /** optional, name of label data column, or a function returning a value */

  47    label?: ValueDef;  

  48    /** optional, name of popup data column, or a function returning a value */

  49    popup?: ValueDef;   

  50    /** optional, name of color data column, or a function returning a value */

  51    color?: ValueDef;   

  52    /** optional stack group. Series with the same group will be stacked on each other */

  53    stacked?:     string;

  54}

  55

  56/** 

  57 * Basic `ValueDef` definition, used in {@link SeriesPlot.SeriesPlot `SeriesPlot`}. 

  58 * - `string`: the name of column in the data set, e.g. `x:'time'`

  59 * - `number`: a constant value, e.g. `r: 5`

  60 * - {@link SeriesPlot.ValueFn `ValueFn`}: a function returning the value.

  61 */

  62export type ValueDef = string|number|ValueFn;

  63

  64/** 

  65 * a function returning the value of a data point 

  66 * The function will be called at runtime once for each row in the data set 

  67 * supplied when calling `render`, and will receive as parameters the 

  68 * `index` of the row in the {@link Graph.DataSet `DataSet's`} rows array.

  69 */

  70export interface ValueFn { (rowIndex:number): DataVal; }

  71

  72/**

  73 * coverts a `DataVal` to a `string`

  74 * @param val 

  75 */

  76export function text(val:DataVal) {

  77    return typeof val==='number'? val.toFixed(1) : val;

  78}

  81/**

  82 * Abstract base class for all series plot types. It manages

  83 * - setting the series' defaults

  84 * - expanding domains to values of this series

  85 * - the rendering lifecycle for the series

  86 * - maintaining a stack mechanism that allows series to be stacked on one another

  87 * To create a series plot, add the desired plot type to the graph:

  88 * ```

  89 * graph.series.add(, {x:, ...});

  90 * ``` 

  91 * where {@link SeriesPlot.ValueDef `ValueDef`} is a data reference, either

  92 * - the name of a column in the data set to use

  93 * - or a function, returning the data to use. 

  94 * To specify a constant of value 5, simply supply `()=>5`.

  95 */

 147    public expandDomains(dataSet:DataSet, domains:Domains) {

 148        this.updateStack(dataSet);

 149        const dims:GraphDimensions = this.dimensions;

 150        Object.keys(dims).map(dim => { // dim='hor', 'ver', size'

 151            const useStack = dim!=='size';  // donst stack-scale marker sizes

 152            const type = this.cfg.graph.defaults.scales.dims[dim].type;

 153            dims[dim].map(colName => { if (colName!==undefined) { 

 154                const valueFn = this.accessor(colName, dataSet.colNames, useStack);

 155                switch(type) {

 156                    case 'ordinal':     

 157                        domains[dim] = this.expandOrdinalDomain(dataSet, domains[dim] || [], valueFn); 

 158                        break;

 159                    default:            

 160                        domains[dim] = this.expandNumDomain(dataSet, domains[dim] || [1e99, -1e99], valueFn);

 161                }

 162            }});

 163        });

 164    }

 166    protected expandNumDomain(dataSet:DataSet, domain:NumDomain, fn:(row?:DataRow, i?:number) => DataVal):NumDomain {

 167        return dataSet.rows.reduce((dom:NumDomain, row:DataRow, i:number):NumDomain => {

 168            const val = fn(row, i);

 169            dom[0] = Math.min(val, dom[0]);

 170            dom[1] = Math.max(val, dom[1]);

 171            return dom;

 172        }, domain);

 173    }

 174    

 175    protected expandOrdinalDomain(dataSet:DataSet, domain:OrdDomain, fn:(row?:DataRow, i?:number) => DataVal):OrdDomain {

 176        return dataSet.rows.reduce((dom:OrdDomain, row:DataRow, i:number):OrdDomain => {

 177            const val = fn(row, i);

 178            if (dom.indexOf(val) < 0) { dom.push(val); }

 179            return dom;

 180        }, domain);

 181    }

 183    /**

 184     * Returns an accessor function `(row:DataVal[], rowIndex:number) => DataVal` to access the numeric value in a data row.

 185     * The type of `v` determines how to access the value: 

 186     * - If `v` is a function it will be valuated for the provided `row` and `rowIndex` to return the result.

 187     * - If `v` is a number it will be returned as constant result.

 188     * - If `v` is a string and contained in `colNames` it specifies the column to index in `row` 

 189     * @param v the `ValueDef` specifying the value

 190     * @param colNames a list of names for the coluymns in the `DataSet`

 191     * @return an accessor function `(row:DataVal[], rowIndex:number) => DataVal` 

 192     * that returns a `DataVal` value. The function

 193     * receives a `DataRow` and the index of the row in the `DataSet` as a parameter. 

 194     */

 195    protected accessor(v:ValueDef, colNames:string[], useStack=true):(row:DataRow, rowIndex:number) => DataVal {

 196        switch (typeof(v)) {

 197            case 'function':return (row, rowIndex) => (v)(rowIndex);

 198            case 'number':  return () => v;

 199            case 'string':

 200            default:        const c = colNames.indexOf(''+v); 

 201                            // try row.data[c] (as passed by d3.arc), then row[c]

 202                            return (row) => row['data']? row['data'][c] : row[c];

 203        }

 204    }

 207    /**

 208     * applies `stroke` and `fill` colors to `items`.

 209     * Colors are determined by the `color` tag in the series definition as in

 210     * ```graph.series.add('column', {x:'item', y:'Mary', color:}); ```

 211     * Various coloring modes are available via the  tag:

 212     * - `string`: 

 213     *     a. addresses a data column containing

 214     *          - `numbers`: used to index the `default.marker.scheme` table

 215     *          - `strings`: used as direct colors; must be in the format '#rgb'  

 216     *     b. if not addressing a data column, the value addresses an entry in an

 217     *        internal color schemes list. Available entries are 

 218     *          [cat10](https://github.com/d3/d3-scale-chromatic#schemeCategory10),

 219     *          [blues](https://github.com/d3/d3-scale-chromatic#schemeBlues),

 220     *          [greens](https://github.com/d3/d3-scale-chromatic#schemeGreens),

 221     *          [greys](https://github.com/d3/d3-scale-chromatic#schemeGreys),

 222     *          [oranges](https://github.com/d3/d3-scale-chromatic#schemeOranges),

 223     *          [purples](https://github.com/d3/d3-scale-chromatic#schemePurples),

 224     *          [reds](https://github.com/d3/d3-scale-chromatic#schemeReds)

 225     * - `number`: used to index the `default.marker.scheme` table

 226     * - `function (rowIndex:number)=>string` a function, returning a '#rgb' string for the 

 227     *    provided row index

 228     * @param items the items being colored

 229     * @param numRows the number of data rows

 230     */

 231    protected d3MarkerColors(items:d3Base, data:DataSet, defaults:SeriesPlotDefaults) {

 232        function getColor(d:number[], i:number):string  {

 233            const scheme = schemes[defaults.marker.scheme];

 234            if (typeof colors === 'function') {

 235                const color = colors(i);

 236                return typeof color === 'number'? scheme((color % 10) / 10) : color;

 237            } else if (typeof colors === 'string') {

 238                const col = data.colNames.indexOf(colors);

 239                if (col >= 0) {

 240                    return d['data']? scheme(d['data'][col]) : scheme(d[col]);

 241                } else if (schemes[colors]) { 

 242                    return schemes[colors](i / data.rows.length);

 243                } else {

 244                    return colors;

 245                }

 246            } else if (typeof colors === 'number') {

 247                return scheme((colors % 10) / 10);

 248            } else {

 249                return '#f00';

 250            }

 251        }

 252        const colors = this.dims.color;

 253        if (colors) {

 254            // items.attr('stroke', getColor)

 255            //      .attr('fill', getColor);

 256            items.attr('color', getColor);

 257        }

 258    }

 260        /**

 261     * formats the popup string to display

 262     * @param colNames 

 263     */

 264    protected d3RenderPopup(data:DataSet):AccessFn {

 265        if (this.dims.popup) {

 266            const format = d3format('.4r');

 267            const popupAccess = this.accessor(this.dims.popup, data.colNames, false);

 268            const abscissa = this.dimensions[this.abscissa][0];

 269            const absAccess     = this.accessor(abscissa, data.colNames, false);

 270            // if dims x present, reflect its column nname; else use 'index'

 271            const absName = typeof abscissa === 'string'? abscissa : this.abscissa;

 272            return (r:DataVal[], i:number) => {

 273                let val = popupAccess(r,i);

 274                if (typeof val === 'number') { val = format(val); }

 275                return typeof this.dims.popup === 'function' ? popupAccess(r,i) :`

 276                    ${this.dims.popup} = ${val}

 277                    ${absName} = ${absAccess(r,i)}

 278                `;

 279            };

 280        }

 281    }
