   1/**

   2 * # SeriesPlot

   3 * 

   4 * Abstract base class for all series plot types.

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.addSeries(, {x:, ...});

   8 * ``` 

   9 * where `dataRef` is a data reference, either

  10 * - the name of a column in the data set to use

  11 * - or a function, returning the data to use. The function will be called at runtime once for each row

  12 * in the data set supplied when calling `render`, and will receive as parameters the `dataRow`, 

  13 * the `index` of the row, and the entire `rows` array.

  14 * To specify a constant of value 5, simply supply `()=>5`.

  15 */

  16

  17/**  */

  21import { defaultMarker }        from "./Settings";

  22import { defaultFill }          from "./Settings";

  23import { GraphCfg }             from "./GraphComponent";

  24import { Area }                 from "./GraphComponent";

  25import { Line }                 from "./GraphComponent";

  27import { DataSet }              from "./Graph";

  31export type d3Selection = d3.Selection;

  33export interface SeriesPlotDefaults {

  34    area:   Area;

  35    line:   Line;

  36    marker: Marker;

  37}

  38

  39

  40export abstract class SeriesPlot { 

  41    /** the base svg element to render the component into */

  42    protected svg: d3Base;

  43    

  44    /** the render tree configuration */

  45    protected cfg: GraphCfg;

  46

  47    /** the unique series key assigned during cinstruction, used to index the series settings. */

  48    protected seriesKey:string;

  49

  50    constructor(cfg:GraphCfg, seriesName:string) {

  51        this.cfg = cfg; 

  52        this.seriesKey = seriesName;

  53    }

  54

  55    public get key() { return this.seriesKey; }

  56

  57    public abstract get dimensions():GraphDimensions;

  58

  59    /** 

  60     * Set the defaults for the series. Called during `addSeries`.

  61     * */

  62    public getDefaults(): SeriesPlotDefaults {

  63        const def:any = {

  64            line:   defaultStroke(5),

  65            marker: defaultMarker(),

  66            area:   defaultFill()

  67        };

  68        def.line.rendered = true;

  69        def.marker.rendered = false;

  70        def.area.rendered = false;

  71        return def;

  72    }

  73    //---------- lifecylce methods --------------------

  74

  75    public initialize(svg:d3Base, color?:string): void {

  76        this.svg = svg.append('g').classed(this.seriesKey, true);

  77        if (color) { this.svg.style('color', color); }

  78    }

  79

  80    public abstract preRender(data:DataSet, domains:Domains): void;

  81

  82    public abstract renderComponent(data:DataSet): void;

  83}
