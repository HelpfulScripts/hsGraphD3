  83export class Axis {

  84    private dir: Direction;

  85    private pos: string;    // left/right or top/bottom

  86    private cfg: GraphCfg;

  87    private svg: d3Base;

  88

  89    constructor(cfg:GraphCfg, dir:Direction) {

  90        this.cfg = cfg;

  91        this.svg = cfg.baseSVG.select('.axes').append('g');

  92        this.dir = dir;

  93        this.svg.classed(`${dir}Axis`, true);

  94    }

  95

  96    public createDefaults():AxisDefaults {

  97        return {

  98            color:      'currentColor',

  99            rendered:   true,

 100            line:       def.defaultLine(1),

 101            tickWidth:  5,

 102            tickMarks:  def.defaultLine(2),

 103            tickLabel:  def.defaultText(),

 104            crossing:   'auto'

 105        };

 106    }

 107

 108    public get defaults():AxisDefaults {

 109        return (this.cfg.defaults.axes)[this.dir];

 110    }

 111    

 112    public renderComponent() {

 113        const trans = this.cfg.transition;

 114        const axisDef = this.cfg.defaults.axes[this.dir];

 115        this.cfg.baseSVG.select('.axes')

 116            .attr('color', this.cfg.defaults.axes.color);

 117

 118        setStroke(this.svg, axisDef.line);

 119        this.setTransform(this.cfg.scales);

 120        const axis:any = this.getD3Axis(this.cfg.scales, axisDef);

 121        this.svg.transition(trans).call(axis);

 122

 123        this.svg.attr('color', axisDef.color);

 124        if (!axisDef.line.rendered) {

 125            this.svg.selectAll('path').attr('style', 'visibility: hidden');

 126        }

 127        if (axisDef.tickLabel.rendered) {

 128            setText(this.svg, axisDef.tickLabel, trans);

 129        } else {

 130            this.svg.selectAll('text').attr('style', 'visibility: hidden');

 131        }

 132        if (axisDef.tickMarks.rendered) {

 133            setStroke(this.svg.selectAll('.tick line'), axisDef.tickMarks);

 134        } else {

 135            this.svg.selectAll('.tick line').attr('style', 'visibility: hidden');

 136        }

 137    }

 139    setTransform(scales:ScaleDims) {

 140        const margins = (this.cfg.defaults.scales).margin;

 141        const trans = this.cfg.transition;

 142        if (this.dir===Direction.horizontal) {

 143            const axisDef = this.cfg.defaults.axes.ver;

 144            const dom = scales.ver.domain();

 145            const cross:ScaleTypes = axisDef.crossing==='auto'? ((dom[0] < 0 && dom[1] > 0)? 0 : dom[0]) : axisDef.crossing;

 146            let yCrossing = scales.ver(cross) + ((scales.ver.type()==='ordinal')? scales.ver.step() : 0);

 147            this.pos = 'bottom';

 148            if (yCrossing < margins.top) {

 149                yCrossing = margins.top;

 150            } else if (yCrossing > this.cfg.viewPort.height-margins.bottom) {

 151                yCrossing = this.cfg.viewPort.height-margins.bottom;

 152                this.pos = 'top';

 153            }

 154            const dx = scales.hor.type()==='ordinal'? scales.hor.paddingOuter()*scales.hor.step()/2 : 0;

 155            this.svg.transition(trans).attr("transform", `translate(${dx}, ${yCrossing})`);

 156        } else {

 157            const axisDef = this.cfg.defaults.axes.hor;

 158            const dom = scales.hor.domain();

 159            const cross:ScaleTypes = axisDef.crossing==='auto'? ((dom[0] < 0 && dom[1] > 0)? 0 : dom[0]) : axisDef.crossing;

 160            let xCrossing = scales.hor(cross);

 161            this.pos = 'left';

 162            if (xCrossing < margins.left) {

 163                xCrossing = margins.left;

 164                this.pos = 'right';

 165            } else if (xCrossing > this.cfg.viewPort.width-margins.right) {

 166                xCrossing = this.cfg.viewPort.width-margins.right;

 167            }

 168            const dy = scales.ver.type()==='ordinal'? scales.ver.paddingOuter()*scales.ver.step()/2 : 0;

 169            this.svg.transition(trans).attr("transform", `translate(${xCrossing}, ${dy})`);

 170        }

 171    }

 173    getD3Axis(scales:ScaleDims, axisDef:AxisDefaults) {

 174        let axis:d3.Axis;

 175        if (this.dir===Direction.horizontal) {

 176            axis = scales.hor.axis(this.pos);

 177            if (scales.hor.type() === 'number') { axis.tickFormat(format('~g')); }

 178        } else {

 179            axis = scales.ver.axis(this.pos);

 180            if (scales.ver.type() === 'number') { axis.tickFormat(format('~g')); }

 181        }

 182        axis.tickSize(axisDef.tickWidth);

 183        return axis;

 184    }
