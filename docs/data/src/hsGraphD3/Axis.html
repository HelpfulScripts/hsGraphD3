  73export class Axis {

  74    private dir: Direction;

  75    private cfg: GraphCfg;

  76    private svg: d3Base;

  77

  78    constructor(cfg:GraphCfg, dir:Direction) {

  79        this.cfg = cfg;

  80        this.svg = cfg.baseSVG.select('.axes').append('g');

  81        this.dir = dir;

  82        this.svg.classed(`${dir}Axis`, true);

  83    }

  84

  85    public createDefaults() {

  86        return {

  87            color:      'currentColor',

  88            line:       def.defaultLine(1),

  89            tickWidth:  10,

  90            tickMarks:  def.defaultLine(2),

  91            tickLabel:  def.defaultText()

  92        };

  93    }

  94    

  95    public renderComponent() {

  96        const horScales = this.cfg.scales.hor;

  97        const verScales = this.cfg.scales.ver;

  98        const style = this.cfg.defaults.axes[this.dir];

  99        let axis:any;

 100        const margins = (this.cfg.defaults.scales).margin;

 101        this.cfg.baseSVG.select('.axes')

 102            .attr('color', this.cfg.defaults.axes.color);

 103

 104        setStroke(this.svg, style.line);

 105

 106        if (this.dir===Direction.horizontal) {

 107            axis = axisTop(horScales);

 108            const yCrossing = Math.max(margins.left, Math.min(verScales(0), this.cfg.viewPort.height-margins.right));

 109            this.svg.transition(d3Transition).attr("transform", `translate(0, ${yCrossing})`);

 110        } else {

 111            axis = axisRight(verScales);

 112            const xCrossing = Math.max(margins.top, Math.min(horScales(0), this.cfg.viewPort.width-margins.bottom));

 113            this.svg.transition(d3Transition).attr("transform", `translate(${xCrossing}, 0)`);

 114        }

 115        axis.tickSize(style.tickWidth);

 116        setText(this.svg, style.tickLabel);

 117        this.svg.attr('color', style.color);

 118        this.svg.transition(d3Transition).call(axis);

 119        setStroke(this.svg.selectAll('text'), style.tickLabel);

 120    }
