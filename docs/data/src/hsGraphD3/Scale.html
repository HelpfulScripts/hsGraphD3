 156/**

 157 * Manages the embedding of scales into the graph (margins, etc.) and provides

 158 * a configuration for each scales used in the graph.

 159 */

 160export class Scales extends GraphComponent {

 161    static type = 'scales';

 162

 163    private scales: { [name:string]: Scale; } = {};

 164

 165    constructor(cfg:GraphCfg) { super(cfg, null); }

 166

 167    public get componentType() { return Scales.type; }

 168    public get defaults():ScalesDefaults { return this.cfg.graph.defaults.scales; }

 169    public get scaleDims() { return this.scales; }

 170

 171    public initialize(): void {} 

 172    public preRender(): void {} 

 173    public renderComponent() {}

 174    public postRender(): void {} 

 175

 176    /** creates a default entry for the component type in `Defaults` */

 177    public createDefaults():ScalesDefaults {

 178        return {

 179            margin: { left:20, top:20, right:20, bottom:30},

 180            dims: {}

 181        };

 182    }

 184    /**

 185     * creates a d3 scale object based on the provided settings.

 186     * @param name name of the new scale, must also match the name of a predefined scale default.

 187     * @param domain the data domain to scale for

 188     * @param range the viewport range to scale for 

 189     */

 190    public createScale(name:string, domain: Domain, range?:Range):Scale {

 191        const scaleDef: ScaleDefaults = this.defaults.dims[name];

 192        if (!scaleDef) { 

 193            log.warn(`can't create scale ${name}; noe default available`);

 194            return; 

 195        }

 196        let scales = {

 197            ordinal:    BandScale,

 198            time:       TimeScale,

 199            log:        LogScale,

 200            linear:     LinearScale,

 201        };

 202        return this.scales[name] = new scales[scaleDef.type](scaleDef, domain, range).getScale();

 203    }

 204}

 214abstract class BaseScale {

 215    constructor(protected d3Scale:any, protected scaleDef: ScaleDefaults, protected domain:Domain, protected range?:Range) {}

 216

 217    getScale():Scale {

 218        function _range():Range;

 219        function _range(r: Range): Scale;

 220        function _range(r?:Range): any { 

 221            if (r) { 

 222                d3Scale.range(r); 

 223                return scale;

 224            } else {

 225                return d3Scale.range(); 

 226            }

 227        }

 228

 229        function _domain(): Domain;

 230        function _domain(d: Domain): Scale;

 231        function _domain(d?:any):any { 

 232            if (d) {

 233                d3Scale.domain(d);

 234                return scale;

 235            } else {

 236                return d3Scale.domain(); 

 237            }

 238        }

 239        

 240        const d3Scale = this.d3Scale;

 241        const scale:Scale = (x:DataVal) => d3Scale(x);

 242    

 243        scale.range = _range;

 244        scale.domain = _domain;

 245        scale.copy = ():Scale => d3Scale.copy();

 246        scale.axis = (loc:string):d3.Axis => axes[loc](d3Scale);

 247        scale.ticks = this.getTicks();

 248        scale.tickCountMajor = 2;

 249        scale.tickCountMinor = 10;

 250

 251        scale.domain(this.getDomain());

 252        scale.range(this.getRange());

 253        scale.type = this.getType;

 254        return scale;

 255    }

 256

 257    protected abstract getType():string; 

 258

 259    protected getTicks() { return (count:number) => this.d3Scale.ticks(count); }

 260

 261    protected getDomain():Domain { return this.d3Scale.domain(); }

 262    

 263    getRange():Range {

 264        const rangeDef = this.scaleDef.range;

 265        return [

 266            (this.range && rangeDef.min === 'auto')? this.range[0] : rangeDef.min, 

 267            (this.range && rangeDef.max === 'auto')? this.range[1] : rangeDef.max

 268        ];

 269    }

 270}

 272class BandScale extends BaseScale {

 273    constructor(protected scaleDef: ScaleDefaults, protected domain:Domain, protected range?:Range) { 

 274        super(d3.scaleBand(), scaleDef, domain, range); 

 275    }

 276

 277    getType() { return 'ordinal'; }

 278

 279    getScale():Scale {

 280        const scale = super.getScale();

 281        scale.bandwidth     = this.d3Scale.bandwidth;

 282        scale.padding       = this.d3Scale.padding;

 283        scale.paddingInner  = this.d3Scale.paddingInner;

 284        scale.paddingOuter  = this.d3Scale.paddingOuter;

 285        scale.step = this.d3Scale.step;

 286        return scale;

 287    }

 288

 289    protected getTicks() { 

 290        return () => this.d3Scale.domain(); 

 291    }

 292

 293    protected getDomain():Domain { 

 294        return this.domain || []; 

 295    }

 296}
