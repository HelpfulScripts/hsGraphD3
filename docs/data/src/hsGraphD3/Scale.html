  70    /**

  71     * creates a d3 scale object based on the provided settings.

  72     * @param scaleDef scale defaults, possibly modified by the application

  73     * @param domain the data domain to scale for

  74     * @param range the viewport range to scale for 

  75     */

  76    public static createScale(scaleDef: ScaleDefaults, domain: [number, number], range?:[UnitVp, UnitVp]):NumberScale {

  77        if (!scaleDef) { return; }

  78        const domDef = scaleDef.domain;

  79        const rangeDef = scaleDef.range;

  80        let scale:d3.ScaleLinear;

  81        switch(scaleDef.type) {

  82            case 'linear':

  83            default:

  84                scale = scaleLinear();

  85                    

  86            }

  87        scale.domain([

  88            domDef.min === 'auto'? domain[0] : domDef.min,

  89            domDef.max === 'auto'? domain[1] : domDef.max

  90        ])

  91        .range([

  92            (range && rangeDef.min === 'auto')? range[0] : rangeDef.min, 

  93            (range && rangeDef.max === 'auto')? range[1] : rangeDef.max

  94        ])

  95        .interpolate(interpolateRound);

  96        return scale;

  97    }
