  48/**

  49 * The `SeriesDimensions` that specify the values to use for different 

  50 * semantic dimension (e.g. 'x' for the x-axis) of 

  51 * each {@link SeriesPlot `SeriesPlot`}.

  52 * For each dimension, the value can be either 

  53 * - a string that identifies the column name in the data set to use

  54 * - or a number constant 

  55 */

  56export interface SeriesDimensions { [dim:string]: ValueDef; }

  61function expandNumDomain(dataSet:DataSet, domain:NumDomain, col:ValueDef, dim:string):NumDomain {

  62    if (!domain) { 

  63        log.debug(`num domain not initialized for dimension '${dim}'`); 

  64        domain = [1e99, -1e99]; 

  65    }

  66

  67    if (typeof(col)==='function') {

  68        const dataDom = extent(dataSet.rows, ((r, i:number) => (col)(i)));

  69        domain[0] = Math.min(domain[0], dataDom[0]);

  70        domain[1] = Math.max(domain[1], dataDom[1]);

  71    } else {

  72        const colNum = dataSet.colNames.indexOf(col);

  73        if (colNum < 0) { log.warn(`did not find column '${col}' in data set ${dataSet.colNames.join(', ')}`); }

  74        else {

  75            const dataDom = extent(dataSet.rows, (r => r[colNum]));

  76            domain[0] = Math.min(domain[0], dataDom[0]);

  77            domain[1] = Math.max(domain[1], dataDom[1]);

  78            if (domain[1] === domain[0]) { 

  79                domain[0] = 0.9*domain[0]; 

  80                domain[1] = 1.1*domain[1];

  81            }

  82        }

  83    }

  84    return domain;

  85}

  87function expandOrdinalDomain(dataSet:DataSet, domain:OrdDomain, col:ValueDef, dim:string):OrdDomain {

  88    if (!domain) { 

  89        log.debug(`ord domain not initialized for dimension '${dim}'`); 

  90        domain = []; 

  91    }

  92    if (typeof(col)==='function') {

  93        dataSet.rows.forEach((d,i) => {

  94            const dataDom = col(i);

  95            if (domain.indexOf(dataDom) < 0) { domain.push(dataDom); }

  96        });

  97    } else {

  98        const colNum = dataSet.colNames.indexOf(col);

  99        if (colNum < 0) { log.warn(`did not find column '${col}' in data set ${dataSet.colNames.join(', ')}`); }

 100        else {

 101            dataSet.rows.forEach((d,i) => {

 102                const dataDom = d[colNum];

 103                if (domain.indexOf(dataDom) < 0) { domain.push(dataDom); }

 104            });

 105       }

 106    }

 107    return domain;

 108}

 110function stretchDomains(dataSet:DataSet, s:SeriesPlot, domains:Domains, defs:ScalesDefaults) {

 111    const dims:GraphDimensions = s.dimensions;

 112    Object.keys(dims).map(dim => {

 113        const type = defs.dims[dim].type;

 114        log.debug(`stretchDomain ${dim}: ${type}`);

 115        dims[dim].map(col => { if (col!==undefined) { 

 116            switch(type) {

 117                case 'ordinal':     domains[dim] = expandOrdinalDomain(dataSet, domains[dim], col, dim); break;

 118                default:            domains[dim] = expandNumDomain(dataSet, domains[dim], col, dim);

 119            }

 120        }});

 121    });

 122    // log.info(log.inspect(domains, null));

 123}

 126export class Series extends GraphComponent {

 127    static type = 'series';

 128

 129    /*------------ Static implementation----- */

 130    /** a map of plot types to corresponding plot functions. New plot types are added via a call to `register`. */

 131    protected static seriesCreatorMap: {[plotKey:string]: PlotFactory} = {};

 132

 133    /** registers a new plot type with corresponding function. */

 134    public static register(key:string, seriesCreator:PlotFactory) {

 135        if (this.seriesCreatorMap[key]) {

 136            this.seriesCreatorMap[key] = seriesCreator;

 137            log.debug(`re-registered plot type '${key}'`);    

 138        } else {

 139            this.seriesCreatorMap[key] = seriesCreator;

 140            log.debug(`registered plot type '${key}'`);

 141        }

 142    }

 143

 144    public static get types():string[] { return Object.keys(this.seriesCreatorMap); }

 145

 146    /*------------ Instance implementation----- */

 147    private series:SeriesPlot[] = [];

 148

 149    constructor(cfg:GraphCfg) {

 150        super(cfg, Series.type);

 151    }

 152

 153    /** returns the component type as a string name */

 154    get componentType() { return Series.type; }

 155

 156    initialize(svg:d3Base): void {

 157        // swap 0 <-> 1, 2 <-> 3, etc.

 158        // const colorSwap = (i:number) => ((i+1)%2)+ (Math.floor(i/2+0.001))*2;

 159        const seriesSVG = svg.selectAll(`.${this.componentType}`);

 160        this.series.forEach((s:SeriesPlot, i:number) => s.initialize(seriesSVG, colors[i % colors.length]));

 161    } 

 162

 163    preRender(data:DataSet | DataSet[], domains:Domains): void {

 164        this.series.forEach((s:SeriesPlot, i:number) => s.preRender((data).colNames? 

 165                data : data[i % this.series.length], 

 166                domains

 167            ) 

 168        );

 169    } 

 170

 171    /** 

 172     * renders the component for the given data.

 173     * If `data` is an array of `DataSets`, each data set will be used to plot a different registered series, 

 174     * in the order they were regeistered.

 175     */

 176    renderComponent(data:DataSet | DataSet[]) {

 177        this.series.forEach((s:SeriesPlot, i:number) => s.renderComponent((data).colNames? 

 178            data : data[i % this.series.length]

 179        ));

 180    }

 181

 182    /** creates a default entry for the component type in `Defaults` */

 183    createDefaults():ComponentDefaults {

 184        return {};

 185    }

 186

 187    /** 

 188     * returns the data domains by data columns across all added series. 

 189     * @param data the data to calculate domains on

 190     * @param domains the current data domains will be accumulated to the 

 191     * provided one under the same data column index. In addition, if domains for the 

 192     * numeric indices `0`, `1`, ... exist, they will be used to merge and initialize 

 193     * all series' data columns in the corresponding index. This way, if series use different 

 194     * data columns to be plotted on the same axis, they all share the same domain accumulation.

 195     * @return an array of [min, max] domains ranges, indexed by data column

 196     */

 197    expandDomain(data:DataSet | DataSet[], domains:Domains):Domains {

 198        if ((data).colNames) {  

 199            // use same dataset for each series

 200            this.series.forEach((s:SeriesPlot) => stretchDomains(data, s, domains, this.cfg.defaults.scales));

 201        } else {

 202            // assign dataset to series based on index

 203            this.series.forEach((s:SeriesPlot, i:number) => {

 204                const dataSet = data[i % (data).length];

 205                stretchDomains(dataSet, s, domains, this.cfg.defaults.scales);

 206            });

 207        }

 208        return domains;

 209    }

 210    

 211    /**

 212     * adds a series to the plot.

 213     * @param type type of plot to use, e.g. 'bubble' or 'scatter'

 214     * @param params the column name of the parameters used to plot the series

 215     */

 216    addSeries(type:string, dims:SeriesDimensions):SeriesPlot {

 217        const seriesCreator = Series.seriesCreatorMap[type];

 218        if (seriesCreator) {

 219            const series = seriesCreator(this.cfg, `${Series.type}${this.series.length}`, dims);

 220            const seriesDefault = this.cfg.defaults.series;

 221            const index = this.series.length;

 222            seriesDefault[index] = seriesDefault[series.key] = series.getDefaults();

 223            this.series.push(series);

 224            log.debug(`added series ${index} on '${log.inspect(dims, null)}'`);

 225            return series;

 226        } else {

 227            log.error(`unknown plot type ${type}; available types are:\n   '${Object.keys(Series.seriesCreatorMap).join("'\n   '")}'`);

 228        }

 229    }
