  53/**

  54 * A class, instantiated as a primary `Graph` component and proxy for all added series.

  55 * `Seires` manages available series types, as well as the specific series added to a `Graph`,

  56 * including their render lifecycle and auto-domaining.

  57 */

  58export class Series extends GraphComponent {

  59    static type = 'series';

  60

  61    /*------------ Static implementation----- */

  62    /** a map of plot types to corresponding plot functions. New plot types are added via a call to `register`. */

  63    protected static seriesCreatorMap: {[plotKey:string]: PlotFactory} = {};

  64

  65    /** registers a new plot type with corresponding function. */

  66    public static register(key:string, seriesCreator:PlotFactory) {

  67        if (this.seriesCreatorMap[key]) {

  68            this.seriesCreatorMap[key] = seriesCreator;

  69            log.debug(`re-registered plot type '${key}'`);    

  70        } else {

  71            this.seriesCreatorMap[key] = seriesCreator;

  72            log.debug(`registered plot type '${key}'`);

  73        }

  74    }

  75

  76    public static get types():string[] { return Object.keys(this.seriesCreatorMap); }

  77

  78    /*------------ Instance implementation----- */

  79    private series:SeriesPlot[] = [];

  80

  81    constructor(cfg:GraphCfg) {

  82        super(cfg, Series.type);

  83    }

  84

  85    /** returns the component type as a string name */

  86    get componentType() { return Series.type; }

  87

  88    public get defaults():SeriesDefaults { return this.cfg.graph.defaults.series; }

  89    

  90

  91    public initialize(svg:d3Base): void {

  92        const seriesSVG = svg.selectAll(`.${this.componentType}`);

  93        this.series.forEach((s:SeriesPlot, i:number) => s.initialize(seriesSVG, defaultColors[i % defaultColors.length]));

  94    } 

  95

  96    public preRender(data:DataSet | DataSet[], domains:Domains): void {

  97        this.series.forEach((s:SeriesPlot, i:number) => 

  98            s.preRender((data).colNames? data : data[i % this.series.length], domains) 

  99        );

 100    } 

 101

 102    /** 

 103     * renders the component for the given data.

 104     * If `data` is an array of `DataSets`, each data set will be used to plot a different registered series, 

 105     * in the order they were regeistered.

 106     */

 107    public renderComponent(data:DataSet | DataSet[]) {

 108        this.series.forEach((s:SeriesPlot, i:number) => { 

 109            s.renderComponent((data).colNames? data : data[i % this.series.length]);

 110        });

 111    }

 112

 113    public postRender(data:DataSet | DataSet[]) {

 114        this.series.forEach((s:SeriesPlot, i:number) => s.postRender((data).colNames? 

 115            data : data[i % this.series.length]

 116        ));

 117    }

 118

 119    /** creates a default entry for the component type in `Defaults` */

 120    public createDefaults():SeriesDefaults {

 121        return {};

 122    }

 123

 124    /** 

 125     * Called during preRendering, prior to all component preRender calls.

 126     * Returns the data domains by data columns across all added series. 

 127     * @param data the data to calculate domains on

 128     * @param domains the current data domains will be accumulated to the 

 129     * provided one under the same data column index. In addition, if domains for the 

 130     * numeric indices `0`, `1`, ... exist, they will be used to merge and initialize 

 131     * all series' data columns in the corresponding index. This way, if series use different 

 132     * data columns to be plotted on the same axis, they all share the same domain accumulation.

 133     * @return an array of [min, max] domains ranges, indexed by data column

 134     */

 135    expandDomains(data:DataSet | DataSet[], domains:Domains):Domains {

 136        if ((data).colNames) {  

 137            // use same dataset for each series

 138            this.series.forEach((s:SeriesPlot) => s.clearStack(data));

 139            this.series.forEach((s:SeriesPlot) => s.expandDomains(data, domains));

 140        } else {

 141            // assign dataset to series based on index

 142            this.series.forEach((s:SeriesPlot, i:number) => {

 143                const dataSet = data[i % (data).length];

 144                s.clearStack(dataSet);

 145            });

 146            this.series.forEach((s:SeriesPlot, i:number) => {

 147                const dataSet = data[i % (data).length];

 148                s.expandDomains(dataSet, domains);

 149            });

 150        }

 151        return domains;

 152    }

 154    /**

 155     * adds a series to the plot, for example

 156     * ```

 157     * graph.series.add('area', {x:'time', y:'costs', r:5})

 158     * ```

 159     * The object literal `dims` specifies the data to use for each 

 160     * semantic dimension the plot uses. For details on the dimensions 

 161     * see {@link Series.SeriesDimensions `SeriesDimensions`}

 162     * @param type type of plot to use, e.g. 'bubble' or 'scatter', See {@link Series `Series`} for available plots to use.

 163     * @param dims an object literal specifying the {@link Series.SeriesDimensions `SeriesDimensions`} to use. 

 164     */

 165    add(type:string, dims:SeriesDimensions):SeriesPlot {

 166        const seriesCreator = Series.seriesCreatorMap[type];

 167        if (seriesCreator) {

 168            const series = seriesCreator(this.cfg, `${Series.type}${this.series.length}`, dims);

 169            const index = this.series.length;

 170            this.defaults[index] = this.defaults[series.key] = series.getDefaults();

 171            this.series.push(series);

 172            log.debug(`added series ${index} on '${log.inspect(dims, null)}'`);

 173            return series;

 174        } else {

 175            log.error(`unknown plot type ${type}; available types are:\n   '${Object.keys(Series.seriesCreatorMap).join("'\n   '")}'`);

 176        }

 177    }
