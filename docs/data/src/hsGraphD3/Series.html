  34import { Data }             from 'hsdatab';

  35import { NumDomain }        from 'hsdatab';

  36import { log as gLog }      from 'hsutil';   const log = gLog('Series');

  37import { d3Base }           from './Settings';

  38import { GraphComponent }   from './GraphComponent'; 

  39import { ComponentDefaults }from './GraphComponent'; 

  40import { GraphCfg }         from './GraphComponent'; 

  41import { SeriesPlot }       from './SeriesPlot';

  43

  44type PlotFactory = (cfg:GraphCfg, seriesName:string, ...params:string[]) => SeriesPlot;

  45

  46export class Series extends GraphComponent {

  47    static type = 'series';

  48

  49    /*------------ Static implementation----- */

  50    /** a map of plot types to corresponding plot functions. New plot types are added via a call to `register`. */

  51    protected static seriesCreatorMap: {[plotKey:string]: PlotFactory} = {};

  52

  53    /** registers a new plot type with corresponding function. */

  54    public static register(key:string, seriesCreator:PlotFactory) {

  55        if (this.seriesCreatorMap[key]) {

  56            this.seriesCreatorMap[key] = seriesCreator;

  57            log.debug(`re-registered plot type '${key}'`);    

  58        } else {

  59            this.seriesCreatorMap[key] = seriesCreator;

  60            log.debug(`registered plot type '${key}'`);

  61        }

  62    }

  63

  64    public static get types():string[] { return Object.keys(this.seriesCreatorMap); }

  65

  66    /*------------ Instance implementation----- */

  67    private series:SeriesPlot[] = [];

  68    private domains: {[dim:string]: NumDomain};

  69

  70    constructor(cfg:GraphCfg) {

  71        super(cfg, Series.type);

  72    }

  73

  74    /** returns the component type as a string name */

  75    get componentType() { return Series.type; }

  76

  77    initialize(svg:d3Base): void {

  78        this.series.forEach((s:SeriesPlot) => s.initialize(svg));

  79    } 

  80

  81    preRender(data:Data): void {

  82        this.series.forEach((s:SeriesPlot) => s.preRender(data, this.domains));

  83    } 

  84

  85    /** renders the component for the given data */

  86    renderComponent(data:Data) {

  87        this.series.forEach((s:SeriesPlot) => s.renderComponent(data));

  88    }

  89

  90    /** creates a default entry for the component type in `Defaults` */

  91    createDefaults():ComponentDefaults {

  92        return {};

  93    }

  94

  95    /** 

  96     * returns the data domains by data columns across all added series. 

  97     * @param data the data to calculate domains on

  98     * @param domains optional; if present, the current data domains will be accumulateds to the 

  99     * provided one under the same data column index. In addition, if domains for the 

 100     * numeric indices `0`, `1`, ... exist, they will be used to merge and initialize 

 101     * all series' data columns in the corresponding index. This way, if series use different 

 102     * data columns to be plotted on the same axis, they all share the same domain accumulation.

 103     * @return an array of [min, max] domains ranges, indexed by data column

 104     */

 105    expandDomain(data:Data, domains:{[dim:string]: NumDomain} = {}):{[dim:string]: NumDomain} {

 106        this.series.forEach(s => {

 107            s.dimensions.map((dim, i) => {

 108                const dataDom:NumDomain = data.findDomain(dim);

 109                if (!domains[i]) { domains[i] = [1e90, -1e90]; }

 110                if (!domains[dim]) { domains[dim] = domains[i] || [1e90, -1e90]; }

 111                domains[dim][0] = Math.min(domains[dim][0], dataDom[0]);

 112                domains[dim][1] = Math.max(domains[dim][1], dataDom[1]);

 113            });

 114        });

 115        return this.domains = domains;

 116    }

 117

 118    

 119    /**

 120     * adds a series to the plot.

 121     * @param type type of plot to use, e.g. 'bubble' or 'scatter'

 122     * @param params the column name of the parameters used to plot the series

 123     */

 124    addSeries(type:string, ...params:string[]) {

 125        const seriesCreator = Series.seriesCreatorMap[type];

 126        if (seriesCreator) {

 127            const series = seriesCreator(this.cfg, `${Series.type}${this.series.length}`, ...params);

 128            const seriesDefault = this.cfg.defaults.series;

 129            const index = this.series.length;

 130            seriesDefault[index] = seriesDefault[series.key] = series.getDefaults();

 131            this.series.push(series);

 132            log.debug(`added series ${index} on '${params}'`);

 133        } else {

 134            log.error(`unknown plot type ${type}; available types are:\n   '${Object.keys(Series.seriesCreatorMap).join("'\n   '")}'`);

 135        }

 136    }
