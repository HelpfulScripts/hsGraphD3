  36/**

  37 * ## Lifecycle calls

  38 * All `GraphComponents` implement these lifecycle methods that are called by the `Graph`:

  39 * - **initialize**: called the first time the `Graph` is rendered, ahead of any other lifecycle method.

  40 * - **preRender**: called each time the `Graph` is rendered, ahead of any other recurring lifecycle method.

  41 * Implementations can assume that 

  42 *     - `initialize` has been called on all `GraphComponents`.

  43 *     - the main axes scales have been properly configured to the data.

  44 * - **renderComponent**: called each time the `Graph` is rendered, to render the component. 

  45 * Implementations can assume that `preRender` has been called on all `GraphComponents`.

  46 */

  47export interface LifecycleCalls {

  48    /** 

  49     * called the first time the `Graph` is rendered, before any of the other lifecycle methods. 

  50     * At this point, all components have been created and the tree-wide transition has been set.

  51     */

  52    initialize(svg:d3Base):void;

  53

  54    /** 

  55     * Called immediately before each call to renderComponent. 

  56     * At this point, the axis scales have been set and all components are initialized.

  57     */

  58    preRender(data:DataSet, domains:Domains): void;

  59

  60    /** 

  61     * renders the component. 

  62     * At this point all components have completed preRendering

  63     */

  64    renderComponent(data:DataSet): void;

  65}

 139    /**

 140     * render the tree with the supplied data.

 141     * @param data the data to render

 142     * @param duration optional period in ms to call `sample` function. the `Graph's` tree-wide transition

 143     * will be timed to end at the specified duration, upon which the `sample` callback will be called.

 144     * @param update optional callback function that allow periodic manipulation of the `data` set. 

 145     * New callbacks will be scheduled at `duration` intervals as long as the callback returns `true`. 

 146     * Otherwise the callbacks will be stopped. Independent of the return value, 

 147     * a render cycle will automatically be triggered after `update` returns.

 148     * See {@link TimeSeries TimeSeries for an example.

 149     */

 150    public render(data:DataSet, duration?:number, update?:(data:DataSet) => boolean): void {

 151        const graph = this;

 152        const graphDef = graph.config.defaults.graph;

 153

 154        function listener() {

 155            if (update(data)) { 

 156                graph.config.transition = graph.config.transition.transition();

 157                graph.config.transition.on('end', listener); 

 158            }

 159            setTimeout(renderGraph,0);  // render on the next event loop pass, outside the transition listener.

 160        }

 161        function renderGraph() {

 162            if (!graph.initialized) {

 163                graph.initialize(graph.config.baseSVG);

 164                graph.initialized = true;

 165            }

 166            const scalesDefaults = graph.config.defaults.scales;

 167            graph.preRender.bind(graph)(data, graph.prepareDomains(scalesDefaults));

 168            graph.renderComponent.bind(graph)(data);

 169        }

 170        

 171        graphDef.transitionTime = duration || graphDef.transitionTime;

 172        const easing = d3[graphDef.easing];

 173        // if (!graph.config.transition) {          

 174            graph.config.transition = graph.config.baseSVG.transition().duration(graphDef.transitionTime).ease(easing);

 175            if (update) {

 176                graph.config.transition.on('end', update? listener : null); 

 177            }

 178        // }

 179        renderGraph();

 180    }

 203    /** Called immediately before each call to renderComponent. */

 204    preRender(data:DataSet, domains:Domains): void {

 205        this.seriesComponent.expandDomain(data, domains);

 206        this.setScales(data);

 207        this.components.forEach((comp:GraphComponent) => comp.preRender(data, domains));

 211    renderComponent(data:DataSet): void {

 212        this.components.forEach((comp:GraphComponent) => comp.renderComponent(data));

 213    }

 237    /** creates and initializes the `Graph`-wide configuration object. */

 238    private initializeCfg():GraphCfg {

 239        return {

 240            baseSVG:  undefined,    

 241            client:   { x:0, y:0, width: 0, height: 0 },

 242            viewPort: {

 243                width: vpWidth,

 244                height: vpWidth * 0.7   // initial height: 70% of width

 245            },

 246            defaults: {},

 247            scales: {},

 248            transition: null

 249        };

 250    }

 252    /** creates the list of `GraphComponents` and determines the rendering order. */

 253    private createComponents(cfg:GraphCfg):GraphComponent[] {

 254        return [

 255            new Scales(cfg),

 256            new Canvas(cfg),

 257            new Grids(cfg),

 258            new Axes(cfg),

 259            this.seriesComponent = new Series(cfg)

 260        ];

 261    }

 263    /** callback on window resize event, adjusts the viewport to the new dimensions  */

 264    private resize() {

 265        const cfg = this.config;

 266        // if (this.root && this.root.clientWidth > 0) {

 267        if (this.root) {

 268            if (this.root.clientWidth !== cfg.client.width || this.root.clientHeight !== cfg.client.height) {

 269                log.debug(`resizing svg: [${cfg.client.width} x ${cfg.client.height}] -> [${this.root.clientWidth} x ${this.root.clientHeight}]`);

 270                cfg.client.width = this.root.clientWidth;

 271                cfg.client.height = this.root.clientHeight;

 272                cfg.viewPort.height = cfg.viewPort.width * this.root.clientHeight / this.root.clientWidth;

 273                this.updateBaseSVG(cfg);

 274            }

 275        }

 276    }

 278    /**

 279     * creates the base SVG element for the entire graph.

 280     * This function is intended to be called only once at creation of the graph.

 281     * @param cfg 

 282     */

 283    private createBaseSVG(cfg: GraphCfg):d3Base {

 284        return d3Select(this.root).append('svg')

 285            .classed('baseSVG', true)

 286            .attr('height', '100%')

 287            .attr('width', '100%')

 288            .attr('preserveAspectRatio', 'xMinYMin meet');

 289    }

 291    /**

 292     * sets the graph's viewBox, typically after a window resize.

 293     * @param cfg 

 294     */

 295    private updateBaseSVG(cfg: GraphCfg) {

 296        cfg.baseSVG.attr('viewBox', `0 0 ${cfg.viewPort.width} ${cfg.viewPort.height}`);

 297    }
