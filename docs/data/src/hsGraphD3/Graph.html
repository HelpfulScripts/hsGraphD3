  38/**

  39 * Function interface, describing the signature of the call back function 

  40 * in {@link RenderChain `RenderChain`}.

  41 * If the function returns `false`, no further callbacks will be initiated

  42 */

  43export interface RenderCallback {

  44    (data:DataSet | DataSet[]): boolean;

  45}

  46

  47/**

  48 * Function interface, describing the signature of the `update` function

  49 * @param duration optional period in ms to call `sample` function. the `Graph's` tree-wide transition

  50 * will be timed to end at the specified duration, upon which the `sample` callback will be called.

  51 * @param callback callback function that allow periodic manipulation of the `data` set. 

  52 * New callbacks will be scheduled at `duration` intervals as long as the callback returns `true`. 

  53 * Otherwise the callbacks will be stopped. Independent of the return value, 

  54 * a render cycle will automatically be triggered after `update` returns.

  55 * See {@link TimeSeries TimeSeries} for an example.

  56 */

  57export interface Update {

  58    (duration:number, callback:RenderCallback): void;

  59}

  60

  61/**

  62 * This object is returned by a call to {@link Graph.render `Graph.render`}, providing an `update` function

  63 * to call in order to render dynamic data updates.

  64 */

  65export interface RenderChain {

  66    /** 

  67     * Update function, can be called to dynamically update the data and trigger a redraw in regular intervals.

  68     * Once called, the `callback` function will be invoked in roughly regular intervals every `duration` milliseconds.

  69     * Returning `false` in the `callback` will stop the callbacks.

  70     * @param duration optional period in ms to call `sample` function. the `Graph's` tree-wide transition

  71     * will be timed to end at the specified duration, upon which the `sample` callback will be called.

  72     * @param callback callback function that allow periodic manipulation of the `data` set. 

  73     * New callbacks will be scheduled at `duration` intervals as long as the callback returns `true`. 

  74     * Otherwise the callbacks will be stopped. Independent of the return value, 

  75     * a render cycle will automatically be triggered after `update` returns.

  76     * See {@link TimeSeries TimeSeries} for an example.

  77     */

  78    update: (duration:number, callback:RenderCallback) => void;

  79}

  81/**

  82 * ## Lifecycle calls

  83 * All `GraphComponents` implement these lifecycle methods that are called by the `Graph`:

  84 * - **initialize**: called the first time the `Graph` is rendered, ahead of any other lifecycle method.

  85 * - **preRender**: called each time the `Graph` is rendered, ahead of any other recurring lifecycle method.

  86 * Implementations can assume that 

  87 *     - `initialize` has been called on all `GraphComponents`.

  88 *     - the main axes scales have been properly configured to the data.

  89 * - **renderComponent**: called each time the `Graph` is rendered, to render the component. 

  90 * Implementations can assume that `preRender` has been called on all `GraphComponents`.

  91 */

  92export interface LifecycleCalls {

  93    /** 

  94     * called the first time the `Graph` is rendered, before any of the other lifecycle methods. 

  95     * At this point, all components have been created and the tree-wide transition has been set.

  96     */

  97    initialize(svg:d3Base):void;

  98

  99    /** 

 100     * Called immediately before each call to renderComponent. 

 101     * At this point, the axis scales have been set and all components are initialized.

 102     */

 103    preRender(data:DataSet | DataSet[], domains:Domains): void;

 104

 105    /** 

 106     * renders the component. 

 107     * At this point all components have completed preRendering

 108     */

 109    renderComponent(data:DataSet | DataSet[]): void;

 110}

 127/**

 128 * ## Graph

 129 * Abstract base Graph.

 130 */

 131export abstract class Graph implements LifecycleCalls {

 132    /** the HTML root element to attach the render tree to. */

 133    protected root:any;

 134

 135    /** the `GraphCfg` object shared by all components in this graph */

 136    protected config: GraphCfg;

 137

 138    /** the plot component, provides access to individual series components */

 139    protected seriesComponent:Series;

 140

 141    /** the list of components to render */

 142    private components: GraphComponent[] = [];

 143

 144    /** tracks whether components have been initialized. */

 145    private initialized = false;

 146

 147    protected cumulativeDomains: Domains = {};

 148

 149    constructor(root:any) { 

 150        this.root = root;

 151        this.config = this.initializeCfg();

 152        this.config.baseSVG = this.createBaseSVG(this.config); 

 153        this.updateBaseSVG(this.config);

 154        this.components = this.createComponents(this.config);

 155        this.makeDefaults();

 156        this.resize();

 157        window.onresize = () => this.resize();

 158    }

 159

 160    public get defaults(): ComponentDefaults {

 161        return this.config.defaults;

 162    }

 163

 164    public get viewport() {

 165        return this.config.viewPort;

 166    }

 167

 168    /** returns the types of all registered `Series` */

 169    public get series():{types: string[]} {

 170        return {

 171            types: Series.types

 172        };

 173    }

 174

 175    /**

 176     * Sets the `Graphs` transition duration as an alternative to `defaults.graph.transitionTime`.

 177     * If `done` is provided, it will be used as a callback for when the transition ends.

 178     * @param duration 

 179     * @param done 

 180     */

 181    public transition(duration:number, done?:(data:DataSet) =>void) {

 182    }

 183

 184    /**

 185     * render the tree with the supplied data.

 186     * @param data the data to render. If `data` is a {@link DataSet DataSet}, it will be used 

 187     * to render all added series. Otherwise `data` can be specified as `DataSet[]` to provide a

 188     * different data source to each of the series. The series' index, starting with 0 in the order of adding it to the graph,

 189     * is used to index the list. If there are more series than data sets ion the list, indexing will restart at index 0.

 190     */

 191    public render(data:DataSet | DataSet[]):RenderChain {

 192        const graphDef = this.config.defaults.graph;

 193

 194        function renderGraph() {

 195            if (!this.initialized) {

 196                this.initialize(this.config.baseSVG);

 197                this.initialized = true;

 198            }

 199            const scalesDefaults = this.config.defaults.scales;

 200            this.preRender(data, this.prepareDomains(scalesDefaults));

 201            this.renderComponent(data);

 202        }

 203

 204        const easing = d3[graphDef.easing];

 205        this.config.transition = this.config.baseSVG.transition().duration(graphDef.transitionTime).ease(easing);

 206

 207        renderGraph.bind(this)();

 208        

 209        const graph = this;

 210        return {

 211            update: (duration:number, callback:RenderCallback):void => {

 212                function listener() {

 213                    try {

 214                        graphDef.transitionTime = duration || graphDef.transitionTime;

 215                        graph.config.transition = graph.config.transition.transition().duration(graphDef.transitionTime);

 216                        if (callback(data) !== false) {

 217                            graph.config.transition.on('end', listener); 

 218                        }

 219                    }

 220                    catch(e) { log.warn(`error in callback: ${e}`); }

 221                    setTimeout(renderGraph.bind(graph),0);  // render on the next event loop pass, outside the transition listener.

 222                }

 223            graph.config.transition.on('end', listener); 

 224            }

 225        };

 226    }

 227

 228    /**

 229     * adds a series to the plot.

 230     * @param type type of plot to use, e.g. 'bubble' or 'scatter'

 231     * @param params the column name of the parameters used to plot the series

 232     */

 233    public addSeries(type:string, x:string, y:string, ...params:string[]):SeriesPlot {

 234        //this.resize();

 235        return this.seriesComponent.addSeries(type, x, y, ...params);

 236    }

 241    /** 

 242     * Called once at the beginning of the first call to `Graph.render()`

 243     * and initializes all known `GraphComponts`.

 244     */

 245    initialize(svg:d3Base): void {

 246        this.components.forEach(comp => comp.initialize(svg));

 247    } 

 248

 249    /** Called immediately before each call to renderComponent. */

 250    preRender(data:DataSet | DataSet[], domains:Domains): void {

 251        this.seriesComponent.expandDomain(data, domains);

 252        this.setScales(data);

 253        this.components.forEach((comp:GraphComponent) => comp.preRender(data, domains));

 254    } 

 255

 256    /** renders the component. */

 257    renderComponent(data:DataSet | DataSet[]): void {

 258        this.components.forEach((comp:GraphComponent) => comp.renderComponent(data));

 259    }

 267    /** called once during construction to create the components defaults. */

 268    protected createDefaults():GraphDefaults {

 269        return {

 270            transitionTime: 1000,

 271            easing: 'easeCubic'

 272        };    

 273    }

 275    protected makeDefaults() {

 276        const defaults = this.config.defaults;

 277        defaults['graph'] = this.createDefaults();

 278        this.components.forEach(comp => defaults[comp.componentType] = comp.createDefaults());

 279    }

 283    /** creates and initializes the `Graph`-wide configuration object. */

 284    private initializeCfg():GraphCfg {

 285        return {

 286            baseSVG:  undefined,    

 287            client:   { x:0, y:0, width: 0, height: 0 },

 288            viewPort: {

 289                width: vpWidth,

 290                height: vpWidth * 0.7   // initial height: 70% of width

 291            },

 292            defaults: {},

 293            scales: {},

 294            transition: null

 295        };

 296    }

 297

 298    /** creates the list of `GraphComponents` and determines the rendering order. */

 299    private createComponents(cfg:GraphCfg):GraphComponent[] {

 300        return [

 301            new Scales(cfg),

 302            new Canvas(cfg),

 303            new Grids(cfg),

 304            new Axes(cfg),

 305            this.seriesComponent = new Series(cfg)

 306        ];

 307    }

 309    /** callback on window resize event, adjusts the viewport to the new dimensions  */

 310    private resize() {

 311        const cfg = this.config;

 312        // if (this.root && this.root.clientWidth > 0) {

 313        if (this.root) {

 314            if (this.root.clientWidth !== cfg.client.width || this.root.clientHeight !== cfg.client.height) {

 315                log.debug(`resizing svg: [${cfg.client.width} x ${cfg.client.height}] -> [${this.root.clientWidth} x ${this.root.clientHeight}]`);

 316                cfg.client.width = this.root.clientWidth;

 317                cfg.client.height = this.root.clientHeight;

 318                cfg.viewPort.height = cfg.viewPort.width * this.root.clientHeight / this.root.clientWidth;

 319                this.updateBaseSVG(cfg);

 320            }

 321        }

 322    }

 324    /**

 325     * creates the base SVG element for the entire graph.

 326     * This function is intended to be called only once at creation of the graph.

 327     * @param cfg 

 328     */

 329    private createBaseSVG(cfg: GraphCfg):d3Base {

 330        return d3Select(this.root).append('svg')

 331            .classed('baseSVG', true)

 332            .attr('height', '100%')

 333            .attr('width', '100%')

 334            .attr('preserveAspectRatio', 'xMinYMin meet');

 335    }

 336

 337    /**

 338     * sets the graph's viewBox, typically after a window resize.

 339     * @param cfg 

 340     */

 341    private updateBaseSVG(cfg: GraphCfg) {

 342        cfg.baseSVG.attr('viewBox', `0 0 ${cfg.viewPort.width} ${cfg.viewPort.height}`);

 343    }
