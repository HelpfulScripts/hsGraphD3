   2 * # Graph

   3 * 

   4 * The base class for Graph types.

   5 * Takes care of the overall orchestration of the Graph, including the following tasks:

   6 * - setting up the SVG environment for the `Graph`

   7 * - creating a central {@link GraphComponent.GraphCfg configuration object} that is used throughout the library.

   8 * - creating and managing the {@link GraphComponent components} to render in ths `Graph`

   9 * - initializing default {@link Settings settings}

  10 * - establishing {@link Graph.LifecycleCalls lifecycle calls}

  11 * - adding {@link SeriesPlot `series`} to render by calling {@link Graph.Graph.addSeries `addSeries`}

  12 * - rendering the graph by calling {@link Graph.Graph.render `render`}

  13 * - managing a central {@link Graph.Graph.transition `transition`} that applies to all components.   

  14 * 

  15 * ### Graph Management Phases

  16 * Graph Management is dividied into the following phases:

  17 * 1. **Graph creation**: 

  18 * `const graph = new xxxGraph(root);`

  19 *     - creates the graph's {@link GraphComponent components} and their 

  20 *       {@link Graph.Graph.createComponents sequence of rendering} in the DOM

  21 *     - sets the factory {@link Settings default settings} for all components

  22 * 2. **Graph configuration**:

  23 *     - adding series: 

  24 *      `graph.series.add(, {y:'State', x:'costs'});`

  25 *     - configuring defaults: e.g., 

  26 *      `graph.grids.defaults.hor.major.rendered = false;`

  27 * 3. **Graph rendering**: 

  28 * `graph.render()`

  29 *     - sets the graph-wide transition timing and easing

  30 *     - starts a {@link Graph.LifecycleCalls render lifecycle} by calling each of the lifecycle methods on all components:

  31 *         - Once, following graph creation, {@Link Graph.LifecycleCalls.initialize initializes} the components

  32 *         - Calls {@link Graph.LifecycleCalls.preRender preRender} for any regular preparation

  33 *         - Calls {@link Graph.LifecycleCalls.renderComponent renderComponent} to render each component

  34 *     - returns a {@link Graph.RenderChain RenderChain} that can be used to dynamically {@link Graph.RenderChain.update update}

  35 *       the data or settings, which triggers a new render lifecycle.

  36 * 

  37 * ### Graph Default Settings:

 215/**

 216 * Function interface, describing the signature of the `update` function.

 217 * See {@link plots.TimeSeries TimeSeries} for an example.

 218 */

 219export interface Update {

 220    /**

 221     * function definition:

 222     * @param duration optional period in ms to call `sample` function. the `Graph's` tree-wide transition

 223     * will be timed to end at the specified duration, upon which the `sample` callback will be called.

 224     * @param callback callback function that allow periodic manipulation of the `data` set. 

 225     * New callbacks will be scheduled at `duration` intervals as long as the callback returns `true`. 

 226     * Otherwise the callbacks will be stopped. Independent of the return value, 

 227     * a render cycle will automatically be triggered after `update` returns.

 228     */

 229    (duration:number, callback:RenderCallback): void;

 230}

 232/**

 233 * This object is returned by a call to {@link Graph.Graph.render `Graph.render`}, providing an `update` function

 234 * to call in order to render dynamic data updates.

 235 */

 236export interface RenderChain {

 237    /** 

 238     * Update function, can be called to dynamically update the data and trigger a redraw in regular intervals.

 239     * Once called, the `callback` function will be invoked in roughly regular intervals every `duration` milliseconds.

 240     * Returning `false` in the `callback` will stop the callbacks.

 241     * @param duration optional period in ms to call `sample` function. the `Graph's` tree-wide transition

 242     * will be timed to end at the specified duration, upon which the `sample` callback will be called.

 243     * @param callback callback function that allow periodic manipulation of the `data` set. 

 244     * New callbacks will be scheduled at `duration` intervals as long as the callback returns `true`. 

 245     * Otherwise the callbacks will be stopped. Independent of the return value, 

 246     * a render cycle will automatically be triggered after `update` returns.

 247     * See {@link plots.TimeSeries TimeSeries} for an example.

 248     */

 249    update: Update;

 252/**

 253 * All `GraphComponents` implement these lifecycle methods that are called by the `Graph` instance

 254 * in response to a `render` call:

 255 * - **initialize**: called the first time the `Graph` is rendered, ahead of any other lifecycle method.

 256 * During initialization, the component's style settings are applied 

 257 * - **preRender**: called each time the `Graph` is rendered, ahead of any other recurring lifecycle method.

 258 * Implementations can assume that 

 259 *     - `initialize` has been called on all `GraphComponents`.

 260 *     - the main axes scales have been properly configured to the data.

 261 * - **renderComponent**: called each time the `Graph` is rendered, to render the component. 

 262 * Implementations can assume that `preRender` has been called on all `GraphComponents`.

 263 * - **postRender**: called each time the `Graph` is rendered, to render the component. 

 264 * Implementations can assume that `preRender` has been called on all `GraphComponents`.

 265 */

 266export interface LifecycleCalls {

 267    /** 

 268     * called the first time the `Graph` is rendered, before any of the other lifecycle methods. 

 269     * At this point, all components have been created, defaults have been defined, 

 270     * and the tree-wide transition has been set.

 271     */

 272    initialize(svg:d3Base):void;

 273

 274    /** 

 275     * Called immediately before each call to renderComponent. 

 276     * At this point, the axis scales have been set and all components are initialized.

 277     */

 278    preRender(data:DataSet | DataSet[], domains:Domains): void;

 279

 280    /** 

 281     * renders the component. 

 282     * At this point all components have completed preRendering

 283     */

 284    renderComponent(data:DataSet | DataSet[]): void;

 285

 286    /** 

 287     * Called immediately after the call to renderComponent. 

 288     * Can be used for cleanup operations.

 289     */

 290    postRender(data:DataSet | DataSet[], domains:Domains): void;

 291}

 408    constructor(root:HTMLElement|Vnode) {

 409        super(initializeCfg());

 410        this.cfg.graph = this;

 411        if (root['attrs'] && root['attrs'].rootID) {

 417    }

 428        }

 429    }

 454    /**

 455     * renders the tree with the supplied data. The call returns a {@link Graph.RenderChain `RenderChain`}

 456     * function that allows rendering to be repeated at fixed intervals with updated data. 

 457     * @param data the data to render. If `data` is a {@link Graph.DataSet DataSet}, it will be used 

 458     * to render all added series. Otherwise `data` can be specified as `DataSet[]` to provide a

 459     * different data source to each of the series. The series' index is used to index the list,

 460     * starting with 0 in the order of adding it to the graph. 

 461     * If there are more series than data sets ion the list, indexing will restart at index 0.

 462     * @return a `RenderChain`.

 463     */

 464    public render(data:DataSet | DataSet[]):RenderChain {

 465        const transitionDef = this.defaults.graph.transition;

 466        const graph = this;

 467

 468        const easing = easings[transitionDef.easing];

 469        this.cfg.transition = this.cfg.baseSVG.transition().duration(transitionDef.duration).ease(easing);

 470

 471        Graph.addGraph(graph.root.id, () => {   // called upon resize

 472            setTimeout(() => {

 473                graph.resize();

 474                graph.renderLifecycle(data);

 475            }, 0);

 476        });

 477        graph.renderLifecycle(data);

 478        

 479        return { update: graph.updateFn(data)};

 480    }

 549    renderLifecycle(data:DataSet | DataSet[]) {

 550        const isRendered = this.isRendered();

 551        if (isRendered) {

 553                this.initialize(this.cfg.baseSVG);

 554                this.initialized = true;

 555            }

 556            const scalesDefaults = this.defaults['scales'];

 557            this.preRender(data, this.prepareDomains(scalesDefaults));

 558            this.renderComponent(data);

 559            this.postRender(data);

 560        } else {

 571    private updateFn(data:DataSet | DataSet[]) {

 572        const graphDef = this.defaults.graph;

 573        const graph = this;

 574        return (duration:number, callback?:RenderCallback):void => {

 575            function listener() {

 576                try {

 577                    graphDef.transition.duration = duration || graphDef.transition.duration;

 578                    graph.cfg.transition = graph.cfg.transition.transition().duration(graphDef.transition.duration);

 579                    

 580                    // set new transition if a) still rendered, b) callback is missing, or c) callback returns undefined or truthy

 581                    if (graph.isRendered() && (!callback || callback(data) !== false)) {

 582                        graph.cfg.transition.on('end', listener); 

 583                    }

 584                }

 585                catch(e) { log.warn(`error in callback: ${e}`); }

 586                setTimeout(() => graph.renderLifecycle(data),0);  // render on the next event loop pass, outside the transition listener.

 587            }

 588            graph.cfg.transition.on('end', listener); 

 589        };

 590    }

 623    /** callback on window resize event, adjusts the viewport to the new dimensions  */

 624    private resize() {

 625        const client = this.cfg.client;

 626        const vp = this.cfg.viewPort;

 627        // if (this.root && this.root.clientWidth > 0) {

 628        if (this.root) {

 629            if (this.root.clientWidth !== client.width || this.root.clientHeight !== client.height) {

 630                log.debug(`resizing svg for ${this.root.id}: [${client.width} x ${client.height}] -> [${this.root.clientWidth} x ${this.root.clientHeight}]`);

 631                client.width = this.root.clientWidth;

 632                client.height = this.root.clientHeight;

 633                vp.width = 2*this.root.clientWidth;

 634                vp.height = vp.width * this.root.clientHeight / this.root.clientWidth;

 635                this.updateBaseSVG(this.cfg);

 636            }

 637        }
