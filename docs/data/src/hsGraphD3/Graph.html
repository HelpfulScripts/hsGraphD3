  33/**

  34 * ## Lifecycle calls

  35 * All `GraphComponents` implement these lifecycle methods that are called by the `Graph`:

  36 * - **initialize**: called the first time the `Graph` is rendered, ahead of any other lifecycle method.

  37 * - **preRender**: called each time the `Graph` is rendered, ahead of any other recurring lifecycle method.

  38 * Implementations can assume that 

  39 *     - `initialize` has been called on all `GraphComponents`.

  40 *     - the main axes scales have been properly configured to the data.

  41 * - **renderComponent**: called each time the `Graph` is rendered, to render the component. 

  42 * Implementations can assume that `preRender` has been called on all `GraphComponents`.

  43 */

  44export interface LifecycleCalls {

  45    /** called the first time the `Graph` is rendered, before any of the other lifecycle methods. */

  46    initialize(svg:d3Base):void;

  47

  48    /** Called immediately before each call to renderComponent. */

  49    preRender(data:DataSet): void;

  50

  51    /** renders the component. */

  52    renderComponent(data:DataSet): void;

  53}

  56/** default settings for the `Graph` component */

  57export interface GraphDefaults extends ComponentDefaults {

  58    /** the duration of the `Graph`-wide transition, restarted whith each `render` call.  */

  59    transitionTime: number; // in ms

  60    easing: string;         // e.g. 'easeCubic'

  61}

  67    rows: DataRow[];

  70/**

  71 * ## Graph

  72 * Abstract base Graph.

  73 */

  74export abstract class Graph implements LifecycleCalls {

  75    /** the HTML root element to attach the render tree to. */

  76    protected root:any;

  77

  78    /** the `GraphCfg` object shared by all components in this graph */

  79    protected config: GraphCfg;

  80

  81    /** the plot component, provides access to individual series components */

  82    protected series:Series;

  83

  84    /** the list of components to render */

  85    private components: GraphComponent[] = [];

  86

  87    /** tracks whether components have been initialized. */

  88    private initialized = false;

  89

  90    constructor(root:any) { 

  91        this.root = root;

  92        this.config = this.initializeCfg();

  93        this.config.baseSVG = this.createBaseSVG(this.config); 

  94        this.updateBaseSVG(this.config);

  95        this.components = this.createComponents(this.config);

  96        this.makeDefaults();

  97        this.resize();

  98        window.onresize = () => this.resize();

  99    }

 100

 101    public get defaults(): ComponentDefaults {

 102        return this.config.defaults;

 103    }

 104

 105    public get viewport() {

 106        return this.config.viewPort;

 107    }

 108

 109    /** returns the types of all registered `Series` */

 110    public get seriesTypes():string[] {

 111        return Series.types;

 112    }

 113

 114    /**

 115     * render the tree with the supplied data.

 116     * @param data the data to render

 117     */

 118    public render(data:DataSet): void {

 119        const graph = this.config.defaults.graph;

 120        const easing = d3[graph.easing];

 121        this.config.transition = transition().duration(graph.transitionTime);

 122        this.config.transition.ease(easing);

 123        if (!this.initialized) {

 124            this.initialize(this.config.baseSVG);

 125            this.initialized = true;

 126        }

 127        this.preRender(data);

 128        this.renderComponent(data);

 129    }

 130

 131    /**

 132     * adds a series to the plot.

 133     * @param type type of plot to use, e.g. 'bubble' or 'scatter'

 134     * @param params the column name of the parameters used to plot the series

 135     */

 136    public addSeries(type:string, x:string, y:string, ...params:string[]) {

 137        //this.resize();

 138        this.series.addSeries(type, x, y, ...params);

 139    }

 143

 144    /** 

 145     * Called once at the beginning of the first call to `Graph.render()`

 146     * and initializes all known `GraphComponts`.

 147     */

 148    initialize(svg:d3Base): void {

 149        this.components.forEach(comp => comp.initialize(svg));

 150    } 

 151

 152    /** Called immediately before each call to renderComponent. */

 153    preRender(data:DataSet): void {

 154        this.setScales(data);

 155        this.components.forEach((comp:GraphComponent) => comp.preRender(data));

 156    } 

 157

 158    /** renders the component. */

 159    renderComponent(data:DataSet): void {

 160        this.components.forEach((comp:GraphComponent) => comp.renderComponent(data));

 161    }

 164    //************** Non-public part **************************/

 165

 166    /** set the scales for the graph prior to rendering components. */

 167    protected abstract setScales(data:DataSet):void;

 168

 169    /** called once during construction to create the components defaults. */

 170    protected createDefaults():GraphDefaults {

 171        return {

 172            transitionTime: 1000,

 173            easing: 'easeCubic'

 174        };    

 175    }

 176

 177    protected makeDefaults() {

 178        const defaults = this.config.defaults;

 179        defaults['graph'] = this.createDefaults();

 180        this.components.forEach(comp => defaults[comp.componentType] = comp.createDefaults());

 181    }

 183    /** creates and initializes the `Graph`-wide configuration object. */

 184    private initializeCfg():GraphCfg {

 185        return {

 186            baseSVG:  undefined,    

 187            client:   { x:0, y:0, width: 0, height: 0 },

 188            viewPort: {

 189                width: vpWidth,

 190                height: vpWidth * 0.7   // initial height: 70% of width

 191            },

 192            defaults: {},

 193            scales: {},

 194            transition: null

 195        };

 196    }

 197

 198    /** creates the list of `GraphComponents` and determines the rendering order. */

 199    private createComponents(cfg:GraphCfg):GraphComponent[] {

 200        return [

 201            new Scales(cfg),

 202            new Canvas(cfg),

 203            new Grids(cfg),

 204            new Axes(cfg),

 205            this.series = new Series(cfg)

 206        ];

 207    }

 209    /** callback on window resize event, adjusts the viewport to the new dimensions  */

 210    private resize() {

 211        const cfg = this.config;

 212        // if (this.root && this.root.clientWidth > 0) {

 213        if (this.root) {

 214            if (this.root.clientWidth !== cfg.client.width || this.root.clientHeight !== cfg.client.height) {

 215                log.info(`resizing svg: [${cfg.client.width} x ${cfg.client.height}] -> [${this.root.clientWidth} x ${this.root.clientHeight}]`);

 216                cfg.client.width = this.root.clientWidth;

 217                cfg.client.height = this.root.clientHeight;

 218                cfg.viewPort.height = cfg.viewPort.width * this.root.clientHeight / this.root.clientWidth;

 219                this.updateBaseSVG(cfg);

 220            }

 221        }

 222    }

 224    /**

 225     * creates the base SVG element for the entire graph.

 226     * This function is intended to be called only once at creation of the graph.

 227     * @param cfg 

 228     */

 229    private createBaseSVG(cfg: GraphCfg):d3Base {

 230        return d3Select(this.root).append('svg')

 231            .classed('baseSVG', true)

 232            .attr('height', '100%')

 233            .attr('width', '100%')

 234            .attr('preserveAspectRatio', 'xMinYMin meet');

 235    }

 236

 237    /**

 238     * sets the graph's viewBox, typically after a window resize.

 239     * @param cfg 

 240     */

 241    private updateBaseSVG(cfg: GraphCfg) {

 242        cfg.baseSVG.attr('viewBox', `0 0 ${cfg.viewPort.width} ${cfg.viewPort.height}`);

 243    }
