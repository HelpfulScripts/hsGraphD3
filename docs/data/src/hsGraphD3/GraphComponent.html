  14/** 

  15 * Configuration parameters for the render tree, passed to each `GraphComponent` during construction. 

  16 */

  17export interface GraphCfg {

  18    /** the base svg for the rendering tree */

  19    baseSVG:    d3Base;

  20

  21    /** dimensions of the HTML client rect containing this graph */

  22    client:     RectDef;

  23

  24    /** top level svg viewBox dimensions */

  25    viewPort: {

  26        width:  UnitVp;

  27        height: UnitVp;

  28    };

  29

  30    /** Default settings for GraphComp[onents in this graph] */

  31    defaults: DefaultsType;

  32    // defaults:   (compName?:string) => ComponentDefaults;

  33

  34    /** 

  35     * mapping of semantic scales to data columns and scaling functions. 

  36     * The scaling function will be set at rendering time for each new rendering frame.

  37     * @param dim the semantic name of the axis to plot and scale. E.g. for 2D cartesian plots, `dim` is `hor` or `ver`.

  38     * @param dataCol the name of the data column that will be plotted along this axis

  39     * @param scale a d3js scale used for scaling

  40     */    

  41    scales: { [dim:string]:  d3.ScaleContinuousNumeric};

  42}

  44/**

  45 * Generic base interface for the default setting for all component types.

  46 */

  47export interface ComponentDefaults {

  48    [filed:string]:any;

  49}

  50

  51/**

  52 * Generic base class for all component types

  53 */

  54export abstract class GraphComponent {

  55    /** the base svg element to render the component into */

  56    protected svg: d3Base;

  57    

  58    /** the render tree configuration */

  59    protected cfg: GraphCfg;

  60

  61    constructor(cfg:GraphCfg, svgBase:d3Base) { 

  62        this.cfg = cfg; 

  63        this.svg = svgBase; 

  64    }

  65

  66    /** returns the component type as a string name */

  67    abstract get componentType(): string;

  68

  69    /** renders the component for the given data */

  70    abstract renderComponent(data:Data): void;

  71

  72    /** creates a default entry for the component type in `Defaults` */

  73    abstract createDefaults(): ComponentDefaults;

  74}
