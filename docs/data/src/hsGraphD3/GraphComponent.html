  35/** 

  36 * Configuration parameters for the render tree, passed to each `GraphComponent` during construction. 

  37 */

  38export interface GraphCfg {

  39    /** the base svg for the rendering tree */

  40    baseSVG:    d3Base;

  41

  42    /** dimensions of the HTML client rect containing this graph */

  43    client:     RectDef;

  44

  45    /** top level svg viewBox dimensions */

  46    viewPort: {

  47        width:  UnitVp;

  48        height: UnitVp;

  49    };

  50

  51    /** Default settings for GraphComp[onents in this graph] */

  52    defaults: DefaultsType;

  53    // defaults:   (compName?:string) => ComponentDefaults;

  54

  55    /** 

  56     * mapping of semantic scales to data columns and scaling functions. 

  57     * The scaling function will be set at rendering time for each new rendering frame.

  58     * @param dim *string* the semantic name of the axis to plot and scale. E.g. for 2D cartesian plots, 

  59     * `dim` is `hor`, `ver`, or `size`. 

  60     */    

  61    scales: { [dim:string]: NumberScale};

  62

  63    /** the currently used transition. A new transition will be set each time `Graph.render()` is called. */

  64    transition: any;

  65}

  74/**

  75 * Generic base class for all component types

  76 */

  77export abstract class GraphComponent implements LifecycleCalls {

  78    /** the base svg element to render the component into */

  79    protected svg: d3Base;

  80    

  81    /** the render tree configuration */

  82    protected cfg: GraphCfg;

  83

  84    constructor(cfg:GraphCfg, compClass:string) { 

  85        this.cfg = cfg; 

  86        if (compClass) {

  87            this.cfg.baseSVG.append('g').classed(compClass, true);

  88        }

  89        this.svg = this.cfg.baseSVG.select(`.${compClass}`);

  90    }

  91

  92    /** returns the component type as a string name */

  93    abstract get componentType(): string;

  94

  95    /** creates a default entry for the component type in `Defaults` */

  96    abstract createDefaults(): ComponentDefaults;

  97

  98    //************** Lifecycle calls **************************/

  99

 100    /** called once during creation of the `Graph`. m*/

 101    abstract initialize(svg:d3Base): void;

 102

 103    /** Called immediately before each call to renderComponent. */

 104    abstract preRender(data:DataSet | DataSet[], domains:Domains): void; 

 105

 106    /** renders the component. */

 107    abstract renderComponent(data:DataSet | DataSet[]): void;

 108}
