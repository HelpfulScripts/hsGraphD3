  39/** 

  40 * Configuration parameters for the render tree, passed to each `GraphComponent` during construction. 

  41 */

  42export interface GraphCfg {

  43    /** the base svg for the rendering tree */

  44    baseSVG:    d3Base;

  45

  46    /** the `graph` object that this config belongs to. */

  47    graph:      Graph;

  48

  49    /** the set of graph components used  */

  50    components: Components,

  51

  52    /** dimensions of the HTML client rect containing this graph in pixel. */

  53    client:     { x:pixel; y:pixel; width:pixel; height:pixel; };

  54

  55    /** top level svg viewBox dimensions in viewport units*/

  56    viewPort: {

  57        orgX:   UnitVp;

  58        orgY:   UnitVp;

  59        width:  UnitVp;

  60        height: UnitVp;

  61    };

  62

  63    /** Default settings for GraphComponents in this graph */

  64    // defaults: DefaultsType;

  65

  66    // /** 

  67    //  * mapping of semantic scales to data columns and scaling functions. 

  68    //  * The scaling function will be set at rendering time for each new rendering frame.

  69    //  * @param dim *string* the semantic name of the axis to plot and scale. E.g. for 2D cartesian plots, 

  70    //  * `dim` is `hor`, `ver`, or `size`. 

  71    //  */    

  72    // scales: { [dim:string]: Scale};

  73

  74    /**

  75     * used to aggregate series into grouped stacks. 

  76     */

  77    stack: { [group:string]: {[abscissa:string]:number}; };

  78

  79    /** the currently used transition. A new transition will be set each time `Graph.render()` is called. */

  80    transition: any;

  89export abstract class GraphBase implements LifecycleCalls {

  90    /** the render tree configuration */

  91    protected cfg: GraphCfg;

  92

  93    constructor(cfg:GraphCfg) { 

  94        this.cfg = cfg; 

  95    }

  96

  97    /** creates a default entry for the component type in `Defaults` */

  98    abstract createDefaults(): ComponentDefaults;

  99

 100    /** getter for the defaults. */

 101    abstract get defaults(): ComponentDefaults;

 102

 103    //************** Lifecycle calls **************************/

 104

 105    /** called once during creation of the `Graph`. m*/

 106    abstract initialize(svg:d3Base): void;

 107

 108    /** Called immediately before each call to renderComponent. */

 109    abstract preRender(data:DataSet | DataSet[]): void; 

 110

 111    /** renders the component. */

 112    abstract renderComponent(data:DataSet | DataSet[]): void;

 113

 114    /** Called immediately after each call to renderComponent. */

 115    abstract postRender(data:DataSet | DataSet[]): void; 

 116}

 118/**

 119 * Generic base class for all component types

 120 */

 121export abstract class GraphComponent extends GraphBase {

 122    /** the base svg element to render the component into */

 123    protected svg: d3Base;

 124    

 125    constructor(cfg:GraphCfg, compClass:string) { 

 126        super(cfg); 

 127        if (compClass) { 

 128            this.svg = this.cfg.baseSVG.append('g').classed(compClass, true);

 129        } else if (compClass !== null) {

 130            console.log('no CompClass');

 131            console.log(new Error().stack);

 132        }

 133    }

 134

 135    /** returns the component type as a string name */

 136    abstract get componentType(): string;

 137}
