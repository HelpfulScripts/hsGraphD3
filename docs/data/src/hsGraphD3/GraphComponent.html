  21/** 

  22 * Configuration parameters for the render tree, passed to each `GraphComponent` during construction. 

  23 */

  24export interface GraphCfg {

  25    /** the base svg for the rendering tree */

  26    baseSVG:    d3Base;

  27

  28    /** dimensions of the HTML client rect containing this graph */

  29    client:     RectDef;

  30

  31    /** top level svg viewBox dimensions */

  32    viewPort: {

  33        width:  UnitVp;

  34        height: UnitVp;

  35    };

  36

  37    /** Default settings for GraphComp[onents in this graph] */

  38    defaults: DefaultsType;

  39    // defaults:   (compName?:string) => ComponentDefaults;

  40

  41    /** 

  42     * mapping of semantic scales to data columns and scaling functions. 

  43     * The scaling function will be set at rendering time for each new rendering frame.

  44     * @param dim the semantic name of the axis to plot and scale. E.g. for 2D cartesian plots, `dim` is `hor` or `ver`.

  45     * @param dataCol the name of the data column that will be plotted along this axis

  46     * @param scale a d3js scale used for scaling

  47     */    

  48    scales: { [dim:string]:  d3.ScaleContinuousNumeric};

  49

  50    /** the currently used transition. A new transition will be set each time `Graph.render()` is called. */

  51    transition: any;

  52}

  56 */

  59}

  61/**

  62 * Generic base class for all component types

  63 */

  64export abstract class GraphComponent implements LifecycleCalls {

  65    /** the base svg element to render the component into */

  66    protected svg: d3Base;

  67    

  68    /** the render tree configuration */

  69    protected cfg: GraphCfg;

  70

  71    constructor(cfg:GraphCfg, compClass:string) { 

  72        this.cfg = cfg; 

  73        if (compClass) {

  74            this.cfg.baseSVG.append('g').classed(compClass, true);

  75        }

  76        this.svg = this.cfg.baseSVG.select(`.${compClass}`);

  77    }

  78

  79    /** returns the component type as a string name */

  80    abstract get componentType(): string;

  81

  82    /** creates a default entry for the component type in `Defaults` */

  83    abstract createDefaults(): ComponentDefaults;

  84

  85    //************** Lifecycle calls **************************/

  86

  87    /** called once during creation of the `Graph`. m*/

  88    abstract initialize(svg:d3Base): void;

  89

  90    /** Called immediately before each call to renderComponent. */

  91    abstract preRender(data:DataSet, domains:Domains): void; 

  92

  93    /** renders the component. */

  94    abstract renderComponent(data:DataSet): void;

  95}
