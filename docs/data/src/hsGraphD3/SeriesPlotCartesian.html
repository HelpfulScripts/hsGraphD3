   3 * 

   4 * Abstract base class for all series plot types on cartesian coordinates.

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.add(, {: , ...});

   8 * ``` 

   9 * - `` is one of the registered types. See plot types for {@link plots.OrdinalSeriesPlot ordinal series} and {@link plots.NumericSeriesPlot numeric series}.

  10 * - `` is the semantic dimension to set. See {@link CartSeriesPlot.CartSeriesDimensions CartSeriesDimensions} for valid dimensions. 

  11 * - `` is the {@link SeriesPlot.ValueDef value definition}. 

  12 */

  37/**

  38 * valid {@link SeriesPlot.ValueDef `Value Definiton`} dimensions on cartesian plots:

  39 * - `x`?:  optional values for the x-axis. If omitted, the index of y-values will be used as x-values

  40 * - `y`:   values for the y-axis.

  41 * - `y0`?: optional values for lower fill border on the y-axis; defaults to `0`

  42 * - `r`?:  optional values for the size of markers. If provided, marker rendering is enabled.

  43 * 

  44 * Inherited from {@link SeriesPlot.SeriesDimensions SeriesDimensions}:

      45 * - `label`?: optional values for item {@link SeriesPlot.SeriesDimensions.label labels}

      46 * - `popup`?: optional values to show in mouse-over {@link SeriesPlot.SeriesDimensions.popup popups}.

      47 * - `color`?: optional values to determine {@link SeriesPlot.SeriesDimensions.color marker colors}

      48 * - `stacked`?: optional {@link SeriesPlot.SeriesDimensions.stacked stack group}. 

      49 *    Series with the same group will be stacked on each other

      50 */

      51export interface CartSeriesDimensions extends SeriesDimensions {

      52    /** 

      53     * optional, name of x-axis data column, or a function returning a value.

      54     * If omitted, the index of y-values will be used as x-values.

      55     */

      56    x?:  ValueDef;    

      57    /** name of y-axis data column, or a function returning a value */ 

      58    y:   ValueDef;    

      59    /** optional, name of y-axis data column for lower fill border, or a function returning a value */

      60    y0?: ValueDef;  

      61    /** optional, name of marker size data column, or a function returning a value */

      62    r?:  ValueDef;   

      63}

      66/**

      67 * Abstract base class for all cartesian plots.

      68 */

      69export abstract class SeriesPlotCartesian extends SeriesPlotScaled {

      70    /** the main data line  */

      71    protected line: string;         // d3Line;

      72

      73    // protected popupDiv:d3Base;

      74

      75    constructor(cfg:GraphCfg, seriesName:string, dims:CartSeriesDimensions) {

      76        super(cfg, seriesName, dims);

      77        this.type = 'cartesian';

      78    }

      79

      80    protected get dims(): CartSeriesDimensions { return super.dims; }

      81

      82    /** return the GraphDimension of the independent axis */

      83    protected abscissa:'hor'|'ver' = 'hor';

      84

      85    /** return the list of scalable Series dimesions for each Graph Dimension */

      86    get dimensions():CartDimensions { 

      87        return {

      88            hor: [this.dims.x],

      89            ver: [this.dims.y, this.dims.y0],

      90            size:[this.dims.r],

      91        };

      92    }

      93

      94    /** 

      95     * Set the defaults for the series. Called during `series.add`.

      96     * */

      97    public getDefaults(): SeriesPlotDefaults {

      98        const def = super.getDefaults();

      99        if (this.dims.r)    { 

     100            def.marker.rendered = true; 

     101            def.marker.stroke = defaultStroke(1);

     102        } else {

     103            def.marker.stroke = defaultStroke(0);

     104        }

     105        if (this.dims.y0)   { 

     106            def.area.rendered = true; 

     107        }

     108        if (this.dims.label){ 

     109            def.label.rendered = true; 

     110            def.label.color = '#000';

     111        }

     112        return def;

     113    }

     114

     115    getStackDim(v:ValueDef) { return this.dimensions[this.abscissa==='hor'? 'ver' : 'hor'].indexOf(v)>=0; }

     116    getAbscissaCol() { return {hor:this.dims.x, ver:this.dims.y}[this.abscissa]; }

     117    getOrdinateCol() { return {hor:this.dims.y, ver:this.dims.x}[this.abscissa]; }

     118

     119    //---------- lifecylce methods --------------------

     120

     121    public initialize(plot:d3Base, color?:string): void {

     122        super.initialize(plot, color);

     123        if (this.dims.popup===undefined) { this.dims.popup = {hor: this.dims.y, ver: this.dims.x}[this.abscissa]; }

     124

     125        // if abscissa data is missing, use implicit index as data

     126        if (this.abscissa === 'hor') {

     127            if (!this.dims.x) { this.dims.x = (i:number)=> i; }

     128        } else {

     129            if (!this.dims.y) { this.dims.y = (i:number)=> i; }

     130        }

     131    }

     132        

     133    //---------- support methods during lifecylce --------------------

     134

     135    protected d3RenderMarkers(plot:d3Base, data:DataSet) {

     136        const shape = this.markerShape();

     137        const defaults = this.defaults;

     138        const popup = this.cfg.components.popup;

     139        if (defaults.marker.rendered) {

     140            plot.select('.markers').selectAll(shape)

     141                .data(data.rows, d => d[0]) // bind to first DataVal, rather than to DataRow, iterate over rows

     142                .join(shape)              

     143                .call(popup.addListener.bind(popup), this.d3RenderPopup(data))

     144                .transition(this.cfg.transition)

     145                .call(this.d3DrawMarker.bind(this), data, defaults)

     146                .call(this.d3MarkerColors.bind(this), data, defaults);

     147        }

     148    }

     149

     150    protected abstract markerShape():string;

     151

     152    protected d3RenderLine(plot:d3Base, data:DataSet) {

     153        this.line = this.line || this.getPath(data.rows, data.colNames, this.dims.y);

     154        return this.getPathElement(plot, '.line').attr('d', (d:any) => this.line);

     155    }

     156

     157    protected d3RenderLabels(plot:d3Base, data:DataSet):void {

     158        const defaults = this.defaults;

     159        const popup = this.cfg.components.popup;

     160        if (defaults.label.rendered) {

     161            plot.select('.label').selectAll("text")

     162                .data(data.rows, (d:any[]) => d[0]) // bind to first DataVal, rather than to DataRow, iterate over rows

     163                .join('text')                       

     164                .call(popup.addListener.bind(popup), this.d3RenderPopup(data))

     165                .transition(this.cfg.transition) 

     166                .call(this.d3DrawLabels.bind(this), data, defaults);

     167        }

     168    }
