 109export class Grid {

 110    private hor: boolean;

 111    private svg: d3Base;

 112 

 113    constructor(protected cfg:GraphCfg, protected dir:Direction, protected type=MajorMinor.major) {

 114        const baseClass = `${dir}-${type}-Grid`;

 115        this.hor = this.dir===Direction.horizontal;

 116        this.svg = cfg.baseSVG.select('.grids').append('g');

 117        this.svg.classed(baseClass, true);

 118        // log.info(`>>> created ${baseClass}`)

 119    }

 120

 121    renderComponent() {

 122        const style = this.cfg.graph.defaults.grids[this.dir][this.type];

 123        if (style.rendered) {

 124            const trans = this.cfg.transition;

 125            const scaleX = this.cfg.components.scales.scaleDims.hor;

 126            const scaleY = this.cfg.components.scales.scaleDims.ver;

 127            setStroke(this.svg, style);

 128            const c = this.hor? 

 129                { range:  scaleX.range(), scale:  scaleY,   dim: { fix:'x', var:'y'}}   // hor grid

 130              : { range:  scaleY.range(), scale:  scaleX,   dim: { fix:'y', var:'x'}};  // ver grid

 131            const count = this.type===MajorMinor.major? c.scale.tickCountMajor : c.scale.tickCountMinor;

 132            try { 

 133                this.svg.selectAll("line")

 134                .data(c.scale.ticks(count), d => <any>d)

 135                .join(

 136                    enter => enter.append('line')

 137                        .attr(`${c.dim.fix}1`, c.range[0])

 138                        .attr(`${c.dim.fix}2`, c.range[1]),         

 139                    update => update,                 

 140                    exit => exit.remove(),

 141                )

 142                .transition(trans)

 143                    .attr(`${c.dim.var}1`, d => c.scale(<number>d))

 144                    .attr(`${c.dim.var}2`, d => c.scale(<number>d))

 145            } catch(e) {

 146                log.warn(`${this.type} ${this.hor?'hor':'ver'} grid: ${e}`)

 147            }

 148        }

 149    }
