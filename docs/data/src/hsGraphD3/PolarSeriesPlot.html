  35/**

  36 * valid {@link SeriesPlot.ValueDef `Value Definiton`} dimensions on polar plots:

  37 * - `phi`?:  optional values for the angular axis. 

  38 *    If omitted, the index of radial values will be used as  angular values

  39 * - `r`:   values for the radial axis.

  40 * - `r0`?: optional values for lower fill border on the  radial axis; defaults to `0`

  41 * - `label`?: optional values for item labels

  42 * - `popup`?: optional values to show in mouse-over popups.

  43 */

  44export interface PolarSeriesDimensions extends SeriesDimensions {

  45    /** 

  46     * optional, name of angular axis data column, or a function returning a value.

  47     * If omitted, the index of  radial values will be used as angular values.

  48     */

  49    phi?:   ValueDef;    

  50    /** name of radial axis data column, or a function returning a value */

  51    r:   ValueDef;    

  52    /** optional, name of  radial axis data column for lower fill border, or a function returning a value */

  53    r0?:   ValueDef;    

  54}

 104    public expandDomains(dataSet:DataSet, domains:Domains) {

 105        super.expandDomains(dataSet, domains);

 111    /**

 112     * Returns an accessor function to access the numeric value in a data row. 

 113     * @param dim the semantic dimension ('hor', 'ver', 'size') in which to aggregate

 114     * @param v data column value definition

 115     * @param colNames 

 116     */

 117    accessor(v:ValueDef, colNames:string[], useStack=true):AccessFn {

 118        if (useStack && this.dims.stacked) {

 119            // stackDim = is 'v' a stackable dimension?

 120            const stackDim = this.dimensions[this.abscissa==='ang'? 'rad' : 'ang'].indexOf(v)>=0;

 121            const abscissaCol = {ang:this.dims.phi, rad:this.dims.r}[this.abscissa];

 122            if (stackDim && typeof abscissaCol === 'string') {

 123                const stackIndex = colNames.indexOf(this.dims.stacked);

 124                const fn = super.accessor(v, colNames);

 125                return (row, rowIndex) => row[stackIndex] + fn(row, rowIndex);

 126            }

 127        }

 128        return super.accessor(v, colNames);

 129    }

 134    public initialize(svg:d3Base, color?:string): void {

 135        super.initialize(svg, color);

 136        const defaults = this.defaults;

 137

 138        // if abscissa data is missing, use implicit index as data

 139        const r = Math.min(this.cfg.viewPort.width, this.cfg.viewPort.height) / 2;

 140        if (!this.dims.r0)  { this.dims.r0= ()=> 0; }

 141        if (this.abscissa === 'ang') {

 142            if (!this.dims.phi) { this.dims.phi = 1; }

 143            if (!this.dims.r)   { this.dims.r   = ()=> r; }

 144        } else {

 145            if (!this.dims.r) { this.dims.r = 1; }

 146        }

 147        if (!this.dims.popup) { this.dims.popup = {ang: this.dims.r, rad: this.dims.phi}[this.abscissa]; }

 148

 149        if (defaults.area.rendered) {

 150            this.svg.append('g').classed('area', true).append('path');

 151            const area = this.svg.select('.area');

 152            setArea(area, defaults.area);

 153        }

 154

 155        if (defaults.line.rendered) {

 156            this.svg.append('g').classed('line', true).append('path');

 157            const line = this.svg.select('.line');

 158            setStroke(line, defaults.line);

 159        }

 160

 161        if (defaults.marker.rendered) {

 162            this.svg.append('g').classed('markers', true);

 163            const markers = this.svg.select('.markers');

 164            setStroke(markers, defaults.marker.stroke);

 165            setFill(markers, defaults.marker.fill);

 166        }

 167        if (defaults.label.rendered) {

 168            this.svg.append('g').classed('label', true);

 169            const label = this.svg.select('.label');

 170            setLabel(label, defaults.label);

 171        }

 172    }

 251    /** clears the stack for this cycle before any series rendering happens. */

 252    public clearStack(data:DataSet) {

 253        const group = this.dims.stacked;

 254        if (group) {

 255            if (data.colNames.indexOf(group) < 0) { 

 256                data.colNames.push(group); 

 257            }

 258            const stackCol = data.colNames.indexOf(group);

 259            data.rows.forEach(row => row[stackCol] = 0);

 260            this.cfg.stack[this.dims.stacked] = {};

 261        }

 262    }

 267    /** update stack after rendering series. */

 268    protected updateStack(data:DataSet) {

 269        const group = this.dims.stacked;

 270        if (group) {

 271            const stack = this.cfg.stack[group];

 272            const stackCol = data.colNames.indexOf(group);

 273            const abscissaCol = {hor:this.dims.phi, ver:this.dims.r}[this.abscissa];

 274            const abscissaIndex = data.colNames.indexOf(abscissaCol);

 275            const ordinateCol = {hor:this.dims.r, ver:this.dims.phi}[this.abscissa];

 276            const ordinateIndex = data.colNames.indexOf(ordinateCol);

 277            data.rows.forEach(row => {

 278                const abscissaKey = ''+row[abscissaIndex];

 279                row[stackCol] = stack[''+abscissaKey] || 0;

 280                stack[''+abscissaKey] = (stack[''+abscissaKey]||0) + row[ordinateIndex];

 281            });

 282        }
