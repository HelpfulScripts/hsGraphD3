  65export interface PopupDefaults extends ComponentDefaults { 

  66    rendered:   boolean;

  67    style:      TextStyle;

  74export class Popup extends GraphComponent {

  77    constructor(cfg:GraphCfg) {

  78        super(cfg, Popup.type);

  79    }

  80

  81    public get componentType() { return Popup.type; }

  82

  83    public get defaults():PopupDefaults { return this.cfg.graph.defaults.popup; }

  84 

  85    public createDefaults():PopupDefaults {

  86        const defs:PopupDefaults = {

  87            rendered:   false,

  88            fill:       defaultArea('#eee', 0.8),

  89            style:      defaultTextStyle(10),

  90            offset:     { xPx: -50, yPx:5 }

  91

  92        };

  93        defs.style.font.weight = 'bold';    // 100-900

  94        return defs;

  95    }

  96

  97    public addListener(items:d3Base, popupAccess:AccessFn) {

  98        items

  99            .on('mouseenter', this.showPopup(popupAccess).bind(this))

 100            .on('mousemove', this.movePopup.bind(this))

 101            .on('mouseleave', this.hidePopup.bind(this));

 102    }

 103    

 104    private showPopup(popupAccess:AccessFn) { 

 105        return (d:number[], i:number) => {

 106        this.svg

 107            .html(popupAccess(d,i))

 108            .transition()

 109            .duration(100)

 110            .style('opacity', this.defaults.fill.opacity)

 111            ;

 112        };

 113    }

 114

 115    private movePopup() {

 116        const o = this.defaults.offset;

 117        this.svg

 118            .style('left', `${event.layerX+o.xPx}px`)

 119            .style('top',  `${(event.layerY+o.yPx)}px`);

 120    }

 121

 122    private hidePopup() {

 123        this.svg

 124        .transition()

 125        .duration(500)

 126        .style('opacity', 0);

 127    }

 129    public initialize(svg:d3Base): void {

 130        const def = this.defaults;

 131        const parentID = (this.cfg.baseSVG)._groups[0][0].parentElement.id;

 132        if (parentID) {

 133            this.svg = select(`#${parentID}`).append('div');

 134            this.svg.attr('id', 'tooltip').attr('style', `

 135                position: absolute;

 136                padding: 5px;

 137                background-color: ${def.fill.color};

 138                border: ${def.fill.border.width}px solid ${def.fill.border.color};

 139                border-radius: 5px;

 140                color: ${def.style.color};

 141                font-family: ${def.style.font.family};

 142                font-size: ${def.style.font.size};

 143                font-style: ${def.style.font.style};

 144                font-weight: ${def.style.font.weight};

 145                text-align: center;

 146                line-height: 1.2em;

 147            `)

 148            .style('opacity', 0);

 149        } else {

 150            this.svg = undefined;

 151        }

 152    }
