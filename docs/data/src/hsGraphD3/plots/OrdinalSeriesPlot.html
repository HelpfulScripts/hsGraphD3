   1/**

   2 * # OrdinalSeriesPlot

   3 * 

   4 * Abstract base class for all ordinal series plot types.

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.series.add(, {: , ...});

   8 * ``` 

   9 * - `` is one of the registered types: 

  10 *     -   {@link plots.Bar `bar`} a horizontal bar chart

  11 *     -   {@link plots.Column `column`} a vertical column chart

  12 * - `` is one of the semantic dimensions defined for the plot. `NumericSeriesPlots` define the 

  13 *   following dimensions:

  14 *     - `x`: required; the value to plot along the x-axis

  15 *     - `y`: required; the value to plot along the y-axis

  16 *     - `stacked`: optional, specifies a group name for which series values will be stacked.

  17 * - `` is the {@link SeriesPlot.ValueDef value definition} to use for the {@link Series.SeriesDimensions semantic dimension}. Values may be specified

  18 *     - by `string` to specify the column name of the data set to use

  19 *     - by `number` to specify a constant value to use.  

  20 */

  60     * Returns an accessor function to access the numeric value in a data row. 

  61     * @param dim the semantic dimension ('hor', 'ver', 'size') in which to aggregate

  62     * @param v data column value definition

  63     * @param colNames

  65    accessor(v:ValueDef, colNames:string[]):(row:DataRow, rowIndex:number) => DataVal {

  66        const stack = this.cfg.stack[this.dims.stacked];

  67        const stackDim = this.dimensions[this.abscissa==='hor'? 'ver' : 'hor'].indexOf(v)>=0;

  68        const abscissaCol = {hor:this.dims.x, ver:this.dims.y}[this.abscissa];

  69        if (this.dims.stacked && stackDim && typeof abscissaCol === 'string') {

  70            const stackIndex = colNames.indexOf(this.dims.stacked);

  71            const fn = super.accessor(v, colNames);

  72            return (row, rowIndex) => {

  73                return row[stackIndex] + fn(row, rowIndex);

  74            };

  75        } else {

  76            return super.accessor(v, colNames);

  77        }

 149    private getParams(colNames:string[]):any[] {

 150        const scales = this.cfg.graph.scales;

 151        const scaleDef = scales.defaults.dims[this.abscissa];

 152        const stackGroup = this.dims.stacked || false;

 153        const allKeys = Object.keys(this.cfg.graph.series.defaults).filter(k => k.indexOf(Series.type)===0);

 154        const myKey = allKeys.indexOf(this.key) || 0;

 155        const step = scales.scaleDims[this.abscissa].step();

 156        const pad = scales.scaleDims[this.abscissa].paddingInner();

 157        const overlap = scaleDef.ordinal.overlap;

 158        const thickness = step * (1-pad) / (stackGroup? 1 : (allKeys.length * (1-overlap) + overlap));

 159        const offset = step*pad/2 + (stackGroup? 0 : thickness*myKey * (1-overlap));

 160        return [offset, thickness];

 161    }

 169    protected d3DrawBar(markers:d3Base, colNames:string[]) {

 170        const [offset, thickness] = this.getParams(colNames);

 171

 172        const xScale = this.cfg.graph.scales.scaleDims.hor;

 173        const yScale = this.cfg.graph.scales.scaleDims.ver;

 174

 175        if (this.abscissa==='hor') { // Column

 176            const x  = this.accessor(this.dims.x, colNames);

 177            const y  = this.accessor(this.dims.y, colNames);

 178            const y0 = this.dims.stacked? this.accessor(this.dims.stacked, colNames) : ()=>0;

 179            markers

 180                .attr("x",  (d:number[], i:number) => this.cached('x', i, ()=>xScale(x(d, i))) + offset)

 181                .attr("y",  (d:number[], i:number) => Math.min(this.cached('y', i, ()=>yScale(y(d, i))), this.cached('y0', i, ()=>yScale(y0(d, i)))))

 182                .attr("width",  () => thickness)

 183                .attr("height", (d:number[], i:number) => Math.abs(this.cached('y', i, ()=>yScale(y(d, i)))-this.cached('y0', i, ()=>yScale(y0(d, i)))));        

 184        } else {                    // Bar

 185            const x0 = this.dims.stacked? this.accessor(this.dims.stacked, colNames) : ()=>0;

 186            const x  = this.accessor(this.dims.x, colNames);

 187            const y  = this.accessor(this.dims.y, colNames);

 188            markers

 189                .attr("x",  (d:number[], i:number) => Math.min(this.cached('x', i, ()=>xScale(x(d, i))), this.cached('x0', i, ()=>xScale(x0(d, i)))))

 190                .attr("y",  (d:number[], i:number) => this.cached('y', i, ()=>yScale(y(d, i))) + offset)

 191                .attr("height",  () => thickness)

 192                .attr("width", (d:number[], i:number) => Math.abs(this.cached('x0', i, ()=>xScale(x0(d, i)))-this.cached('x', i, ()=>xScale(x(d, i)))));

 193        }

 194    }

 218                  + Math.abs(this.cached('y', i, ()=>yScale(y(d, i))) -this.cached('y0', i, ()=>yScale(y0(d, i)))) * ypos

 219                );

 220        } else {                        // Bar

 221            const x0 = this.dims.stacked? this.accessor(this.dims.stacked, colNames) : ()=>0;

 222            labels

 225                  + Math.abs(this.cached('x', i, ()=>xScale(x(d, i)))-this.cached('x0', i, ()=>xScale(x0(d, i))))* xpos

 226                )

 227                .attr("y",  (d:number[], i:number) => this.cached('y', i, ()=>yScale(y(d, i))) + offset + thickness* ypos);

 228        }

 229        labels.text((d:number[], i:number) => text(l(d, i)));

 232    protected getLine(rows:DataRow[], colNames:string[]):string {

 233        const hor = this.abscissa==='hor';

 234        const x  = this.accessor(this.dims.x, colNames);

 235        const y  = this.accessor(this.dims.y, colNames);

 236        const [offset, thickness] = this.getParams(colNames);

 237

 238        const xScale = this.cfg.graph.scales.scaleDims.hor;

 239        const yScale = this.cfg.graph.scales.scaleDims.ver;

 240

 241        const line = hor?

 242            stepLine(parseInt(''+thickness), 'hor')

 243                .x((d:number[], i:number) => this.cached('x', i, ()=>xScale(x(d, i)))+offset)

 244                .y((d:number[], i:number) => this.cached('y', i, ()=>yScale(y(d, i))))

 245          : stepLine(parseInt(''+thickness), 'ver')

 246                .x((d:number[], i:number) => this.cached('x', i, ()=>xScale(x(d, i))))

 247                .y((d:number[], i:number) => this.cached('y', i, ()=>yScale(y(d, i)))+offset);

 248        return line(rows);

 249    }

 251    protected labelPos(cfg:Label, labels:d3Base) {

 252        let xShift = 0;

 253        let yShift = 0.35;

 254        let xpos = 0.5; // 0: left aligned, 1: right aligned

 255        let ypos = 0.5;   // 0: top of bar, 1: bottom of bar

 256        let anchor = 'middle';

 257        switch(cfg.xpos) { 

 258            case TextHAlign.left:   xpos = 0; xShift = cfg.inside?0.2:-0.2; anchor = cfg.inside?'start':'end'; break;

 259            case TextHAlign.center: break;

 260            case TextHAlign.right:  xpos = 1; xShift = cfg.inside?-0.2:0.2; anchor = cfg.inside?'end':'start';  break;

 261            default: log.warn(`illegal TextHAlign: ${cfg.xpos}`);

 262        }

 263        switch(cfg.ypos) { // additional y 'em' shift

 264            case TextVAlign.top:    ypos = 0; yShift = cfg.inside? 1 : -0.2; break;

 265            case TextVAlign.center: break;

 266            case TextVAlign.bottom: ypos = 1; yShift = cfg.inside? -0.2 : 1; break;

 267            default:  log.warn(`illegal TextVAlign: ${cfg.ypos}`);

 268        }

 269        labels.style('text-anchor', anchor)

 270              .attr('dx', ((cfg.hOffset||0)+xShift).toFixed(1) + 'em')

 271              .attr('dy', ((cfg.vOffset||0)+yShift).toFixed(1) + 'em');

 272    return [xpos, ypos];

 273    }

 298    /** update stack after rendering series. */

 299    protected updateStack(data:DataSet) {

 300        const group = this.dims.stacked;

 301        if (group) {

 302            const stack = this.cfg.stack[this.dims.stacked];

 303            const stackCol = data.colNames.indexOf(group);

 304            const abscissaCol = {hor:this.dims.x, ver:this.dims.y}[this.abscissa];

 305            const abscissaIndex = data.colNames.indexOf(abscissaCol);

 306            const ordinateCol = {hor:this.dims.y, ver:this.dims.x}[this.abscissa];

 307            const ordinateIndex = data.colNames.indexOf(ordinateCol);

 308            data.rows.forEach(row => {

 309                const abscissaKey = ''+row[abscissaIndex];

 310                row[stackCol] = stack[abscissaKey] || 0;

 311                stack[abscissaKey] += row[ordinateIndex];

 312            });

 313        }

 314    }

 318function stepLine(step:number, axis:'hor'|'ver') {

 319    interface Accessor { (d:DataVal[], i:number): number; }

 320    let xAccess:Accessor;

 321    let yAccess:Accessor;

 322    const accessors = (rows: DataRow[]):string => {

 323        const result = rows.map((r, i) => {

 324            const x = parseInt(''+xAccess(r, i));

 325            const y = parseInt(''+yAccess(r, i));

 326            return (i===0?'M':'L') + x + ' ' + 

 327                ((axis==='hor')? (y + 'L' + (x+step)) : ((y+step) + 'L' + x ))

 328                + ' ' + y;

 329        }).join('');

 330        return result;

 331    };

 332    accessors.x = (fn:Accessor) => {

 333        xAccess = fn;

 334        return accessors;

 335    };

 336    accessors.y = (fn:Accessor) => {

 337        yAccess = fn;

 338        return accessors;

 339    };

 340    return accessors;

 341}
