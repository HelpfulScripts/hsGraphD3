   1/**

   2 * # OrdinalSeriesPlot

   3 * 

   4 * Abstract base class for all ordinal series plot types, i.e. plots with a ordinal abscissa and a numeric ordinate.

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.series.add(, {: , ...});

   8 * ``` 

   9 * - `` is one of the registered types: 

  10 *     -   {@link plots.Bar `bar`} a horizontal bar chart

  11 *     -   {@link plots.Column `column`} a vertical column chart

  12 * - `` is the semantic dimension to set. See {@link CartSeriesPlot.CartSeriesDimensions CartSeriesDimensions} for valid dimensions. 

  13 * - `` is the {@link SeriesPlot.ValueDef value definition}. 

  14 */

  89    private getParams(colNames:string[]):any[] {

  90        const scales = this.cfg.graph.scales;

  91        const scaleDef = scales.defaults.dims[this.abscissa];

  92        const stackGroup = this.dims.stacked || false;

  93        const allKeys = Object.keys(this.cfg.graph.series.defaults).filter(k => k.indexOf(Series.type)===0);

  94        const myKey = allKeys.indexOf(this.key) || 0;

  95        const step = scales.scaleDims[this.abscissa].step();

  96        const pad = scales.scaleDims[this.abscissa].paddingInner();

  97        const overlap = scaleDef.ordinal.overlap;

  98        const thickness = step * (1-pad) / (stackGroup? 1 : (allKeys.length * (1-overlap) + overlap));

  99        const offset = step*pad/2 + (stackGroup? 0 : thickness*myKey * (1-overlap));

 100        return [offset, thickness];

 101    }

 109    protected d3DrawMarker(markers:d3Base, data:DataSet, defaults:SeriesPlotDefaults) {

 110        const [offset, thickness] = this.getParams(data.colNames);

 111

 112        const xScale = this.cfg.graph.scales.scaleDims.hor;

 113        const yScale = this.cfg.graph.scales.scaleDims.ver;

 114        const x  = this.accessor(this.dims.x, data.colNames);

 115        const x0 = this.dims.stacked? this.accessor(this.dims.stacked, data.colNames) : ()=>0;

 116        const y  = this.accessor(this.dims.y, data.colNames);

 117        const y0 = this.dims.stacked? this.accessor(this.dims.stacked, data.colNames) : ()=>0;

 118        const xAttr  = (d:number[], i:number) => this.cached('x', i, ()=>xScale(x(d, i)));

 119        const x0Attr = (d:number[], i:number) => this.cached('x0',i, ()=>xScale(x0(d, i)));

 120        const yAttr  = (d:number[], i:number) => this.cached('y', i, ()=>yScale(y(d, i)));

 121        const y0Attr = (d:number[], i:number) => this.cached('y0',i, ()=>yScale(y0(d, i)));

 122

 123        if (this.abscissa==='hor') { // Column

 124            markers

 125                .attr("x",  (d:number[], i:number) => xAttr(d,i) + offset)

 126                .attr("y",  (d:number[], i:number) => Math.min(yAttr(d,i), y0Attr(d,i)))

 127                .attr("width",  () => thickness)

 128                .attr("height", (d:number[], i:number) => Math.abs(yAttr(d,i)-y0Attr(d,i)));        

 129        } else {                    // Bar

 130            markers

 131                .attr("x",  (d:number[], i:number) => Math.min(xAttr(d,i), x0Attr(d,i)))

 132                .attr("y",  (d:number[], i:number) => yAttr(d,i) + offset)

 133                .attr("height", () => thickness)

 134                .attr("width",  (d:number[], i:number) => Math.abs(x0Attr(d,i)-xAttr(d,i)));

 135        }

 136    }

 143    protected d3DrawLabels(labels:d3Base, data:DataSet, defaults:SeriesPlotDefaults) {

 144        const [offset, thickness] = this.getParams(data.colNames);

 145

 146        const xScale = this.cfg.graph.scales.scaleDims.hor;

 147        const yScale = this.cfg.graph.scales.scaleDims.ver;

 148        

 149        const l = this.accessor(this.dims.label, data.colNames);

 150        const cfg:Label = this.defaults.label;

 151

 152        const [xpos, ypos] = this.labelPos(cfg, labels);

 153        const x  = this.accessor(this.dims.x, data.colNames);

 154        const x0 = this.dims.stacked? this.accessor(this.dims.stacked, data.colNames) : ()=>0;

 155        const y  = this.accessor(this.dims.y, data.colNames);

 156        const y0 = this.dims.stacked? this.accessor(this.dims.stacked, data.colNames) : ()=>0;

 157        const xAttr  = (d:number[], i:number) => this.cached('x', i, ()=>xScale(x(d, i)));

 158        const x0Attr = (d:number[], i:number) => this.cached('x0', i, ()=>xScale(x0(d, i)));

 159        const yAttr  = (d:number[], i:number) => this.cached('y', i, ()=>yScale(y(d, i)));

 160        const y0Attr = (d:number[], i:number) => this.cached('y0', i, ()=>yScale(y0(d, i)));

 161        const _ = {

 162            hor: {x:'x', atX:xAttr, xPos:xpos, y:'y', atY:yAttr, atY0:y0Attr, yPos:ypos},

 163            ver: {x:'y', atX:yAttr, xPos:ypos, y:'x', atY:xAttr, atY0:x0Attr, yPos:xpos}

 164        }[this.abscissa];

 165        labels

 166            .attr(_.x,  (d:number[], i:number) => _.atX(d, i) + offset + thickness*_.xPos)

 167            .attr(_.y,  (d:number[], i:number) =>  Math.min(_.atY(d,i), _.atY0(d,i)) + Math.abs(_.atY(d,i) -_.atY0(d,i)) * _.yPos)

 168            .text((d:number[], i:number) => text(l(d, i)));

 169    }

 171    protected getPath(rows:DataRow[], colNames:string[]):string {

 172        const hor = this.abscissa==='hor';

 173        const x  = this.accessor(this.dims.x, colNames);

 174        const y  = this.accessor(this.dims.y, colNames);

 175        const [offset, thickness] = this.getParams(colNames);

 176

 177        const xScale = this.cfg.graph.scales.scaleDims.hor;

 178        const yScale = this.cfg.graph.scales.scaleDims.ver;

 179        const xAttr  = (d:number[], i:number) => this.cached('x', i, ()=>xScale(x(d, i)));

 180        const yAttr  = (d:number[], i:number) => this.cached('y', i, ()=>yScale(y(d, i)));

 181

 182        const line = hor?

 183            stepLine(parseInt(''+thickness), 'hor')

 184                .x((d:number[], i:number) => xAttr(d,i)+offset)

 185                .y((d:number[], i:number) => yAttr(d,i))

 186          : stepLine(parseInt(''+thickness), 'ver')

 187                .x((d:number[], i:number) => xAttr(d,i))

 188                .y((d:number[], i:number) => yAttr(d,i)+offset);

 189        return line(rows);

 190    }

 192    protected labelPos(cfg:Label, labels:d3Base) {

 193        let xShift = 0;

 194        let yShift = 0.35;

 195        let xpos = 0.5; // 0: left aligned, 1: right aligned

 196        let ypos = 0.5;   // 0: top of bar, 1: bottom of bar

 197        let anchor = 'middle';

 198        switch(cfg.xpos) { 

 199            case TextHAlign.left:   xpos = 0; xShift = cfg.inside?0.2:-0.2; anchor = cfg.inside?'start':'end'; break;

 200            case TextHAlign.center: break;

 201            case TextHAlign.right:  xpos = 1; xShift = cfg.inside?-0.2:0.2; anchor = cfg.inside?'end':'start';  break;

 202            default: log.warn(`illegal TextHAlign: ${cfg.xpos}`);

 203        }

 204        switch(cfg.ypos) { // additional y 'em' shift

 205            case TextVAlign.top:    ypos = 0; yShift = cfg.inside? 1 : -0.2; break;

 206            case TextVAlign.center: break;

 207            case TextVAlign.bottom: ypos = 1; yShift = cfg.inside? -0.2 : 1; break;

 208            default:  log.warn(`illegal TextVAlign: ${cfg.ypos}`);

 209        }

 210        labels.style('text-anchor', anchor)

 211              .attr('dx', ((cfg.hOffset||0)+xShift).toFixed(1) + 'em')

 212              .attr('dy', ((cfg.vOffset||0)+yShift).toFixed(1) + 'em');

 213    return [xpos, ypos];

 214    }

 218function stepLine(step:number, axis:'hor'|'ver') {

 219    interface Accessor { (d:DataVal[], i:number): number; }

 220    let xAccess:Accessor;

 221    let yAccess:Accessor;

 222    const accessors = (rows: DataRow[]):string => {

 223        const result = rows.map((r, i) => {

 224            const x = parseInt(''+xAccess(r, i));

 225            const y = parseInt(''+yAccess(r, i));

 226            return (i===0?'M':'L') + x + ' ' + 

 227                ((axis==='hor')? (y + 'L' + (x+step)) : ((y+step) + 'L' + x ))

 228                + ' ' + y;

 229        }).join('');

 230        return result;

 231    };

 232    accessors.x = (fn:Accessor) => {

 233        xAccess = fn;

 234        return accessors;

 235    };

 236    accessors.y = (fn:Accessor) => {

 237        yAccess = fn;

 238        return accessors;

 239    };

 240    return accessors;
