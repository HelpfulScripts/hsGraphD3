   1/**

   2 * # OrdinalSeriesPlot

   3 * 

   4 * Abstract base class for all ordinal series plot types.

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.series.add(, {: , ...});

   8 * ``` 

   9 * - `` is one of the registered types: 

  10 *     -   {@link plots.Bar `bar`} a horizontal bar chart

  11 *     -   {@link plots.Column `column`} a vertical column chart

  12 * - `` is one of the semantic dimensions defined for the plot. `NumericSeriesPlots` define the 

  13 *   following dimensions:

  14 *     - `x`: required; the value to plot along the x-axis

  15 *     - `y`: required; the value to plot along the y-axis

  16 *     - `stacked`: optional, specifies a group name for which series values will be stacked.

  17 * - `` is the value to use for the {@link Series.SeriesDimensions semantic dimension}. Values may be specified

  18 *     - by `string` to specify the column name of the data set to use

  19 *     - by `number` to specify a constant value to use.  

  20 */

  39/**

  40 * Returns a function to access the numeric value in a data row.

  41 * The numeric value is specified by the `ValueDef` `v`. 

  42 * - If `v` is a number, returns `v` as the result.

  43 * - If `v` is contained in `colNames`, uses the position of `v` in `colNames` to index the 

  44 * supplied `row`, applies the `scale`, and returns the result.

  45 * - Otherwise, if `v` ends with 'u', interprets `v` to be Viewport Units and 

  46 * returns `v` without aplying `scale`. This allows for absolute positioning inside the 

  47 * viewport window.

  48 * @param v the `ValueDef` specifying the value

  49 * @param colNames a list of names for the coluymns in the `DataSet`

  50 * @param def defaults to `0`; the default to return if `v` is `undefined`.

  51 */

  52export function accessor(v:ValueDef, colNames:string[], def=0):(row?:number[], i?:number) => DataVal {

  53    const index = colNames.indexOf(''+v);

  54    const fn = typeof(v)==='function';

  55    return v===undefined? 

  56        () => def :      // always return default

  57        (row?:number[], i?:number) => fn? (v)(i) : row[index];

  58}

  61/**

  62 * Abstract base class of a  cartesian series plot. 

  63 */

  64export abstract class OrdinalSeriesPlot extends CartSeriesPlot { 

  65    getDefaults(): SeriesPlotDefaults {

  66        const scaleDef = this.cfg.graph.defaults.scales.dims[this.independentAxis];

  67        scaleDef.type = 'ordinal';

  68        scaleDef.ordinal = scaleDef.ordinal || { gap:0.3, overlap:0};

  69

  70        const def = super.getDefaults();

  71        def.area.rendered = true;

  72        def.marker.rendered = false;

  73        def.line.rendered = false;

  74        def.line.width = 1;

  75        return def;

  76    } 

  77

  78    value(dim:string, v:ValueDef, colNames:string[]):(row?:DataRow, i?:number) => DataVal {

  79        const stackDim = this.independentAxis==='hor'? 'ver' : 'hor';

  80        if (this.dims.stacked && dim===stackDim) {

  81            const fn = typeof(v)==='function';

  82            const stack = this.cfg.stack[this.dims.stacked];

  83            const index = colNames.indexOf(''+v);

  84            return (row, i) => stack[i] += (fn? (v)(i) : row[index]);    

  85        } else {

  86            return super.value(dim, v, colNames);

  87        }

  88    }

  90    /** ensures that 0 is in the domain, since the columns extend down to 0. */

  91    protected expandNumDomain(dataSet:DataSet, domain:NumDomain, fn:(row?:DataRow, i?:number) => DataVal):NumDomain {

  92        domain = super.expandNumDomain(dataSet, domain, fn);

  93        domain[0] = Math.min(0, domain[0]);

  94        domain[1] = Math.max(0, domain[1]);

  95        return domain;

  96    }

 124    protected d3RenderFill(svg:d3Base, data:DataSet) {

 125        const defaults = this.defaults.area;

 126        if (defaults.rendered) {

 127            const samples:any = svg.select('.area').selectAll("rect")

 128                .data(data.rows, d => d[0]);                    // bind to data, iterate over rows

 129            samples.exit().remove();                            // remove unneeded rects

 130            samples.enter().append('rect')                      // add new rects

 131                .call(this.d3DrawBar.bind(this), this, data.colNames)

 132                .merge(samples).transition(this.cfg.transition) // draw markers

 133                .call(this.d3DrawBar.bind(this), this, data.colNames);

 134        }

 137    private getParams(plot:OrdinalSeriesPlot, colNames:string[]):any[] {

 138        const scales = plot.cfg.graph.scales;

 139        const scaleDef = scales.defaults.dims[this.independentAxis];

 140        const stackGroup = plot.dims.stacked || false;

 141        const allKeys = Object.keys(plot.cfg.graph.series.defaults).filter(k => k.indexOf(Series.type)===0);

 142        const myKey = allKeys.indexOf(plot.key) || 0;

 143        const step = scales.scaleDims[plot.independentAxis].step();

 144        const pad = scales.scaleDims[plot.independentAxis].paddingInner();

 145        const overlap = scaleDef.ordinal.overlap;

 146        const thickness = step * (1-pad) / (stackGroup? 1 : (allKeys.length * (1-overlap) + overlap));

 147        const offset = step*pad/2 + (stackGroup? 0 : thickness*myKey * (1-overlap));

 148        return [offset, thickness, stackGroup];

 149    }

 151    protected d3DrawBar(markers:d3Base, plot:OrdinalSeriesPlot, colNames:string[]) {

 152        const [offset, thickness, stackGroup] = this.getParams(plot, colNames);

 153        const stack = stackGroup? (i:number)=>plot.cfg.stack[stackGroup][i] : ()=>0;

 154        const xScale = plot.cfg.graph.scales.scaleDims.hor;

 155        const yScale = plot.cfg.graph.scales.scaleDims.ver;

 156

 157        if (plot.independentAxis==='hor') { // Column

 158            const index = colNames.indexOf(plot.dims.y);

 159            const x  = accessor(plot.dims.x, colNames);

 160            const y = (r:DataRow,i:number) => r[index] + stack(i);

 161            const y0 = accessor(stack, colNames);

 162            markers

 163                .attr("x",  (d:number[]) => xScale(x(d)) + offset)

 164                .attr("y",  (d:number[], i:number) => Math.min(yScale(y(d,i)), yScale(y0(d,i))))

 165                .attr("width",  () => thickness)

 166                .attr("height", (d:number[], i:number) => Math.abs(yScale(y0(d,i))-yScale(y(d,i))));        

 167        } else {                            // Bar

 168            const index = colNames.indexOf(plot.dims.x);

 169            const x = (r:DataRow,i:number) => r[index] + stack(i);

 170            const x0 = accessor(stack, colNames);

 171            const y  = accessor(plot.dims.y, colNames);

 172            markers

 173                .attr("x",  (d:number[], i:number) => Math.min(xScale(x(d,i)), xScale(x0(d,i))))

 174                .attr("y",  (d:number[]) => yScale(y(d)) + offset)

 175                .attr("height",  () => thickness)

 176                .attr("width", (d:number[], i:number) => Math.abs(xScale(x0(d,i))-xScale(x(d,i))));

 177        }

 178    }

 180    protected getLine(plot:OrdinalSeriesPlot, rows:DataRow[], colNames:string[]):string {

 181        const hor = plot.independentAxis==='hor';

 182        const x  = accessor(plot.dims.x, colNames);

 183        const y  = accessor(plot.dims.y, colNames);

 184        const [offset, thickness, stackGroup] = this.getParams(plot, colNames);

 185        const stack = stackGroup? (i:number)=>plot.cfg.stack[stackGroup][i] : ()=>0;

 186        const xScale = plot.cfg.graph.scales.scaleDims.hor;

 187        const yScale = plot.cfg.graph.scales.scaleDims.ver;

 188

 189        const line = hor?

 190            stepLine(parseInt(''+thickness), 'hor')

 191                .x((d:number[], i:number) => xScale(x(d))+offset)

 192                .y((d:number[], i:number) => yScale(y(d) + stack(i)))

 193          : stepLine(parseInt(''+thickness), 'ver')

 194                .x((d:number[], i:number) => xScale(x(d)+stack(i)))

 195                .y((d:number[], i:number) => yScale(y(d))+offset);

 196        return line(rows);

 197    }

 200function stepLine(step:number, axis:'hor'|'ver') {

 201    let xAccess:(d:DataVal[], i:number)=> number;

 202    let yAccess:(d:DataVal[], i:number)=> number;

 203    const accessors = (rows: DataRow[]):string => {

 204        const result = rows.map((r, i) => {

 205            const x = parseInt(''+xAccess(r, i));

 206            const y = parseInt(''+yAccess(r, i));

 207            return (i===0?'M':'L') + x + ' ' + 

 208                ((axis==='hor')? (y + 'L' + (x+step)) : ((y+step) + 'L' + x ))

 209                + ' ' + y;

 210        }).join('');

 211        return result;

 212    };

 213    accessors.x = (fn:(d:DataVal[], i:number)=> number) => {

 214        xAccess = fn;

 215        return accessors;

 216    };

 217    accessors.y = (fn:(d:DataVal[], i:number)=> number) => {

 218        yAccess = fn;

 219        return accessors;

 220    };

 221    return accessors;
