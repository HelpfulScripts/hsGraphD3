 108export class Pie extends SeriesPlotPolar {

 109    constructor(cfg:GraphCfg, seriesName:string, dims:PolarSeriesDimensions) {

 110        super(cfg, seriesName, dims);

 111        this.abscissa = 'rad';

 112    }

 113

 114    getDefaults(): PolarPlotDefaults {

 115        const def = super.getDefaults();

 116        def.area.rendered   = false;

 117        def.marker.rendered = true;

 118        def.line.rendered   = false;

 119        def.cornerRadius    = 0;

 120        def.padAngle        = 0;

 121        return def;

 122    } 

 123

 124    protected markerShape():string {

 125        return 'path';

 126    }

 127

 128    protected d3RenderMarkers(plot:d3Base, data:DataSet) {

 129        const defaults = this.defaults;

 130        if (defaults.marker.rendered) {

 131            const popup = this.cfg.components.popup;

 132            const shape = this.markerShape();

 133            const phiAccess = this.accessor(this.dims.phi, data.colNames, false);

 134            const pie = d3Pie().value((d:any, i:number) => phiAccess(d, i));

 135            plot.select('.markers').selectAll(shape)

 136                .data(pie(data.rows), (d:any) => d.data[3]) // bind to first DataVal, rather than to DataRow, iterate over rows

 137                .join(shape)                  

 138                .call(popup.addListener.bind(popup), this.d3RenderPopup(data))

 139                .transition(this.cfg.transition) // draw markers

 140                .call(this.d3DrawMarker.bind(this), data, defaults)

 141                .call(this.d3MarkerColors.bind(this), data, defaults);

 142        }

 143    }

 145    /** no line for pies. */

 146    protected d3RenderLine(plot:d3Base, data:DataSet) {}

 147

 148    /** no fill for pies. */

 149    protected d3RenderFill(plot:d3Base, data:DataSet) {}

 150

 151

 152    protected d3RenderLabels(plot:d3Base, data:DataSet):void {

 153        const defaults = this.defaults;

 154        const popup = this.cfg.components.popup;

 155        if (defaults.label.rendered) {

 156            const phiAccess = this.accessor(this.dims.phi, data.colNames, false);

 157            const pie = d3Pie().value((d:any, i:number) => phiAccess(d, i));

 158            const arcs = pie(data.rows);

 159            plot.select('.label').selectAll('text')

 160                .data(arcs, d => d[0]) // bind to first DataVal, rather than to DataRow, iterate over rows

 161                .join('text')              

 162                .call(popup.addListener.bind(popup), this.d3RenderPopup(data))

 163                .transition(this.cfg.transition) // draw markers

 164                .call(this.d3DrawLabels.bind(this), data, defaults);

 165        }

 166    }

 169    protected d3DrawMarker(markers:d3Base, data:DataSet, defaults:PolarPlotDefaults):void {

 170        const scales = this.cfg.components.scales.scaleDims;

 171        const rAccess = this.accessor(this.dims.r, data.colNames, false);

 172        const r0Access = this.accessor(this.dims.r0, data.colNames, false);

 173        const arc = d3Arc()

 174            .padAngle(defaults.padAngle)

 175            .cornerRadius(defaults.cornerRadius)

 176            .innerRadius((d:any, i:number) => scales.rad(r0Access(d.data, i)))

 177            .outerRadius((d:any, i:number) => scales.rad(rAccess(d.data, i)));

 178        markers.attr("d", arc);

 179    }

 180

 181    protected d3DrawLabels(labels:d3Base, data:DataSet, defaults:PolarPlotDefaults):void {

 182        const scales = this.cfg.components.scales.scaleDims;

 183        const rAccess = this.accessor(this.dims.r, data.colNames, false);

 184        const r0Access = this.accessor(this.dims.r0, data.colNames, false);

 185        const lAccess = this.accessor(this.dims.label, data.colNames, false);

 186        const centroid = (d:any, i:number, pos:number) => {

 187            const r = scales.rad(rAccess(d.data, i))*pos + scales.rad(r0Access(d.data, i))*(1-pos);

 188            const a = (d.startAngle + d.endAngle) / 2 - Math.PI / 2;

 189            return [Math.cos(a) * r, Math.sin(a) * r];

 190        };

 191        const cfg:Label = this.defaults.label;

 192        const pos = textPos(cfg.xpos, cfg.ypos, cfg.inside);

 193        labels.attr("transform", (d:any, i:number) => `translate(${centroid(d, i, pos.x.pos)})`)

 194              .text((d:any, i:number) => text(lAccess(d.data, i)))

 195              .attr('text-anchor', pos.x.anchor)

 196              .attr('dominant-baseline', pos.y.baseline)

 197              .attr('dx', ((cfg.hOffset||0)+pos.x.shift).toFixed(1) + 'em')

 198              .attr('dy', ((cfg.vOffset||0)+pos.y.shift).toFixed(1) + 'em');

 199    }

 201    protected getPath(rows:DataRow[], colNames:string[], yDef?: ValueDef, useStack?:boolean):string {

 202        return '';

 203    }
