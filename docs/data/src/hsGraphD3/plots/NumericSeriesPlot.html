   1/**

   2 * # NumericSeriesPlot

   3 * 

   4 * Abstract base class for all numeric series plot types.

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.addSeries(, {: , ...});

   8 * ``` 

   9 * - `` is one of the registered types: 

  10 *     -   {@link plots.Line `line`} a 2D line plot

  11 *     -   {@link plots.Bubble `bubble`} a 2D scatter plot with marker sizes driven by the data

  12 *     -   {@link plots.Area `area`} a 2D area plot filling to the x-axis

  13 *     -   {@link plots.TimeSeries `timeseries`} a 2D scatter plot with marker sizes driven by the data

  14 *     -   {@link plots.Voronoi `voronoi`} a voronoi diagrom with centroids and partition

  15 * - `` is one of the semantic dimensions defined for the plot. `NumericSeriesPlots` define the 

  16 *   following dimensions:

  17 *     - `x`: required; the value to plot along the x-axis

  18 *     - `y`: required; the value to plot along the y-axis

  19 *     - `y0`: optional; for areas, the lower bound of the area. 

  20 *       Specifying a value for `y0` will automatically enable area rendering

  21 *     - `r`: optional; for markers, the value to use for the size

  22 *       Specifying a value for `r` will automatically enable markers

  23 * - `` is the value to use for the {@link Series.SeriesDimensions semantic dimension}. Values may be specified

  24 *     - by `string` to specify the column name of the data set to use

  25 *     - by `number` to specify a constant value to use.  

  26 */

  41/**

  42 * Returns a function to access the numeric value in a data row.

  43 * The numeric value is specified by the `ValueDef` `v`. 

  44 * - If `v` is a number, applies the `scale` to `v`, and returns the result.

  45 * - If `v` is contained in `colNames`, uses the position of `v` in `colNames` to index the 

  46 * supplied `row`, applies the `scale`, and returns the result.

  47 * - Otherwise, if `v` ends with 'u', interprets `v` to be Viewport Units and 

  48 * returns `v` without aplying `scale`. This allows for absolute positioning inside the 

  49 * viewport window.

  50 * @param v the `ValueDef` specifying the value

  51 * @param colNames a list of names for the coluymns in the `DataSet`

  52 * @param scale a numeric scale to apply 

  53 * @param def defaults to `0`; the default to return if `v` is `undefined`.

  54 */

  55export function accessor(v:ValueDef, colNames:string[], scale:Scale, def=0):(row?:number[], i?:number) => number {

  56    const index = colNames.indexOf(''+v);

  57    const fn = typeof(v)==='function';

  58    // const unit = (''+v).slice(-1);

  59    // if (index < 0 && unit === 'u') {}

  60    return v===undefined? 

  61        () => def :      // always return default

  62        (row?:number[], i?:number) => scale(fn? (v)(i) : row[index]);

  63}

  66/**

  67 * Abstract base class of a  cartesian series plot. 

  68 */

  69export abstract class NumericSeriesPlot extends CartSeriesPlot { 

  70    /** the main data line  */

  71    protected line: string;         // d3Line;

  72

  73    //---------- lifecylce methods --------------------

  74

  75    initialize(svg:d3Base, color?:string): void {

  76        super.initialize(svg, color);

  77    }

  78

  79    preRender(data:NumericDataSet, domains:Domains): void {

  80        super.preRender(data, domains);

  81        const defaults = (this.cfg.defaults.series[this.key]);

  82        if (defaults.area.rendered && this.dims.y0===undefined) { this.dims.y0 = ()=>0; } 

  83        this.line = undefined;

  84    }

  85

  86    //---------- support methods during lifecylce --------------------

  87

  88    protected d3RenderMarkers(svg:d3Base, data:NumericDataSet) {

  89        const defaults = (this.cfg.defaults.series[this.key]).marker;

  90        if (defaults.rendered) {

  91            const samples:any = svg.select('.markers').selectAll("circle")

  92                .data(data.rows, d => d[0]);                // bind to data, iterate over rows

  93            samples.exit().remove();                        // remove unneeded circles

  94            samples.enter().append('circle')                // add new circles

  95                .call(this.d3DrawMarker, this, data.colNames)

  96            .merge(samples).transition(this.cfg.transition) // draw markers

  97                .call(this.d3DrawMarker, this, data.colNames);

  98        }

  99    }

 100

 101    protected d3RenderPath(svg:d3Base, data:NumericDataSet) {

 102        this.line = this.line || this.getLine(data.rows, data.colNames, this.dims.y);

 103        return this.getPathElement(svg, '.line').attr('d', (d:any) => this.line);

 104    }

 105

 106    protected d3RenderFill(svg:d3Base, data:NumericDataSet) {

 107        this.line = this.line || this.getLine(data.rows, data.colNames, this.dims.y);

 108        let line0 = '';

 109        if (this.dims.y0!==undefined) {

 110            const max = data.rows.length-1;

 111            const xmax = accessor(this.dims.x, data.colNames, this.cfg.scales.hor)(data.rows[max]);

 112            const x0   = accessor(this.dims.x, data.colNames, this.cfg.scales.hor)(data.rows[0]);

 113            const y    = accessor(this.dims.y0, data.colNames, this.cfg.scales.ver)(data.rows[max]);

 114            line0 = `L${xmax},${y}`;

 115            line0 += (typeof(this.dims.y0)==='function')? `L${x0},${y}` :

 116                this.getLine(data.rows.reverse(), data.colNames, this.dims.y0).slice(8);  // remove first 'M' command

 117        }

 118        return this.getPathElement(svg, '.area').attr('d', (d:any) => this.line + line0);

 119    }

 120

 121    protected d3DrawMarker(markers:d3Base, plot:NumericSeriesPlot, colNames:string[]) {

 122        const xAccess = accessor(plot.dims.x, colNames, plot.cfg.scales.hor);

 123        const yAccess = accessor(plot.dims.y, colNames, plot.cfg.scales.ver);

 124        const rDefault = plot.cfg.defaults.series[plot.key].marker.size;

 125        const rAccess = accessor(plot.dims.r, colNames, plot.cfg.scales.size, rDefault);

 126        markers

 127            .attr("cx", (d:number[]) => xAccess(d))

 128            .attr("cy", (d:number[]) => yAccess(d))

 129            .attr("r",  (d:number[]) => rAccess(d));

 130    }

 131    

 132    /**

 133     * returns the path rendering for the main data line 

 134     * @param data the data set to render from

 135     * @param yDef a constant (defaults to 0), or the data column to render from

 136     */

 137    protected getLine(rows:NumericDataRow[], colNames:string[], yDef: ValueDef = () => 0):string {

 138        const line = d3line()

 139            .x(accessor(this.dims.x, colNames, this.cfg.scales.hor))

 140            .y(accessor(yDef, colNames, this.cfg.scales.ver))

 141            .curve(curveCatmullRom.alpha(0.2));

 142        return line(<[number, number][]>rows);

 143    }

 144

 145    protected getPathElement(svg:d3Base, cls:string):any {

 146        return svg.select(cls).selectAll('path').transition(this.cfg.transition);

 147    }

 148}
