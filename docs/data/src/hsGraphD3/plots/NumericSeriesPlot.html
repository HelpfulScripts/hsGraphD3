   1/**

   2 * # NumericSeriesPlot

   3 * 

   4 * Abstract base class for all numeric series plot types.

   5 * To create a series plot, add the desired plot type to the graph:

   6 * ```

   7 * graph.series.add(, {: , ...});

   8 * ``` 

   9 * - `` is one of the registered types: 

  10 *     -   {@link plots.Line `line`} a 2D line plot

  11 *     -   {@link plots.Bubble `bubble`} a 2D scatter plot with marker sizes driven by the data

  12 *     -   {@link plots.Area `area`} a 2D area plot filling to the x-axis

  13 *     -   {@link plots.TimeSeries `timeseries`} a 2D scatter plot with marker sizes driven by the data

  14 *     -   {@link plots.Voronoi `voronoi`} a voronoi diagrom with centroids and partition

  15 * - `` is one of the semantic dimensions defined for the plot. `NumericSeriesPlots` define the 

  16 *   following dimensions:

  17 *     - `x`: required; the value to plot along the x-axis

  18 *     - `y`: required; the value to plot along the y-axis

  19 *     - `y0`: optional; for areas, the lower bound of the area. 

  20 *       Specifying a value for `y0` will automatically enable area rendering

  21 *     - `r`: optional; for markers, the value to use for the size

  22 *       Specifying a value for `r` will automatically enable markers

  23 * - `` is the value to use for the {@link Series.SeriesDimensions semantic dimension}. Values may be specified

  24 *     - by `string` to specify the column name of the data set to use

  25 *     - by `number` to specify a constant value to use.  

  26 */

  27

  53    preRender(data:NumericDataSet, domains:Domains): void {

  54        super.preRender(data, domains);

  55        const defaults = this.defaults;

  56        if (defaults.area.rendered && this.dims.y0===undefined) { 

  57            this.dims.y0 = this.dims.stacked? this.dims.stacked : ()=>0; 

  58        } 

  59    }

  63    protected markerShape() { return 'circle'; }

  65    protected d3RenderFill(svg:d3Base, data:NumericDataSet) {

  66        const scales = this.cfg.graph.scales.scaleDims;

  67        let line = this.line = this.line || this.getLine(data.rows, data.colNames, this.dims.y);

  68        if (this.dims.y0!==undefined) {

  69            line += `L` + this.getLine(data.rows.reverse(), data.colNames, this.dims.y0, false).slice(1); // replace first 'M' with 'L'

  70        }

  71        return this.getPathElement(svg, '.area').attr('d', (d:any) => line);

  72    }

  80    protected d3DrawMarker(markers:d3Base, colNames:string[]) {

  81        const scales = this.cfg.graph.scales.scaleDims;

  82        const xAccess = this.accessor(this.dims.x, colNames);

  83        const yAccess = this.accessor(this.dims.y, colNames);

  84        // don't scale markers as 'stacked markers' -> use super instead of this:

  85        const rAccess = this.accessor(this.dims.r, colNames, false);

  86        const rDefault = this.defaults.marker.size;

  87        markers

  88            .attr("cx", (d:number[], i:number) => scales.hor(xAccess(d, i)))

  89            .attr("cy", (d:number[], i:number) => scales.ver(yAccess(d, i)))

  90            .attr("r",  (d:number[], i:number) => this.dims.r? scales.size(rAccess(d, i)) : rDefault);

  91    }

  93    protected d3DrawLabels(labels:d3Base, colNames:string[]) {

  94        const scales = this.cfg.graph.scales.scaleDims;

  95        const xAccess = this.accessor(this.dims.x, colNames);

  96        const yAccess = this.accessor(this.dims.y, colNames);

  97        const rAccess = this.accessor(this.dims.r, colNames, false);

  98        const rDefault = this.defaults.marker.size;

  99        const lAccess = this.accessor(this.dims.label, colNames, false);

 100        const cfg:Label = this.defaults.label;

 101

 102        const [xpos, ypos, yShift] = this.labelPos(cfg, labels);

 103        labels

 104            .attr("x", (d:number[], i:number) => scales.hor(xAccess(d, i))

 105                + (this.dims.r? scales.size(rAccess(d, i)) : rDefault) * xpos)

 106            .attr("y", (d:number[], i:number) => scales.ver(yAccess(d, i)) 

 107                + (this.dims.r? scales.size(rAccess(d, i)) : rDefault) * ypos)

 108            .text((d:number[], i:number) => text(lAccess(d, i)));

 109        }

 111    /**

 112     * returns the path rendering for the main data line 

 113     * @param rows the data rows set to render from

 114     * @param yDef a constant (defaults to 0), or the data column to render from

 115     */

 116    protected getLine(rows:NumericDataRow[], colNames:string[], yDef: ValueDef = () => 0, useStack=true):string {

 117        const scales = this.cfg.graph.scales.scaleDims;

 118        const xAccess = this.accessor(this.dims.x, colNames, useStack);

 119        const yAccess = this.accessor(yDef, colNames, useStack);

 120        const line = d3line()

 121            .x((d:number[], i:number) => scales.hor(xAccess(d, i)))

 122            .y((d:number[], i:number) => scales.ver(yAccess(d, i)))

 123            .curve(curveCatmullRom.alpha(0.2));

 124        return line(<[number, number][]>rows);

 125    }

 127    protected labelPos(cfg:Label, labels:d3Base) {

 128        let xShift = 0;

 129        let yShift = 0.35;

 130        let xpos = 0;   // 0: left aligned, 1: right aligned

 131        let ypos = 0;   // 0: top of bar, 1: bottom of bar

 132        let anchor = 'middle';

 133        switch(cfg.xpos) { 

 134            case TextHAlign.left:   xpos = -1; xShift = -0.4; anchor = 'end'; break;

 135            case TextHAlign.center: break;

 136            case TextHAlign.right:  xpos = 1;  xShift = +0.4; anchor = 'start';  break;

 137            default: log.warn(`illegal TextHAlign: ${cfg.xpos}`);

 138        }

 139        switch(cfg.ypos) { // additional y 'em' shift

 140            case TextVAlign.top:    ypos = -1; yShift = -0.4; break;

 141            case TextVAlign.center: break;

 142            case TextVAlign.bottom: ypos = 1;  yShift = 1.0;  break;

 143            default:  log.warn(`illegal TextVAlign: ${cfg.ypos}`);

 144        }

 145        labels.style('text-anchor', anchor)

 146              .attr('dx', ((cfg.hOffset||0)+xShift).toFixed(1) + 'em')

 147              .attr('dy', ((cfg.vOffset||0)+yShift).toFixed(1) + 'em');

 148    return [xpos, ypos];

 149    }
