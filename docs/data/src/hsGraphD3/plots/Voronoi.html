 105export class Voronoi extends SeriesPlot {

 106    private voronoi: d3Voronoi;

 108    renderComponent(data:DataSet): void {

 109        const scales = this.cfg.scales;

 110        const x = this.cols[0];

 111        const y = this.cols[1];

 112        const m = this.cfg.defaults.scales.margin;

 113        this.voronoi = Delaunay.from(data.rows, 

 114            r => scales.hor(r[x]),

 115            r => scales.ver(r[y])

 116        ).voronoi([m.left, m.top, this.cfg.viewPort.width-m.right, this.cfg.viewPort.height-m.bottom]);

 117        super.renderComponent(data);

 118    }

 120    d3RenderPath(svg:d3Base, data:DataSet) {

 121        const defaults = (this.cfg.defaults.series[this.key]).line;

 122        if (defaults.rendered) {

 123            const path = svg.selectAll('path')

 126        }

 129    /**

 130     * Calculates the nearest `anchor` to the `sample` vector ands returns the index in the `anchors` array.

 131     * @param sample 

 132     * @param anchors 

 133     */

 134    nearest(sample:number[], anchors:number[][]) {

 135        let best = 0;

 136        let dist = 1e99;

 137        anchors.forEach((r, i) => {

 138            const d = Math.pow(r[0] - sample[0], 2) + Math.pow(r[1] - sample[1], 2);

 139            if (d < dist) {

 140                dist = d;

 141                best = i;

 142            }

 143        });

 144        return best;

 145    }

 147    /**

 148     * Calculates the centroid (average) vector over the array od sample vectors.

 149     * @param samples 

 150     * @param weight 

 151     * @return the centroid vector

 152     */

 153    centroid(samples:number[][]) {

 154        const sum = [0, 0];

 155        let count = 0;

 156        

 157        samples.forEach(row => {

 158            sum[0] = sum[0] + row[0];

 159            sum[1] = sum[1] + row[1];

 160            count++;

 161        });

 162        sum[0] = count? sum[0]/count : 0;

 163        sum[1] = count? sum[1]/count : 0;

 164        return {

 165            avg: sum,

 166            count: count

 167        };

 168    }
