   2 * # Sankey Plot

   3 * 

   4 * plots Sankley lines. 

   5 * 

   6 * ## Usage

   7 * `graph.add('sankey', {keys:[<node-col>, ...], value:<value-col>, color:<color-scheme>});`

   8 * - `keys` the columns to use as nodes. At least 2 keys are required. The sequence of keys determines the sequence of the graph flow.

   9 * - `value` is the {@link SeriesPlot.ValueDef value definition}.

  10 * - `color` is the color scheme to use, e.g. 'cat10'

  11 * 

  12 * 

  13 * ## Example

  14 * <example height=200px libs={hsGraphD3:'hsGraphD3'}>

  15 * <file name='script.js'>

  16 * const data = {

  17 *    colNames:['from', 'via', 'to', 'count'], 

  18 *    rows:[['a', 'sheep', 'one', 45], ['b', 'wolf', 'one', 15], ['c', 'sheep', 'two', 22],

  19 *          ['c', 'sheep', 'one', 40], ['c', 'wolf', 'three', 23], ['a', 'fence', 'three', 5]

  20 *    ]

  21 * };

  22 * const graph = new hsGraphD3.Graph(root);

  23 * graph.add('sankey', {keys:['from', 'via', 'to'], value:'count', color:'cat10'});

  24 * graph.render(data);

  26 * </example>

 146    public preRender(data:DataSet): void {

 147        super.preRender(data);

 148        const margin = this.cfg.graph.defaults.scales.margin;

 149        switch((<SankeyDefaults>this.defaults).align) {

 150            case 'left':        this.align = sankeyLeft; break;

 151            case 'right':       this.align = sankeyRight; break;

 152            case 'center':      this.align = sankeyCenter; break;

 153            case 'justified':   this.align = sankeyJustify; break;

 154        }

 155        if (this.sankey) {

 156            this.sankey.update(this.createNodesAndLinks(data))

 157        } else {

 158            this.sankey = d3Sankey()

 159                .nodeId((d:any) => d.name)

 160                .nodeAlign(this.align)

 161                .nodeWidth((<SankeyDefaults>this.defaults).width)

 162                .nodePadding((<SankeyDefaults>this.defaults).padding)

 163                .extent([[margin.left, margin.top], [this.cfg.viewPort.width - margin.left - margin.right, this.cfg.viewPort.height - margin.top - margin.bottom]]);

 164            const {nodes, links} = this.sankey(this.createNodesAndLinks(data));

 165            this.nodes = nodes;

 166            this.links = links;

 167        }

 170    private createNodesAndLinks(data:DataSet):{nodes:Nodes, links:any[]} {

 171        const keys = (<string[]>this.dims.keys).map(k => data.colNames.indexOf(k));

 172        const v = data.colNames.indexOf(<string>this.dims.value);

 173        const nodes:any = this.nodes || [];

 174        nodes.names = nodes.names || {};

 175        const links:any = this.links || [];

 176        links.myKeys = links.myKeys || {};

 177        data.rows.forEach((r:number[]|string[]) => { 

 178            for (let i=1; i<keys.length; i++) {

 179                const s = keys[i-1];

 180                const d = keys[i];

 181                if (!nodes.names[r[s]]) { nodes.push(nodes.names[r[s]] = { name: r[s]})}

 182                if (!nodes.names[r[d]]) { nodes.push(nodes.names[r[d]] = { name: r[d]})}

 183                const key = `${r[s]}-${r[d]}`;

 184                if (!links.myKeys[key]) {

 185                    links.push(links.myKeys[key] = {

 186                        source: r[s],

 187                        target: r[d],

 188                        names:  [r[s], r[d]],

 189                        value: 0

 190                    });

 191                }

 192                links.myKeys[key].value += r[v];

 193            }

 194        });

 195        return {

 196            nodes: nodes,

 197            links: links

 198        };

 199    }

 214    protected d3RenderMarkers(plot:d3Base, data:DataSet) {

 215        const defaults = this.defaults;

 216        const popup = this.cfg.components.popup;

 217        if (defaults.marker.rendered) {

 218            plot.select('.markers')

 219                .selectAll("rect")

 220                .data(this.nodes)

 221                .join("rect")

 222                .call(popup.addListener.bind(popup), this.d3RenderPopup(data))

 223                .transition(this.cfg.transition)

 224                .call(this.d3DrawMarker.bind(this), data, defaults)

 225                .call(this.d3MarkerColors.bind(this), data, defaults);

 226        }

 227    }

 229    protected d3RenderFill(plot:d3Base, data:DataSet) {

 230        const defaults = this.defaults;

 231        if (defaults.area.rendered) {

 232            plot.select('.area')

 233                .attr('fill-opacity', 0)

 234                .selectAll("g")

 235                .data(this.links)

 236                .join("path")

 237                    .style("mix-blend-mode", "multiply")

 238                    .attr("d", sankeyLinkHorizontal())

 239                    .attr("stroke", d => defaults.line.color)

 240                    .attr("stroke-width", d => Math.max(1, d.width))

 241                    .call(this.d3FillColors.bind(this), data, defaults)

 242                    .append("title")

 243                    .text((d:any) => `${d.source.name} â†’ ${d.target.name}:${d.value}`);

 244        }

 245    }

 247    protected d3RenderLabels(plot:d3Base, data:DataSet):void {

 248        const defaults = this.defaults;

 249        const width = this.cfg.viewPort.width;

 250        if (defaults.label.rendered) {

 251            plot.select('.label')

 252                .attr("font-family", "sans-serif")

 253                .attr("font-size", 10)

 254                .selectAll("text")

 255                .data(this.nodes)

 256                .join("text")

 257                    .attr("x", d => d.x0 < width / 2 ? d.x1 + 6 : d.x0 - 6)

 258                    .attr("y", d => (d.y1 + d.y0) / 2)

 259                    // .attr("dy", "0.35em")

 260                    .attr("text-anchor", d => d.x0 < width / 2 ? "start" : "end")

 261                    .text((d:any) => d.name);

 262            }
