   2 * # Sankey Plot

   3 * 

   4 * plots Sankley lines. 

   5 * 

   6 * ## Usage

   7 * `graph.series.add('sankey', {keys:[, ...], value:, color:});`

   8 * - `keys` the columns to use as nodes. At least 2 keys are required. The sequence of keys determines the sequence of the graph flow.

   9 * - `value` is the {@link SeriesPlot.ValueDef value definition}.

  10 * - `color` is the color scheme to use, e.g. 'cat10'

  11 * 

  12 * 

  13 * ## Example

  14 * 'hsGraphD3'}>

  15 * 'script.js'>

  16 * const data = {

  17 *    colNames:['from', 'via', 'to', 'count'], 

  18 *    rows:[['a', 'sheep', 'one', 45], ['b', 'wolf', 'one', 15], ['c', 'sheep', 'two', 22],

  19 *          ['c', 'sheep', 'one', 40], ['c', 'wolf', 'three', 23], ['a', 'fence', 'three', 5]

  20 *    ]

  21 * };

  22 * const graph = new hsGraphD3.GraphCartesian(root);

  23 * graph.series.add('sankey', {keys:['from', 'via', 'to'], value:'count', color:'cat10'});

  24 * graph.render(data);

  25 *

  91    public preRender(data:DataSet, domains:Domains): void {

  92        const margin = this.cfg.graph.defaults.scales.margin;

  93        super.preRender(data, domains);

  94        

  95        const sankey = d3Sankey()

  96            .nodeId((d:any) => d.name)

  97            // .nodeAlign(d3[`sankey${align[0].toUpperCase()}${align.slice(1)}`])

  98            .nodeWidth((this.defaults).width)

  99            .nodePadding((this.defaults).padding)

 100            .extent([[margin.left, margin.top], [this.cfg.viewPort.width - margin.left - margin.right, this.cfg.viewPort.height - margin.top - margin.bottom]]);

 101        const {nodes, links} = sankey(this.createNodesAndLinks(data));

 106    private createNodesAndLinks(data:DataSet):{nodes:any[], links:any[]} {

 107        const keys = (this.dims.keys).map(k => data.colNames.indexOf(k));

 108        const v = data.colNames.indexOf(this.dims.value);

 109        const nodes = <{name:string}[]>[];

 110        const links = [];

 111        const linkByKey = {};

 112        data.rows.forEach(r => { 

 113            for (let i=1; i

 114                const s = keys[i-1];

 115                const d = keys[i];

 116                nodes.push({ name: r[s]});

 117                nodes.push({ name: r[d]});

 118                const key = `${s}-${d}`;

 119                let link = linkByKey[key];

 120                if (!link) {

 121                    links.push(link = {

 122                        source: r[s],

 123                        target: r[d],

 124                        names:  [r[s], r[d]],

 125                        value: 0

 126                    });

 127                }

 128                link.value += r[v];

 129            }

 130        });

 131        return {

 132            nodes: uniquefy(nodes, 'name'),

 133            links: links

 134        };

 135    }

 150    protected d3RenderMarkers(plot:d3Base, data:DataSet) {

 151        const defaults = this.defaults;

 152        const popup = this.cfg.graph.popup;

 153        if (defaults.marker.rendered) {

 154            plot.select('.markers')

 155                .selectAll("rect")

 156                .data(this.nodes)

 157                .join("rect")

 158                .call(popup.addListener.bind(popup), this.d3RenderPopup(data))

 159                .transition(this.cfg.transition)

 160                .call(this.d3DrawMarker.bind(this), data, defaults)

 162        }

 165    protected d3RenderFill(plot:d3Base, data:DataSet) {

 166        const defaults = this.defaults;

 167        if (defaults.area.rendered) {

 168            plot.select('.area')

 169                .selectAll("g")

 170                .data(this.links)

 171                .join("path")

 172                    .style("mix-blend-mode", "multiply")

 173                    .attr("d", sankeyLinkHorizontal())

 174                    .attr("stroke", d => defaults.line.color)

 175                    .attr("stroke-width", d => Math.max(1, d.width))

 176                    .call(this.d3FillColors.bind(this), data, defaults)

 177                    .append("title")

 178                    .text((d:any) => `${d.source.name} â†’ ${d.target.name}`);

 179        }

 182    protected d3RenderLabels(plot:d3Base, data:DataSet):void {

 183        const defaults = this.defaults;

 184        const width = this.cfg.viewPort.width;

 185        if (defaults.label.rendered) {

 186            plot.select('.label')

 187                .attr("font-family", "sans-serif")

 188                .attr("font-size", 10)

 189                .selectAll("text")

 190                .data(this.nodes)

 191                .join("text")

 192                    .attr("x", d => d.x0 < width / 2 ? d.x1 + 6 : d.x0 - 6)

 193                    .attr("y", d => (d.y1 + d.y0) / 2)

 194                    // .attr("dy", "0.35em")

 195                    .attr("text-anchor", d => d.x0 < width / 2 ? "start" : "end")

 196                    .text((d:any) => d.name);

 197            }
