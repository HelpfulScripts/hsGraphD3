  16 * let index = 0;

  17 * const data = {

  18 *    colNames:['date', 'time', 'volume'], 

  19 *    rows:[],

  21 * while (index<11) { data.rows.push([index++, Math.random()+1, Math.random()])}

  23 * // create the graph and define the series to plot:

  26 * graph.addSeries('timeseries', {x:'date', y:'volume', r:'time'});

  82    preRender(data:NumericDataSet, domains:Domains): void {

  83        super.preRender(data, domains);

  84        if (typeof(this.dims.x)==='number') { log.warn(`preRender: unsupported const x=${this.dims.x} in timeseries`); }

  85        const x = data.colNames.indexOf(this.dims.x);

  86        if (data.rows.length>1) { // artificially shorten the x-axis by 1 unit

  87            const xUnit = data.rows[1][x] - data.rows[0][x];

  88            const domain = this.cfg.scales.hor.domain();

  89            if (domain[1] - domain[0] > xUnit) { domain[0] += xUnit; }

  90            this.cfg.scales.hor.domain(domain);         

  91        }   

  92    }

  94    d3RenderMarkers(svg:d3Base, data:NumericDataSet) {

  95        if (data.rows.length<2) { return super.d3RenderMarkers(svg, data); }

  96        const defaults = (this.cfg.defaults.series[this.key]).marker;

  97        if (typeof(this.dims.x)==='number') { log.warn(`d3RenderMarkers: unsupported const x=${this.dims.x} in timeseries`); }

  98        const x = data.colNames.indexOf(this.dims.x);

  99        if (defaults.rendered) {

 100            const xUnit = data.rows[1][x] - data.rows[0][x];

 101            const samples:any = svg.select('.markers').selectAll("circle").data(data.rows, d => d[0]);

 102            samples.exit().remove();            // remove unneeded circles

 103            samples.enter().append('circle')    // add new circles

 104                .call(this.d3DrawMarker, this, data.colNames)

 105                .attr("transform", `translate(${this.cfg.scales.hor(xUnit) - this.cfg.scales.hor(0)})`)

 106            .merge(samples).transition(this.cfg.transition)   // draw markers

 107                .call(this.d3DrawMarker, this, data.colNames)

 108                .attr("transform", `translate(0)`);

 109        }

 110    }
