   1/**

   2 * # Time Series Plot

   3 * 

   4 * plots a 2D line with markers and smooth x-axis tranisitions .

   5 * 

   6 * ## Usage

   7 * Invoke a `timeseries` by adding a new series to the graph

   8 * `graph.addSeries('timeseries', , , []);`

   9 * 

  10 * 

  11 * ## Example

  12 * 'hsGraphD3'}>

  13 * 'script.js'>

  14 * // create data set:

  15 * const data = {

  16 *    colNames:['date', 'time', 'volume'], 

  17 *    rows:[], 

  18 * };

  19 * for (let i=0; i<11; i++) { data.rows.push([i, Math.random(), Math.random()+1])}

  20 * 

  21 * // create the graph and define the series to plot:

  22 * const graph = new hsGraphD3.GraphCartesian(root);

  23 * graph.addSeries('timeseries', 'date', 'time');

  24 * graph.addSeries('timeseries', 'date', 'volume');

  25 * 

  26 * // adjust some settings:

  27 * graph.defaults.scales.hor.aggregateOverTime = false;  // forget early indexes

  28 * graph.defaults.series.series0.line.color = '#00a';  // first line blue

  29 * graph.defaults.series.series1.line.color = '#0a0';  // first line blue

  30 * 

  31 * // trigger the update loop to plot the data

  32 * let index = 11;

  33 * update();

  34 * 

  35 * function update() {

  36 *      index++;

  37 *      // modify the data in this round:

  38 *      data.rows.push([index, Math.random(), Math.random()+1]);

  39 * 

  40 *      // render the graph: 

  41 *      graph.render(data);

  42 *      if (data.rows.length > 10) { data.rows.shift(); }

  43 * 

  44 *      // trigger next update in 1.5s:

  45 *      setTimeout(update, 1050);

  46 * }

  47 * 

  48 * 

  49 */

  82     d3RenderMarkers(svg:d3Base, data:DataSet) {

  83        const defaults = this.cfg.defaults.series[this.key].marker;

  84        const rScale = this.cfg.scales[this.dims[2]];

  85

  86        const samples:any = svg.select('.markers').selectAll("circle").data(data.rows, d => d[0]);

  87        samples.exit().remove();            // remove unneeded circles

  88        samples.enter().append('circle')    // add new circles

  89            .call(this.d3DrawMarker, this)

  90            .attr("transform", `translate(${this.cfg.scales.hor(1) - this.cfg.scales.hor(0)})`)

  91        .merge(samples).transition(this.cfg.transition)   // draw markers

  92            .call(this.d3DrawMarker, this)

  93            .attr("transform", `translate(0)`);

  94    }

  96    d3RenderPath(svg:d3Base, data:DataSet) {

  97        const path = this.svg.select('.lines').selectAll('path'); //.data([data.rows]);

  98        path.attr('d', d => this.lines(data.rows))

  99            .attr("transform", `translate(${this.cfg.scales.hor(1) - this.cfg.scales.hor(0)})`)

 100         .transition(this.cfg.transition)

 101            .attr("transform", `translate(0)`)

 102        ;

 103    }
