  84export class Column extends OrdinalSeriesPlot {

  85    preRender(data:DataSet, domains:Domains): void {

  86        super.preRender(data, domains);

  87        const def = (this.cfg.defaults.series[this.key]);

  88        this.cfg.scales.hor.padding(def.gap);

  89    }

  91    protected d3DrawBar(markers:d3Base, plot:Column, colNames:string[]) {

  92        const xAccess  = accessor(plot.dims.x, colNames, plot.cfg.scales.hor);

  93        const yAccess  = accessor(plot.dims.y, colNames, plot.cfg.scales.ver);

  94        const y0 = accessor(()=>0, colNames, plot.cfg.scales.ver)([]);

  95        const step = plot.cfg.scales.hor.step();

  96        const pad = this.cfg.scales.hor.padding();

  97        const width = step * (1 - pad);

  98        markers

  99            .attr("x",  (d:number[]) => xAccess(d) + step*pad/2)

 100            .attr("y",  (d:number[]) => Math.min(yAccess(d), y0))

 101            .attr("width",  () => width)

 102            .attr("height", (d:number[]) => Math.abs(y0-yAccess(d)));

 103    }
