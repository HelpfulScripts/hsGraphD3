  99 export interface PolarDimensions extends GraphDimensions { 

 100     ang:ValueDef[]; 

 101     rad:ValueDef[]; 

 102 }

 103 

 104 export class GraphPolar extends Graph {

 105     /**

 106      * Called during `Graph` construction to create component defaults and 

 107      * scale defaults for the `GraphDimensions` used in cartedian plots.

 108      */

 109     protected makeDefaults() {

 110         super.makeDefaults();

 111         const scalesDefaults = this.defaults.scales;

 112         scalesDefaults.dims.ang  = scalesDefaults.dims.ang  || scaleDefault('linear');    // auto viewport range

 113         scalesDefaults.dims.rad  = scalesDefaults.dims.rad  || scaleDefault('linear');    // auto viewport range

 114     }

 115

 116    /** 

 117     * disables components for polar plots

 118     */

 119    protected createComponents(cfg:GraphCfg) {

 120        super.createComponents(cfg);

 121        this.components.axes    = undefined;

 122        this.components.grids   = undefined;

 123    }

 124

 125    /**

 126     * set scales, called during `prerender`

 127     * @param data 

 128     */

 129    protected setScales() {

 130        const margins = this.scales.defaults.margin;

 131        this.scales.createScale('ang', this.cumulativeDomains.ang, [0,  2*Math.PI]);

 132        this.scales.createScale('rad', this.cumulativeDomains.rad, [0, Math.min(this.viewport.height-margins.bottom-margins.top, this.viewport.width-margins.left-margins.right)/2]);

 133    }

 134

 135    /**

 136     * returns a `Domains` object 

 137     * @param scalesDefaults 

 138     */

 139    protected prepareDomains(scalesDefaults:ScalesDefaults):Domains {

 140        const dom = this.cumulativeDomains;

 141        const dims = scalesDefaults.dims;

 142        dom.ang  = (dom.ang  && dims.ang.aggregateOverTime)?  dom.ang  : undefined;

 143        dom.rad  = (dom.rad  && dims.rad.aggregateOverTime)?  dom.rad  : undefined;

 144        return dom;

 145    }

 146

 147    protected updateBaseSVG(cfg: GraphCfg) {

 148        const hor = cfg.viewPort.width;

 149        const ver = cfg.viewPort.height;

 150        cfg.viewPort.orgX = -hor/2;

 151        cfg.viewPort.orgY = -ver/2;

 152        cfg.baseSVG.attr('viewBox', `${-hor/2} ${-ver/2} ${hor} ${ver}`);

 153    }
