  69/**

  70 * creates the base SVG element for the entire graph.

  71 * This function is intended to be called only once at creation of the graph.

  72 * @param cfg 

  73 */

  74function createBaseSVG(cfg: GraphCfg):d3Base {

  75    const base = d3.select(cfg.root);

  76    base.selectAll('div').remove();

  77    base.selectAll('svg').remove();

  78    const svg = base.append('svg')

  79        .classed('baseSVG', true)

  80        .attr('height', '100%')

  81        .attr('width', '100%')

  82        .attr('preserveAspectRatio', 'xMinYMin meet')

  83        ;

  84    cfg.baseSVG = svg;

  85    svg.append('rect')

  86        .classed('baseRect', true)

  87        .attr('x', 0)

  88        .attr('y', 0);

  89    return svg;

  90}

  91

  92/**

  93 * regualrly sets the graph's viewBox, typically after a window resize.

  94 * @param cfg 

  95 */

  96function updateBaseSVG(cfg: GraphCfg) {

  97    cfg.baseSVG.attr('viewBox', `0 0 ${cfg.viewPort.width} ${cfg.viewPort.height}`);

  98}

 101export class Graph extends GraphComponent {

 102    private plot:Plot;

 103    private axes:Axis[] = [];

 104    private cumulativeDomains: {[colName:string]: [number, number]} = {};

 105

 106    constructor(root:any) { 

 107        super();

 108        this.config.root = root;

 109        log.info('creating Graph');

 110        const base = createBaseSVG(this.config); 

 111        updateBaseSVG(this.config);

 112        this.plot = new Plot(this.config);

 113        this.axes.push(new Axis(this.config, Direction.Horizontal));

 114        this.axes.push(new Axis(this.config, Direction.Vertical));

 115        window.onresize = () => this.resize();

 116    }

 117

 118    public get defaults(): Defaults {

 119        return this.config.defaults;

 120    }

 121

 122    resize() {

 123        const cfg = this.config;

 124        if (cfg.root.clientWidth > 0) {

 125            if (cfg.root.clientWidth !== cfg.client.width || cfg.root.clientHeight !== cfg.client.height) {

 126                log.info(`resizing svg: [${cfg.client.width} x ${cfg.client.height}] -> [${cfg.root.clientWidth} x ${cfg.root.clientHeight}]`);

 127                cfg.client.width = cfg.root.clientWidth;

 128                cfg.client.height = cfg.root.clientHeight;

 129                cfg.viewPort.height = cfg.viewPort.width * cfg.root.clientHeight / cfg.root.clientWidth;

 130                updateBaseSVG(cfg);

 131            }

 132        }

 133    }

 135    /**

 136     * renders all Graph elements using `data`

 137     * @param data 

 138     */

 139    public render(data:Data) {

 140        this.setScales(data);

 141        this.drawCanvas(this.config);

 142        this.plot.setBorders(10, 10, 10, 10);

 143        this.plot.render(data);

 144        this.axes.forEach(a => a.render(data));

 145    }

 148     * adds a series to the plot.

 149     * @param type type of plot to use, e.g. 'bubble' or 'scatter'

 150     * @param params the column name of the parameters used to plot the series

 151     */

 152    public addSeries(type:string, x:string, y:string, ...params:string[]) {

 153        this.resize();

 154        this.config.scales.hor.dataCol = x;

 155        this.config.scales.ver.dataCol = y;

 156        this.plot.addSeries(type, x, y, ...params);

 157    }

 159    /**

 160     * renders the Graph's background canvas

 161     * @param cfg 

 162     */

 163    private drawCanvas(cfg: GraphCfg) {

 164        const canvas = cfg.defaults.Graph.canvas;

 165        d3.select('.baseRect')

 166        .attr('width', cfg.viewPort.width)

 167        .attr('height', cfg.viewPort.height)

 168        .attr('rx', canvas.rx)

 169        .attr('ry', canvas.ry)

 170        .attr('stroke', canvas.stroke.color)

 171        .attr('stroke-width', canvas.stroke.width)

 172        .attr('stroke-opacity', canvas.stroke.opacity)

 173        .attr('fill', canvas.fill.color)

 174        .attr('fill-opacity', canvas.fill.opacity);

 175    }

 177    private setScales(data:Data) {

 178        /** expands `domains[colName]` to include the range of the current `data` domain. */

 179        function expandDomain(domains:{[colName:string]:NumDomain}, colName:string) {

 180            domains[colName] = domains[colName] || [1e90, -1e90];

 181            const dataDom:NumDomain = data.findDomain(colName);

 182            domains[colName][0] = Math.min(domains[colName][0], dataDom[0]);

 183            domains[colName][1] = Math.max(domains[colName][1], dataDom[1]);

 184            return domains[colName];

 185        }

 186        const hor = this.config.scales.hor;

 187        const ver = this.config.scales.ver;

 188        hor.scale = d3.scaleLinear()

 189            .domain(expandDomain(this.cumulativeDomains, hor.dataCol))

 190            .range([margin, this.config.viewPort.width-2*margin]);

 191        ver.scale = d3.scaleLinear()

 192            .domain(expandDomain(this.cumulativeDomains, ver.dataCol))

 193            .range([this.config.viewPort.height-2*margin, margin]);

 194    }
