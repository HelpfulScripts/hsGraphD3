  69/**

  71 * @param area  the fill color

  72 * @param borderWidth the border width in pixel

  76    return {

  77        rx: 0,

  78        ry: 0,

  79        fill: {

  80            color: area,

  81            opacity: 1

  82        },

  83        stroke: {

  84            width: borderWidth,

  85            color: borderColor,

  86            opacity: 1

  87        }

  88    };

  89}
