  90export class Graph2DCartesian extends AbstractGraph {

  91    private cumulativeDomains: {[colName:string]: [number, number]} = {};

  92

  93    constructor(root:any) { 

  94        super(root);

  95        const scales = this.config.defaults('scales');

  96        const margins = scales.margin;

  97        this.config.scales.hor = { dataCol: undefined, scale: undefined};

  98        this.config.scales.ver = { dataCol: undefined, scale: undefined};

  99        scales.dims['hor'] = defaultDimScale(margins.left, this.viewport.width-margins.right);

 100        scales.dims['ver'] = defaultDimScale(margins.top, this.viewport.height-margins.bottom);

 101        log.info('creating Graph');

 102    }

 103

 104    /**

 105     * adds a series to the plot.

 106     * @param type type of plot to use, e.g. 'bubble' or 'scatter'

 107     * @param params the column name of the parameters used to plot the series

 108     */

 109    public addSeries(type:string, x:string, y:string, ...params:string[]) {

 110        super.addSeries(type, x, y, ...params);

 111    }

 113    protected setScales(data:Data) {

 114        /** expands `domains[colName]` to include the range of the current `data` domain. */

 115        function expandDomain(domains:{[colName:string]:NumDomain}, colName:string) {

 116            domains[colName] = domains[colName] || [1e90, -1e90];

 117            const dataDom:NumDomain = data.findDomain(colName);

 118            domains[colName][0] = Math.min(domains[colName][0], dataDom[0]);

 119            domains[colName][1] = Math.max(domains[colName][1], dataDom[1]);

 120            return domains[colName];

 121        }

 122        const scales = (this.config.defaults('scales')).dims;

 123        const horCol = this.config.scales.hor.dataCol;

 124        const verCol = this.config.scales.ver.dataCol;

 125        const hor = this.config.scales.hor;

 126        const ver = this.config.scales.ver;

 127        if (scales[horCol].type === 'linear') {

 128            hor.scale = d3.scaleLinear()

 129            .domain(expandDomain(this.cumulativeDomains, hor.dataCol))

 130            .range([scales[horCol].range.min, scales[horCol].range.max]);

 131        }

 132        if (scales[verCol].type === 'linear') {

 133            ver.scale = d3.scaleLinear()

 134            .domain(expandDomain(this.cumulativeDomains, ver.dataCol))

 135            .range([scales[verCol].range.min, scales[verCol].range.max]);

 136        }

 137    }
