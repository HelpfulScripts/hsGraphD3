  36    /*------------ Static implementation----- */

  37    /** a map of plot types to corresponding plot functions. New plot types are added via a call to `register`. */

  38    protected static plotFnMap: {[plotKey:string]: PlotFactory} = {};

  39

  40    /** registers a new plot type with corresponding function. */

  41    public static register(key:string, factory:PlotFactory) {

  42        this.plotFnMap[key] = factory;

  43        log.info(`registered plot type '${key}'`);

  44    }

  46    /*------------ Instance implementation----- */

  47    private desc: PlotCfg;

  50    constructor(cfg:GraphCfg) {

  52        this.desc = {

  60    /**

  61     * adds a series to the plot.

  62     * @param type type of plot to use, e.g. 'bubble' or 'scatter'

  63     * @param params the column name of the parameters used to plot the series

  64     */

  65    addSeries(type:string, ...params:string[]) {

  66        const factory = Plot.plotFnMap[type];

  67

  68        // this.series[seriesKey] = [type].concat(params);

  69        if (factory) {

  70            const svg = this.desc.plotBase; 

  71            svg.append('g').classed(`series${this.series.length}`,true);

  72            const seriesKey = `${type} ${params.join(' ')}`;

  73            const plot = factory.newPlot(this.desc, ...params);

  74            this.series.push(plot);

  75            log.info(`added series ${this.series.length} as '${seriesKey}'`);

  76        } else {

  77            log.error(`unknown plot type ${type}; available types are:\n   '${Object.keys(Plot.plotFnMap).join("'\n   '")}'`);

  78        }

  79    }

  81    /**

  82     * renders the plot area and all added series using `data`

  85    render(data:Data) {

  86//        this.renderPlotArea();

  87        const svg = this.desc.cfg.baseSVG; 

  88        this.series.forEach((s:PlotClass, i:number) => {

  89            s.render(data, svg.selectAll(`.series${i}`));

  90        });

  91    }
