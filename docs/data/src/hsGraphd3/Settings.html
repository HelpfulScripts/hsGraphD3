 104export function setLine(svg:d3Base, settings:d.Line) {

 105    svg 

 106    .attr('stroke',         settings.color)

 124/**

 125 * convenience function to create a default `RectStyle` object with configurable fill color and border. 

 126 * @param area  the fill color

 127 * @param borderWidth the border width in pixel

 128 * @param borderColor the border color

 129 */

 130export const defaultRect = (areaFill:d.Color, borderWidth:UnitVp=0, borderColor:d.Color='currentColor'):d.RectStyle => {

 131    return {

 132        rx: 0,

 133        ry: 0,

 134        fill:   defaultFill(areaFill),

 135        stroke: defaultLine(borderWidth, borderColor)

 136    };

 137};

 138

 139export function setRect(svg:d3Base, settings:d.RectStyle) {

 140    svg .attr('rx', settings.rx)

 141        .attr('ry', settings.ry);

 142    setLine(svg, settings.stroke);

 143    setFill(svg, settings.fill); 

 144}
