  53                tickMarks:  def.defaultLine(2),

  54                tickLabel:  def.defaultText()

  55            },

  56            ver: {

  57                color:      'currentColor',

  58                line:       def.defaultLine(2),

  59                tickMarks:  def.defaultLine(2),

  60                tickLabel:  def.defaultText()

  71export class Axis extends GraphComponent {

  72    private dir: Direction;

  73

  74    constructor(cfg:GraphCfg, dir:Direction) {

  75        super(cfg, cfg.baseSVG.select('.axes').append('g'));

  76        this.dir = dir;

  77        this.svg.classed(`${dir}Axis`, true);

  78    }

  79

  80    public get componentType() { return 'axis'; }

  81

  82    public createDefaults() {}

  83    

  84    public renderComponent() {

  85        const scales = this.cfg.scales;

  86        const style = this.cfg.defaults('axes')[this.dir];

  87        let axis;

  88        this.cfg.baseSVG.select('.axes')

  89            .attr('color', (this.cfg.defaults('axes')).color);

  90        this.svg

  91            .attr('stroke', style.line.color)

  92            .attr('stroke-width', style.line.width)

  93            .attr('stroke-opacity', style.line.opacity);

  94

  95        if (this.dir===Direction.horizontal) {

  96            axis = d3.axisTop(scales.hor.scale);

  97            const yCrossing = Math.max(axisWidth, Math.min(scales.ver.scale(0), this.cfg.viewPort.height-axisWidth));

  98            this.svg.attr("transform", `translate(0, ${yCrossing})`);

  99        } else {

 100            axis = d3.axisRight(scales.ver.scale);

 101            const xCrossing = Math.max(axisWidth, Math.min(scales.hor.scale(0), this.cfg.viewPort.width-axisWidth));

 102            this.svg.attr("transform", `translate(${xCrossing}, 0)`);

 103        }

 104        axis.tickSize(tickWidth);

 105        this.svg

 106            .attr('color', style.color)

 107            .style('font-family',  style.tickLabel.font.family)

 108            .style('font-size',   `${style.tickLabel.font.size}px`)

 109            .style('font-style',  style.tickLabel.font.style)

 110            .style('font-weight', style.tickLabel.font.weight);

 111        this.svg.call(axis);

 112        this.svg.selectAll('text')

 113            .transition().duration(1000)

 114            .attr('stroke', style.tickLabel.color || 'currentColor')

 115            .attr('stroke-width', 1);

 116        }
