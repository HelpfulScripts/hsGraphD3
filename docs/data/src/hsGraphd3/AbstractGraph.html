  23const vpWidth:number    = 1000;

  26export abstract class AbstractGraph {

  27    /** the HTML root element to attach the render tree to. */

  28    protected root:any;

  29

  30    /** the `GraphCfg` object shared by all components in this graph */

  31    protected config: GraphCfg;

  32

  33    /** the plot component, provides access to individual series components */

  34    protected plot:Series;

  35

  36    /** the list of components to render */

  37    private components: GraphComponent[] = [];

  38

  39    constructor(root:any) { 

  40        this.root = root;

  41        this.config = this.initializeCfg();

  42        this.config.baseSVG = this.createBaseSVG(this.config); 

  43        this.updateBaseSVG(this.config);

  44        this.components = this.createComponents();

  45        this.components.map(c => this.config.defaults[c.componentType] = c.createDefaults());

  46        window.onresize = () => this.resize();

  47    }

  48

  49    public get defaults(): ComponentDefaults {

  50        return this.config.defaults;

  51        // return this.config.defaults();

  52    }

  53

  54    public get viewport() {

  55        return this.config.viewPort;

  56    }

  57

  58    /**

  59     * render the tree with the supplied data.

  60     * @param data the data to render

  61     */

  62    public render(data:Data|DataSet|DataTable): void {

  63        let d:Data = (data instanceof Data)? data : new Data(data);

  64        this.setScales(d);

  65        this.components.forEach((comp:GraphComponent) => comp.renderComponent(d));

  66    }

  67

  68    /**

  69     * adds a series to the plot.

  70     * @param type type of plot to use, e.g. 'bubble' or 'scatter'

  71     * @param params the column name of the parameters used to plot the series

  72     */

  73    public addSeries(type:string, x:string, y:string, ...params:string[]) {

  74        this.resize();

  75        this.plot.addSeries(type, x, y, ...params);

  76    }

  77

  78

  79    //************** Non-public part **************************/

  80

  81    /** set the scales for the graph prior to rendering components. */

  82    protected abstract setScales(data:Data):void;

  83

  84    private initializeCfg():GraphCfg {

  85        return {

  86            baseSVG:  undefined,    

  87            client:   { x:0, y:0, width: 0, height: 0 },

  88            viewPort: {

  89                width: vpWidth,

  90                height: vpWidth * 0.7   // initial height: 70% of width

  91            },

  92            defaults: {},

  93            scales: {} 

  94        };

  95    }

  96

  97    private createComponents():GraphComponent[] {

  98        return [

  99            new Scales(),

 100            new Canvas(this.config),

 101            new Grids(this.config),

 102            this.plot = new Series(this.config),

 103            new Axes(this.config)

 104        ];

 105    }

 106

 107    /** callback on window resize event;  */

 108    private resize() {

 109        const cfg = this.config;

 110        if (this.root && this.root.clientWidth > 0) {

 111            if (this.root.clientWidth !== cfg.client.width || this.root.clientHeight !== cfg.client.height) {

 112                log.info(`resizing svg: [${cfg.client.width} x ${cfg.client.height}] -> [${this.root.clientWidth} x ${this.root.clientHeight}]`);

 113                cfg.client.width = this.root.clientWidth;

 114                cfg.client.height = this.root.clientHeight;

 115                cfg.viewPort.height = cfg.viewPort.width * this.root.clientHeight / this.root.clientWidth;

 116                this.updateBaseSVG(cfg);

 117            }

 118        }

 119    }

 120

 121    /**

 122     * creates the base SVG element for the entire graph.

 123     * This function is intended to be called only once at creation of the graph.

 124     * @param cfg 

 125     */

 126    private createBaseSVG(cfg: GraphCfg):d3Base {

 127        const svg = d3.select(this.root).append('svg')

 128            .classed('baseSVG', true)

 129            .attr('height', '100%')

 130            .attr('width', '100%')

 131            .attr('preserveAspectRatio', 'xMinYMin meet')

 132            ;

 133        // determine order of rendering:

 134        svg.append('rect').classed('graphArea', true).attr('x', 0).attr('y', 0);

 135        svg.append('g').classed('grids', true);

 136        svg.append('g').classed('axes', true);

 137        svg.append('g').classed('series', true);

 138        svg.append('rect').classed('graphBorder', true).attr('x', 0).attr('y', 0);

 139        return svg;

 140    }

 141

 142    /**

 143     * regualrly sets the graph's viewBox, typically after a window resize.

 144     * @param cfg 

 145     */

 146    private updateBaseSVG(cfg: GraphCfg) {

 147        cfg.baseSVG.attr('viewBox', `0 0 ${cfg.viewPort.width} ${cfg.viewPort.height}`);

 148    }
