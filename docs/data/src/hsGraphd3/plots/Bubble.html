  16export class BubbleFatory extends PlotFactory {

  17    newPlot(desc:PlotCfg, cx:string, cy:string, r?:string): PlotClass {

  18        return new Bubble(desc, cx, cy, r);

  22class Bubble extends PlotClass {

  23    /**

  24     * plot constructor

  25     * @param cx string column name for x-center coordinates

  26     * @param cy string column name for y-center coordinates

  27     * @param r  string column name for radius coordinates

  28     */

  29    constructor(protected desc: PlotCfg, protected cx:string, protected cy:string, protected r?:string) {

  30        super(desc, cx, cy);

  31        desc.cfg.defaults.scales[r] = desc.cfg.defaults.scales[r] || desc.cfg.defaults.defaultScale();

  32    }

  33

  34    /**

  35     * 

  36     * @param data a {@link hsDatab:Data `Data`} object containing the 

  37     */

  38    render(data:Data, series:d3Selection) {  

  39        const ix = data.colNumber(this.cx);

  40        const iy = data.colNumber(this.cy);

  41        const ir = data.colNumber(this.r);

  42        const scaleX = this.desc.cfg.scales.hor.scale;

  43        const scaleY = this.desc.cfg.scales.ver.scale;

  44        const defR = this.desc.cfg.defaults.scales[this.r];

  45        const scaleR = d3.scaleLinear().domain(data.findDomain(this.r)).range([defR.range.min, defR.range.max]);

  46        const circles = series.selectAll("circle").data(data.getData());

  47            

  48        circles.exit().remove();            // remove unneeded circles

  49        circles.enter().append('circle');   // add new circles

  50        

  51        circles.transition().duration(1000)

  52            .attr("cx", d => scaleX(d[ix]))

  53            .attr("cy", d => scaleY(d[iy]))

  54            .attr("r",  d => scaleR(ir===undefined? DEF_RADIUS : d[ir]))

  55            .attr('fill', (d,i) => ['#f00', '#0f0', '#00f', '#ff0', '#f0f', '#0ff'][i])

  56            ;        

  57    }

  58}
