  50    renderComponent(data:Data) {  

  51        const ix = data.colNumber(this.cx);

  59        const defaults = this.cfg.defaults.series[this.key];

  60            

  61        circles.exit().remove();            // remove unneeded circles

  63        

  64        circles.transition(def.d3Transition)

  65            .attr("cx", (d:number[]) => scaleX(d[ix]))

  66            .attr("cy", (d:number[]) => scaleY(d[iy]))

  67            .attr("r",  (d:number[]) => scaleR(ir===undefined? defaults.marker.size : d[ir]))
